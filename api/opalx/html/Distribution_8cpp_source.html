187size_t Distribution::getNumOfLocalParticlesToCreate(size_t n) {
188 size_t locNumber = n / ippl::Comm->size();
189
190 // make sure the total number is exact
191 size_t remainder = n % ippl::Comm->size();
192 size_t myNode = ippl::Comm->rank();
193 if (myNode < remainder)
194 ++locNumber;
195
196 return locNumber;
197}

300void Distribution::setDistParametersMultiVariateGauss() {
301
302 cutoffR_m = 3.;
303 cutoffP_m = 3.;
304
305 // initialize the covariance matrix to identity
306 for (unsigned int i = 0; i < 6; ++ i) {
307 for (unsigned int j = 0; j < 6; ++ j) {
308 if (i==j)
309 correlationMatrix_m[i][j] = 1.0;
310 else
311 correlationMatrix_m[i][j] = 0.0;
312 }
313 }
314
315 // set diagonal elements first
316 setSigmaR_m();
317 setSigmaP_m();
318
319 for (unsigned int i = 0; i < 3; ++ i){
320 correlationMatrix_m[2*i ][2*i ] = sigmaR_m[i]*sigmaR_m[i];
321 correlationMatrix_m[2*i+1][2*i+1] = sigmaP_m[i]*sigmaP_m[i];
322 }
323
324 std::vector<double> cr = Attributes::getRealArray(itsAttr[DISTRIBUTION::CORR]);
325
326 if (!cr.empty()) {
327 // read off-diagonal correlation matrix from input file
328 if (cr.size() == 15) {
329 *gmsg << "* Use r to specify correlations" << endl;
330 unsigned int k = 0;
331 for (unsigned int i = 0; i < 5; ++ i) {
332 for (unsigned int j = i + 1; j < 6; ++ j, ++ k) {
333 correlationMatrix_m[j][i] = cr.at(k)*cr.at(k);
334 correlationMatrix_m[i][j] = correlationMatrix_m[j][i]; // impose symmetry
335 }
336 }
337 }
338 else {
339 throw OpalException("Distribution::SetDistParametersGauss",
340 "Inconsistent set of correlations specified, check manual");
341 }
342 }
343
344 avrgpz_m = 0.0;
345}

347void Distribution::setDistParametersFlatTop() {
348 // avrgpz_m is not reset here; TrackRun sets it from BEAM (getMomentum()/getMass() = beta*gamma)
349 // so that initial and emitted particles get the correct reference momentum.
350
351 cutoffR_m = 3.;
352 cutoffP_m = 3.;
353
354 // set diagonal elements first
355 setSigmaR_m();
356 setSigmaP_m();
357
358 // initialize the covariance matrix to identity
359 for (unsigned int i = 0; i < 6; ++ i) {
360 for (unsigned int j = 0; j < 6; ++ j) {
361 if (i==j)
362 correlationMatrix_m[i][j] = 1.0;
363 else
364 correlationMatrix_m[i][j] = 0.0;
365 }
366 }
367
368 cutoffR_m = ippl::Vector<double, 3>(
369 std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::CUTOFFX])),
370 std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::CUTOFFY])),
371 std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::CUTOFFLONG])));
372
373 correlationMatrix_m[1][0] = Attributes::getReal(itsAttr[DISTRIBUTION::CORRX]);
374 correlationMatrix_m[3][2] = Attributes::getReal(itsAttr[DISTRIBUTION::CORRY]);
375 correlationMatrix_m[5][4] = Attributes::getReal(itsAttr[DISTRIBUTION::CORRT]);
376
377 if (Attributes::getReal(itsAttr[DISTRIBUTION::CORRZ]) != 0.0)
378 correlationMatrix_m[5][4] = Attributes::getReal(itsAttr[DISTRIBUTION::CORRZ]);
379
380 emitting_m = Attributes::getBool(itsAttr[DISTRIBUTION::EMITTED]);
381
382 if (emitting_m) {
383 sigmaR_m[2] = 0.0;
384
385 sigmaTRise_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::SIGMAT]));
386 sigmaTFall_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::SIGMAT]));
387 tPulseLengthFWHM_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TPULSEFWHM]));
388
389 FTOSCAmplitude_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::FTOSCAMPLITUDE]));
390 FTOSCPeriods_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::FTOSCPERIODS]));
391
392 // If TRISE and TFALL are defined > 0.0 then these attributes
393 // override SIGMAT.
394 //
395 if (std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TRISE])) > 0.0
396 || std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TFALL])) > 0.0) {
397
398 double timeRatio = std::sqrt(2.0 * std::log(10.0)) - std::sqrt(2.0 * std::log(10.0 / 9.0));
399
400 if (std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TRISE])) > 0.0)
401 sigmaTRise_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TRISE]))
402 / timeRatio;
403
404 if (std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TFALL])) > 0.0)
405 sigmaTFall_m = std::abs(Attributes::getReal(itsAttr[DISTRIBUTION::TFALL]))
406 / timeRatio;
407 }
408
409 // For an emitted beam, the longitudinal cutoff >= 0.
410 cutoffR_m[2] = std::abs(cutoffR_m[2]);
411
412 }
