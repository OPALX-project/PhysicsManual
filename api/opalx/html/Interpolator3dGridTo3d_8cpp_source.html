33 Interpolator3dGridTo3d::Interpolator3dGridTo3d(const Interpolator3dGridTo3d& rhs) {
34 coordinates_m = new ThreeDGrid(*rhs.coordinates_m);

39 void Interpolator3dGridTo3d::function(const double Point[3], double Value[3]) const {
40 if (Point[0] > coordinates_m->maxX() || Point[0] < coordinates_m->minX()
41 || Point[1] > coordinates_m->maxY() || Point[1] < coordinates_m->minY()
42 || Point[2] > coordinates_m->maxZ() || Point[2] < coordinates_m->minZ()) {
43 Value[0] = 0;
44 Value[1] = 0;
45 Value[2] = 0;
46 return;
47 }
48
49 interpolator_m[0]->function(Point, &Value[0]);
50 interpolator_m[1]->function(Point, &Value[1]);
51 interpolator_m[2]->function(Point, &Value[2]);
52 }

54 void Interpolator3dGridTo3d::setAll(
55 ThreeDGrid* grid, double*** Bx, double*** By, double*** Bz,
56 interpolationAlgorithm algo) {
57 if (coordinates_m != nullptr) coordinates_m->remove(this);
58 grid->add(this);
59 coordinates_m = grid;
60
61 for (int i = 0; i < 3; i++)
62 if (interpolator_m[i] != nullptr) delete interpolator_m[i];
63
64 switch (algo) {
65 case TRILINEAR:
66 interpolator_m[0] = new TriLinearInterpolator(grid, Bx);
67 interpolator_m[1] = new TriLinearInterpolator(grid, By);
68 interpolator_m[2] = new TriLinearInterpolator(grid, Bz);
69 break;
70 default:
71 throw(LogicalError(
72 "Interpolator3dGridTo3d::setAll",
73 "Did not recognise interpolation algorithm"));
74 }
75 }
