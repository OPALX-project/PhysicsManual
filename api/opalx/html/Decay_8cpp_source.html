29size_t Decay::apply(
30 ParticleContainer<double, 3>& pc, double dt, [[maybe_unused]] long long globalTrackStep,
31 [[maybe_unused]] size_t containerIdx) {
32 using pc_size_type = ippl::detail::size_type;
33
34 if (dt <= 0.0 || !(tau0_m > 0.0) || !std::isfinite(tau0_m)) {
35 return 0;
36 }
37
38 const pc_size_type nLocal = pc.getLocalNum();
39 if (nLocal == 0 && !daughterPC_m) {
40 return 0;
41 }
42
43 /* Phase 1: Mark decayed particles using relativistic decay probability. */
44 Kokkos::View<bool*> invalid("Decay::invalid", nLocal);
45 const pc_size_type localDestroyNum = markDecayedParticles(nLocal, dt, pc.P.getView(), invalid);
46
47 pc_size_type globalDestroyNum = 0;
48 ippl::Comm->allreduce(localDestroyNum, globalDestroyNum, 1, std::plus<pc_size_type>());
49 if (globalDestroyNum == 0) {
50 return 0;
51 }
52
53 if (daughterPC_m) {
54 /* Phase 2: Gather kinematics of decayed parents into compact views. */
55 const DecayedParentViews parents = collectDecayedParents(
56 nLocal, localDestroyNum, invalid, pc.R.getView(), pc.P.getView(), pc.dt.getView());
57
58 /* Phase 3: Create daughters — subclass-specific momentum sampling. */
59 const pc_size_type oldDaughterLocal = daughterPC_m->getLocalNum();
60 daughterPC_m->create(localDestroyNum);
61
62 if (localDestroyNum > 0) {
63 createDaughterParticles(
64 localDestroyNum, oldDaughterLocal, parents.R, parents.P, parents.dt);
65 }
66 }
67
68 /* Phase 4: Destroy decayed parent particles. */
69 pc.destroy(invalid, localDestroyNum);
70 return static_cast<size_t>(globalDestroyNum);
71}

73ippl::detail::size_type Decay::markDecayedParticles(
74 ippl::detail::size_type nLocal, double dt, Kokkos::View<ippl::Vector<double, 3>*> Pview,
75 Kokkos::View<bool*> invalid) {
76 using pc_size_type = ippl::detail::size_type;
77
78 // Local copies — KOKKOS_LAMBDA captures these, not `this`.
79 const double tau0 = tau0_m;
80 auto pool = randPool_m;
81
82 pc_size_type localDestroyNum = 0;
83 Kokkos::parallel_reduce(
84 "Decay::mark", nLocal,
85 KOKKOS_LAMBDA(const pc_size_type i, pc_size_type& count) {
86 auto generator = pool.get_state();
87 const double p2 = Pview(i)[0] * Pview(i)[0] + Pview(i)[1] * Pview(i)[1]
88 + Pview(i)[2] * Pview(i)[2];
89 const double gamma = Kokkos::sqrt(1.0 + p2);
90 const double tauLab = gamma * tau0;
91 const double pDecay = 1.0 - Kokkos::exp(-dt / tauLab);
92 const bool decayed = generator.drand(0.0, 1.0) < pDecay;
93 invalid(i) = decayed;
94 count += decayed;
95 pool.free_state(generator);
96 },
97 localDestroyNum);
98 Kokkos::fence();
99 return localDestroyNum;
100}

102Decay::DecayedParentViews Decay::collectDecayedParents(
103 ippl::detail::size_type nLocal, ippl::detail::size_type localDestroyNum,
104 Kokkos::View<bool*> invalid, Kokkos::View<ippl::Vector<double, 3>*> Rview,

107
108 Kokkos::View<pc_size_type*> compactIdx("Decay::compactIdx", nLocal);
109 Kokkos::parallel_scan(
110 "Decay::compact", nLocal,
111 KOKKOS_LAMBDA(const pc_size_type i, pc_size_type& runningTotal, const bool isFinal) {
112 if (invalid(i)) {
113 if (isFinal) {
114 compactIdx(i) = runningTotal;
115 }
116 ++runningTotal;
117 }
118 });
119 Kokkos::fence();
120
121 DecayedParentViews out{
122 Kokkos::View<ippl::Vector<double, 3>*>("Decay::parentR", localDestroyNum),
123 Kokkos::View<ippl::Vector<double, 3>*>("Decay::parentP", localDestroyNum),
124 Kokkos::View<double*>("Decay::parentDt", localDestroyNum)};
125
126 Kokkos::parallel_for(
127 "Decay::collectParents", nLocal, KOKKOS_LAMBDA(const pc_size_type i) {
128 if (invalid(i)) {
129 const pc_size_type j = compactIdx(i);
130 out.R(j) = Rview(i);
131 out.P(j) = Pview(i);
132 out.dt(j) = dtView(i);
133 }
134 });
135 Kokkos::fence();
