75inline double gsl_sf_fact(unsigned int n) {
76 if (n > 170) {
77 // Factorial of numbers > 170 overflows double
78 // Use gamma function for large numbers
79 return std::tgamma(static_cast<double>(n + 1));
80 }
81 double result = 1.0;
82 for (unsigned int i = 2; i <= n; ++i) {
83 result *= static_cast<double>(i);
84 }
85 return result;
86}

93inline double gsl_sf_choose(unsigned int n, unsigned int m) {
94 if (m > n) return 0.0;
95 if (m == 0 || m == n) return 1.0;
96
97 // Use symmetry: choose(n, m) = choose(n, n-m)
98 if (m > n / 2) m = n - m;
99
100 // Compute using iterative method to avoid overflow
101 double result = 1.0;
102 for (unsigned int i = 0; i < m; ++i) {
103 result = result * (n - i) / (i + 1);
104 }
105 return result;
106}
