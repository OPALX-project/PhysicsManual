20 /* Kinematics of pi -> mu + nu_mu (two-body decay).
21 *
22 * With one massless neutrino, energy-momentum conservation in the pion
23 * rest frame fixes the muon momentum uniquely:
24 * E_mu = (M_pi^2 + m_mu^2) / (2 M_pi)
25 * |p| = (M_pi^2 - m_mu^2) / (2 M_pi)
26 *
27 * The direction is isotropic (pion is spin-0).
28 *
29 * Procedure per daughter:
30 * 1. Compute the fixed rest-frame momentum (done once, outside the loop).
31 * 2. Pick an isotropic direction in the parent rest frame.
32 * 3. Lorentz-boost the rest-frame 4-momentum to the lab frame using the
33 * parent's beta*gamma vector.
34 * 4. Store the result as beta*gamma = p_lab / m_daughter. */
35 const double M = parentMassGeV_m;
36 const double md = daughterMassGeV_m;
37 const double pFixed = (M * M - md * md) / (2.0 * M);
38 const double eDaughter = Kokkos::sqrt(pFixed * pFixed + md * md);
39
40 Kokkos::parallel_for(
41 "PionDecay::createDaughters", static_cast<pc_size_type>(localDestroyNum),

43 auto gen = pool.get_state();

45 /* Step 2: Isotropic direction in parent rest frame. */
46 const double cosTheta = 2.0 * gen.drand(0.0, 1.0) - 1.0;
47 const double sinTheta = Kokkos::sqrt(1.0 - cosTheta * cosTheta);
48 const double phi = 2.0 * Physics::pi * gen.drand(0.0, 1.0);
49 const double pxRF = pFixed * sinTheta * Kokkos::cos(phi);
50 const double pyRF = pFixed * sinTheta * Kokkos::sin(phi);
51 const double pzRF = pFixed * cosTheta;
52
53 /* Step 3: Lorentz boost to lab frame.
54 * Parent momentum is stored as beta*gamma (dimensionless).
55 * beta = P / gamma, with gamma = sqrt(1 + |P|^2).
56 * The boost formula for the spatial components is:
57 * p_lab = p_RF + [(gamma-1)(beta . p_RF)/beta^2 + gamma*E] * beta */
58 const auto& parentMom = parentP(j);
59 const double bg2 = parentMom[0] * parentMom[0] + parentMom[1] * parentMom[1]
60 + parentMom[2] * parentMom[2];
61 const double gamma = Kokkos::sqrt(1.0 + bg2);
62 const double betaX = parentMom[0] / gamma;
63 const double betaY = parentMom[1] / gamma;
64 const double betaZ = parentMom[2] / gamma;
65 const double beta2 = bg2 / (gamma * gamma);
66
67 double pxLab, pyLab, pzLab;
68 if (beta2 > 0.0) {
69 const double betaDotP = betaX * pxRF + betaY * pyRF + betaZ * pzRF;
70 const double factor = (gamma - 1.0) * betaDotP / beta2 + gamma * eDaughter;
71 pxLab = pxRF + factor * betaX;
72 pyLab = pyRF + factor * betaY;
73 pzLab = pzRF + factor * betaZ;
74 } else {
75 pxLab = pxRF;
76 pyLab = pyRF;
77 pzLab = pzRF;
78 }
79
80 /* Step 4: Write daughter attributes.
81 * Position inherits from parent. Momentum stored as beta*gamma = p / m. */
82 const pc_size_type idx = static_cast<pc_size_type>(oldDaughterLocal) + j;
83 dR(idx) = parentR(j);
84 dP(idx)[0] = pxLab / daughterMass;
85 dP(idx)[1] = pyLab / daughterMass;
86 dP(idx)[2] = pzLab / daughterMass;
87 dDt(idx) = parentDt(j);

90 });
