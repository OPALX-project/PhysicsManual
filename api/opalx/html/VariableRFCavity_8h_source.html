233KOKKOS_INLINE_FUNCTION bool VariableRFCavity::computeField(
234 const Vector_t<double, 3>& R, Vector_t<double, 3>& E, const double E0,
235 const double integralF, const double phi, const double halfWidth, const double halfHeight) {
236 E[2] += E0 * Kokkos::sin(Physics::two_pi * integralF + phi);
237 const bool outsideAperture = Kokkos::abs(R[0]) > halfWidth || Kokkos::abs(R[1]) > halfHeight;
238 return outsideAperture;
239}
