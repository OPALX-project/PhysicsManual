96 KOKKOS_INLINE_FUNCTION
97 static void computeField(
98 const Vector_t<double, 3>& R,
99 Vector_t<double, 3>& E,
100 Vector_t<double, 3>& B,
101 const ViewType& FourCoefs,
102 double zbegin,
103 double length,
104 double xlrep,
105 int accuracy)
106 {
107 const double RR2 = R(0) * R(0) + R(1) * R(1);
108
109 const double two_pi = Physics::two_pi;
110 const double pi = Physics::pi;
111 const double dk = Physics::two_pi / length;
112 const double kz = two_pi * (R(2) - zbegin) / length + pi;
113
114 double ez = FourCoefs(0);
115 double ezp = 0.0;
116 double ezpp = 0.0;
117 double ezppp = 0.0;
118
119 int n = 1;
120 for (int l = 1; l < accuracy; ++l, n += 2) {
121 const double base = dk * l;
122
123 const double coskzl = Kokkos::cos(kz * l);
124 const double sinkzl = Kokkos::sin(kz * l);
125
126 ez += FourCoefs(n) * coskzl - FourCoefs(n + 1) * sinkzl;
127 ezp += base * (-FourCoefs(n) * sinkzl - FourCoefs(n + 1) * coskzl);
128 ezpp += base * base * (-FourCoefs(n) * coskzl + FourCoefs(n + 1) * sinkzl);
129 ezppp += base * base * base * ( FourCoefs(n) * sinkzl + FourCoefs(n + 1) * coskzl);
130 }
131
132 const double f = -(ezpp + ez * xlrep * xlrep) / 16.0;
133 const double fp = -(ezppp + ezp * xlrep * xlrep) / 16.0;
134
135 const double EfieldR = -(ezp / 2.0 + fp * RR2);
136 const double BfieldT = (ez / 2.0 + f * RR2) * xlrep / Physics::c;
137
138 E(0) += EfieldR * R(0);
139 E(1) += EfieldR * R(1);
140 E(2) += ez + 4.0 * f * RR2;
141
142 B(0) -= BfieldT * R(1);
143 B(1) += BfieldT * R(0);
144 }
