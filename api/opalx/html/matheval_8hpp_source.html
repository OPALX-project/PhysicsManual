355 template <typename real_t> struct ConstantFolder {
357 using result_type = typename expr_ast<real_t>::tree_t;
358
360 result_type operator()(nil /*unused*/) const { return 0; }
361
363 result_type operator()(real_t n) const { return n; }
364
366 result_type operator()(std::string const &c) const { return c; }

369 result_type operator()(expr_ast<real_t> const &ast) const {
370 return std::visit(*this, ast.tree);
371 }
372
374 result_type operator()(binary_op<real_t> const &tree) const {
375 auto lhs = std::visit(*this, tree.lhs->tree);
376 auto rhs = std::visit(*this, tree.rhs->tree);
377
378 /* If both operands are known, we can directly evaluate the function,
379 * else we just update the children with the new expressions. */
380 if (holds_alternative<real_t>(lhs) && holds_alternative<real_t>(rhs)) {
381 return tree.op(std::get<real_t>(lhs), std::get<real_t>(rhs));
382 }
383 return binary_op<real_t>(tree.op, expr_ast<real_t>(lhs), expr_ast<real_t>(rhs));
384 }
385
387 result_type operator()(unary_op<real_t> const &tree) const {
388 auto rhs = std::visit(*this, tree.rhs->tree);
389 /* If the operand is known, we can directly evaluate the function. */
390 if (holds_alternative<real_t>(rhs)) {
391 return tree.op(std::get<real_t>(rhs));
392 }
393 return unary_op<real_t>(tree.op, expr_ast<real_t>(rhs));
394 }
395
397 result_type operator()(recursive_wrapper<expr_ast<real_t>> const& w) const {
398 return std::visit(*this, w.get().tree);
399 }
400
402 result_type operator()(recursive_wrapper<binary_op<real_t>> const& w) const {
403 return (*this)(w.get());
404 }
405
407 result_type operator()(recursive_wrapper<unary_op<real_t>> const& w) const {
408 return (*this)(w.get());
409 }

414 template <typename real_t>
415 class SimpleMathParser {
416 public:
417 SimpleMathParser(const std::string& input)
418 : input_(input), pos_(0) {}
419
420 expr_ast<real_t> parse() {
421 pos_ = 0;
422 skipWhitespace();
423 auto result = parseExpression();
424 skipWhitespace();
425 if (pos_ < input_.length()) {
426 throw std::runtime_error("Unexpected characters at end of expression");
427 }
428 return result;
429 }
430
431 private:
432 std::string input_;
433 size_t pos_;
434
435 void skipWhitespace() {
436 while (pos_ < input_.length() && std::isspace(static_cast<unsigned char>(input_[pos_]))) {
437 pos_++;
438 }
439 }
440
441 bool match(char c) {
442 skipWhitespace();
443 if (pos_ < input_.length() && input_[pos_] == c) {
444 pos_++;
445 return true;
446 }
447 return false;
448 }
449
450 bool match(const std::string& str) {
451 skipWhitespace();
452 if (input_.compare(pos_, str.length(), str) == 0) {
453 pos_ += str.length();
454 return true;
455 }
456 return false;
457 }
458
459 void expect(char c) {
460 if (!match(c)) {
461 throw std::runtime_error(std::string("Expected '") + c + "'");
462 }
463 }
464
465 expr_ast<real_t> parseExpression() {
466 auto result = parseTerm();
467 while (true) {
468 if (match('+')) {
469 result += parseTerm();
470 } else if (match('-')) {
471 result -= parseTerm();
472 } else {
473 break;
474 }
475 }
476 return result;
477 }
478
479 expr_ast<real_t> parseTerm() {
480 auto result = parseFactor();
481 while (true) {
482 if (match('*')) {
483 result *= parseFactor();
484 } else if (match('/')) {
485 result = expr_ast<real_t>(binary_op<real_t>(std::divides<real_t>{}, result, parseFactor()));
486 } else if (match('%')) {
487 auto fmod = static_cast<real_t(*)(real_t,real_t)>(&std::fmod);
488 result = expr_ast<real_t>(binary_op<real_t>(fmod, result, parseFactor()));
489 } else {
490 break;
491 }
492 }
493 return result;
494 }
495
496 expr_ast<real_t> parseFactor() {
497 auto result = parsePrimary();
498 while (match("**")) {
499 auto pow = static_cast<real_t(*)(real_t,real_t)>(&std::pow);
500 result = expr_ast<real_t>(binary_op<real_t>(pow, result, parseFactor()));
501 }
502 return result;
503 }
504
505 expr_ast<real_t> parsePrimary() {
506 skipWhitespace();
507
508 // Number
509 if (std::isdigit(static_cast<unsigned char>(input_[pos_])) ||
510 (pos_ < input_.length() && input_[pos_] == '.')) {
511 return parseNumber();
512 }
513
514 // Parentheses
515 if (match('(')) {
516 auto result = parseExpression();
517 expect(')');
518 return result;
519 }
520
521 // Unary operators
522 if (match('-')) {
523 return expr_ast<real_t>(unary_op<real_t>(std::negate<real_t>{}, parsePrimary()));
524 }
525 if (match('+')) {
526 return parsePrimary();
527 }
528
529 // Functions
530 std::string identifier = parseIdentifier();
531 if (identifier.empty()) {
532 throw std::runtime_error("Unexpected character");
533 }
534
535 // Check for constants
536 const auto& constants = getConstants();
537 if (constants.find(identifier) != constants.end()) {
538 return expr_ast<real_t>(constants.at(identifier));
539 }
540
541 // Check for unary functions
542 const auto& ufuncs = getUnaryFunctions();
543 if (ufuncs.find(identifier) != ufuncs.end()) {
544 expect('(');
545 auto arg = parseExpression();
546 expect(')');
547 return expr_ast<real_t>(unary_op<real_t>(ufuncs.at(identifier), arg));
548 }
549
550 // Check for binary functions
551 const auto& bfuncs = getBinaryFunctions();
552 if (bfuncs.find(identifier) != bfuncs.end()) {
553 expect('(');
554 auto arg1 = parseExpression();
555 expect(',');
556 auto arg2 = parseExpression();
557 expect(')');
558 return expr_ast<real_t>(binary_op<real_t>(bfuncs.at(identifier), arg1, arg2));
559 }
560
561 // Variable
562 return expr_ast<real_t>(identifier);
563 }
564
565 expr_ast<real_t> parseNumber() {
566 skipWhitespace();
567 size_t start = pos_;
568 bool has_dot = false;
569 bool has_exp = false;
570
571 if (pos_ < input_.length() && (input_[pos_] == '+' || input_[pos_] == '-')) {
572 pos_++;
573 }
574
575 while (pos_ < input_.length()) {
576 char c = input_[pos_];
577 if (std::isdigit(static_cast<unsigned char>(c))) {
578 pos_++;
579 } else if (c == '.' && !has_dot) {
580 has_dot = true;
581 pos_++;
582 } else if ((c == 'e' || c == 'E') && !has_exp) {
583 has_exp = true;
584 pos_++;
585 if (pos_ < input_.length() && (input_[pos_] == '+' || input_[pos_] == '-')) {
586 pos_++;
587 }
588 } else {
589 break;
590 }
591 }
592
593 if (pos_ > start) {
594 try {
595 real_t value = static_cast<real_t>(std::stod(input_.substr(start, pos_ - start)));
596 return expr_ast<real_t>(value);
597 } catch (...) {
598 throw std::runtime_error("Invalid number");
599 }
600 }
601 throw std::runtime_error("Expected number");
602 }
603
604 std::string parseIdentifier() {
605 skipWhitespace();
606 size_t start = pos_;
607 if (pos_ < input_.length() && (std::isalpha(static_cast<unsigned char>(input_[pos_])) || input_[pos_] == '_')) {
608 pos_++;
609 while (pos_ < input_.length() &&
610 (std::isalnum(static_cast<unsigned char>(input_[pos_])) || input_[pos_] == '_')) {
611 pos_++;
612 }
613 return input_.substr(start, pos_ - start);
614 }
615 return "";
616 }
