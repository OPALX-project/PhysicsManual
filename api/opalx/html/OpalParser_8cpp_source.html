62void OpalParser::parse(Statement& stat) const {
63 if (stat.keyword("SHARED")) {
64 // "SHARED ...": Shared object definition.
65 parseDefine(stat);
66 } else if (
67 stat.keyword("CONSTANT") || stat.keyword("CONST") || stat.keyword("BOOL")
68 || stat.keyword("REAL") || stat.keyword("STRING") || stat.keyword("VECTOR")) {
69 // Keywords introducing variable definitions.
70 parseAssign(stat);
71 } else {
72 std::string name = parseString(stat, "Identifier or keyword expected.");
73
74 if (stat.delimiter('?')) {
75 // "<class>?": give help for class.
76 printHelp(name);
77 } else if (stat.delimiter(':')) {
78 // "<object>:<class>...": labeled command.
79 parseDefine(stat);
80 } else if (stat.delimiter('(')) {
81 // "<macro>(...)...": macro definition or call.
82 // We are positioned just after the '(' of the argument list.
83 parseMacro(name, stat);
84 } else if (stat.delimiter(',') || stat.delimiter(';') || stat.atEnd()) {
85 // "<class>" or "<class>,<attributes>": Executable command.
86 parseAction(stat);
87 } else {
88 // Assignment beginning with a name.
89 stat.mark();
90 stat.start();
91 std::string objName = parseString(stat, "Object name expected.");
92 if (OpalData::getInstance()->find(objName) == 0) {
93 Token tok = stat.getCurrent();
94 stat.restore();
95
96 std::string name = tok.getLex();
97 std::string hint = getHint(name);
98 unsigned int position = stat.position();
99 std::string positionIndicator = std::string(position, ' ') + "^\n";
100 std::ostringstream statStr;
101 stat.print(statStr);
102 if (!hint.empty()) {
103 throw ParseError(
104 "OpalParser::parse()",
105 statStr.str() + positionIndicator
106 + "Syntax error, either the keyword REAL is missing or\n" + hint);
107 } else {
108 throw ParseError(
109 "OpalParser::parse()", statStr.str() + positionIndicator
110 + "Syntax error, the keyword REAL is missing\n");
111 }
112 }
113 parseAssign(stat);
114 }
115 }

118void OpalParser::execute(Object* object, const std::string& name) const {
119 // Trace execution.
120 if (Options::mtrace && object->shouldTrace()) {
121 double time = double(clock()) / double(CLOCKS_PER_SEC);
122 *gmsg << "\nBegin execution: \"" << name << "\", CPU time = " << time << " seconds.\n"
123 << endl;
124 }
125
126 // Force updating of all attributes which might have been changed.
127 if (object->shouldUpdate()) {
128 OpalData::getInstance()->update();
129 }
130
131 // Execute or check the command.
132 object->execute();
133
134 // Trace execution.
135 if (Options::mtrace && object->shouldTrace()) {
136 double time = double(clock()) / double(CLOCKS_PER_SEC);
137 *gmsg << "\nEnd execution: \"" << name << "\", CPU time = " << time << " seconds.\n"
138 << endl;
139 }

146void OpalParser::parseAction(Statement& stat) const {
147 stat.start();
148 std::string cmdName = parseString(stat, "Command name expected");
149
150 if (cmdName == "STOP") {
151 stopFlag = true;
152 } else if (cmdName == "QUIT") {
153 stopFlag = true;
154 } else if (cmdName == "HELP" && stat.delimiter(',')) {
155 cmdName = parseString(stat, "Object name expected");
156 printHelp(cmdName);
157 } else if (Object* object = find(cmdName)) {
158 Object* copy = 0;
159 try {
160 copy = object->clone("");
161 copy->parse(stat);
162 parseEnd(stat);
163 execute(copy, cmdName);
164 delete copy;
165 } catch (...) {
166 delete copy;
167 throw;
168 }
169 } else {
170 std::string hint = getHint(cmdName, "command");
171 if (!hint.empty()) {
172 throw ParseError("OpalParser::parseAction()", "Syntax error, " + hint);
173 }
174
175 throw ParseError("OpalParser::parseAction()", "Command \"" + cmdName + "\" is unknown.");
176 }

179void OpalParser::parseAssign(Statement& stat) const {
180 stat.start();
181
182 // Find various model objects.
183 /*static*/
184 Object* boolConstant = OpalData::getInstance()->find("BOOL_CONSTANT");
185 /*static*/
186 Object* realConstant = OpalData::getInstance()->find("REAL_CONSTANT");
187 /*static*/
188 Object* realVariable = OpalData::getInstance()->find("REAL_VARIABLE");
189 /*static*/
190 Object* realVector = OpalData::getInstance()->find("REAL_VECTOR");
191 /*static*/
192 Object* stringConstant = OpalData::getInstance()->find("STRING_CONSTANT");
193
194 // Gobble up any prefix.
195 int code = 0x00;
196 while (true) {
197 if (stat.keyword("CONSTANT") || stat.keyword("CONST")) {
198 code |= 0x01;
199 } else if (stat.keyword("BOOL")) {
200 code |= 0x02;
201 } else if (stat.keyword("REAL")) {
202 code |= 0x04;
203 } else if (stat.keyword("STRING")) {
204 code |= 0x08;
205 } else if (stat.keyword("VECTOR")) {
206 code |= 0x10;
207 } else {
208 break;
209 }
210 }
211
212 std::string objName = parseString(stat, "Object name expected.");
213 // Test for attribute name.
214 Object* object = 0;
215 std::string attrName;
216
217 if (stat.delimiter("->")) {
218 // Assignment to object attribute.
219 attrName = parseString(stat, "Attribute name expected.");
220
221 if (code != 0) {
222 throw ParseError(
223 "OpalParser::parseAssign()", "Invalid type specification for this value.");
224 } else if ((object = OpalData::getInstance()->find(objName)) == 0) {
225 throw ParseError(
226 "OpalParser::parseAssign()", "The object \"" + objName + "\" is unknown.");
227 }
228 } else {
229 // Assignment to variable-like object.
230 if ((object = OpalData::getInstance()->find(objName)) == 0) {
231 Object* model = 0;
232 switch (code) {
233 case 0x01: // CONSTANT
234 case 0x05: // CONSTANT REAL
235 model = realConstant;
236 break;
237 case 0x02: // BOOL
238 case 0x03: // BOOL CONSTANT
239 model = boolConstant;
240 break;
241 case 0x00: // empty <type>.
242 case 0x04: // REAL
243 model = realVariable;
244 break;
245 case 0x10: // VECTOR
246 case 0x11: // CONSTANT VECTOR
247 case 0x14: // REAL VECTOR
248 case 0x15: // CONSTANT REAL VECTOR
249 model = realVector;
250 break;
251 case 0x08: // STRING
252 case 0x09: // STRING CONSTANT
253 model = stringConstant;
254 break;
255 default:
256 break;
257 }
258
259 if (model != 0) {
260 object = model->clone(objName);
261 OpalData::getInstance()->define(object);
262 } else {
263 throw ParseError("OpalParser::parseAssign()", "Invalid <type> field.");
264 }
265 } else if (object->isTreeMember(realConstant)) {
266 throw ParseError(
267 "OpalParser::parseAssign()",
268 "You cannot redefine the constant \"" + objName + "\".");
269 }
270
271 attrName = "VALUE";
272 }
273
274 // Test for index; it is evaluated immediately.
275 int index = 0;
276
277 if (stat.delimiter('[')) {
278 index = int(std::round(parseRealConst(stat)));
279 parseDelimiter(stat, ']');
280
281 if (index <= 0) {
282 throw ParseError("Expressions::parseReference()", "Index must be positive.");
283 }
284 }
285
286 if (object != 0) {
287 if (Attribute* attr = object->findAttribute(attrName)) {
288 if (stat.delimiter('=') || object->isTreeMember(realConstant)) {
289 if (index > 0) {
290 attr->parseComponent(stat, true, index);
291 } else {
292 attr->parse(stat, true);
293 }
294 } else if (stat.delimiter(":=")) {
295 if (index > 0) {
296 attr->parseComponent(stat, false, index);
297 } else {
298 attr->parse(stat, false);
299 }
300 }
301 } else {
302 throw ParseError(
303 "OpalParser::parseAssign()",
304 "Object \"" + objName + "\" has no attribute \"" + attrName + "\".");
305 }
306
307 parseEnd(stat);
308 OpalData::getInstance()->makeDirty(object);
309 }
310}

312void OpalParser::parseDefine(Statement& stat) const {
313 stat.start();
314 bool isShared = stat.keyword("SHARED");
315 std::string objName = parseString(stat, "Object name expected.");
316
317 if (stat.delimiter(':')) {
318 std::string clsName = parseString(stat, "Class name expected.");
319 Object* classObject = find(clsName);
320
321 if (classObject == 0) {
322 if (clsName == "SURFACEPHYSICS")
323 throw ParseError(
324 "OpalParser::parseDefine()",
325 "The object \"" + clsName + "\" is changed to \"PARTICLEMATTERINTERACTION\".");
326 else
327 throw ParseError(
328 "OpalParser::parseDefine()", "The object \"" + clsName + "\" is unknown.");
329 }
330
331 Object* copy = 0;
332 try {
333 if (stat.delimiter('(')) {
334 // Macro-like objects are always classes, instances never.
335 // There is no further check required.
336 copy = classObject->makeInstance(objName, stat, this);
337 } else {
338 copy = classObject->clone(objName);
339 copy->parse(stat);
340 copy->setShared(isShared);
341 }
342
343 parseEnd(stat);
344 execute(copy, clsName);
345 OpalData::getInstance()->define(copy);
346 } catch (...) {
347 delete copy;
348 throw;
349 }
350 } else {
351 // Redefine an object to be a class.
352 Object* classObject = find(objName);
353 Object* copy = classObject->clone(objName);
354 copy->parse(stat);
355 copy->setShared(isShared);
356 }
357}

372void OpalParser::parseMacro(const std::string& macName, Statement& stat) const {
373 // Record the position just after the '(' of the argument list.
374 stat.mark();
375
376 // Skip argument list.
377 int par_level = 1;
378 while (true) {
379 if (stat.delimiter('(')) {
380 ++par_level;
381 } else if (stat.delimiter(')')) {
382 if (--par_level == 0)
383 break;
384 } else {
385 stat.getCurrent();
386 }
387 }
388
389 if (stat.delimiter(':')) {
390 // Macro definition.
391 std::string className = parseString(stat, "Class name expected.");
392
393 if (Object* macro = OpalData::getInstance()->find(className)) {
394 // Backtrack to first argument.
395 stat.restore();
396
397 if (Object* copy = macro->makeTemplate(macName, *inputStack.back(), stat)) {
398 OpalData::getInstance()->define(copy);
399 } else {
400 throw ParseError(
401 "OpalParser::parseMacro()",
402 "Command \"" + macName + "\" cannot be defined with arguments.");
403 }
404 } else {
405 throw ParseError(
406 "OpalParser::parseMacro()", "Object \"" + className + "\" is unknown.");
407 }
408 } else {
409 // Macro call.
410 if (Object* macro = OpalData::getInstance()->find(macName)) {
411 // Backtrack to first argument.
412 stat.restore();
413 Object* instance = 0;
414 try {
415 instance = macro->makeInstance(macName, stat, this);
416 execute(instance, macName);
417 } catch (...) {
418 delete instance;
419 throw;
420 }
421 } else {
422 throw ParseError("OpalParser::parseMacro()", "Macro \"" + macName + "\" is unknown.");
423 }
424 }

437void OpalParser::parseBracketList(char close, Statement& stat) {
438 Token token = readToken();
439
440 while (!token.isEOF()) {
441 stat.append(token);
442
443 if (token.isDel('(')) {
444 parseBracketList(')', stat);
445 } else if (token.isDel('[')) {
446 parseBracketList(']', stat);
447 } else if (token.isDel('{')) {
448 parseBracketList('}', stat);
449 } else if (token.isDel(close)) {
450 return;
451 }
452
453 token = readToken();
454 }
455}

457void OpalParser::parseTokenList(Statement& stat) {
458 Token token = readToken();
459
460 while (!token.isEOF()) {
461 // End of list if semicolon occurs outside of brackets.
462 if (token.isDel(';'))
463 break;
464 stat.append(token);
465
466 if (token.isDel('(')) {
467 parseBracketList(')', stat);
468 } else if (token.isDel('[')) {
469 parseBracketList(']', stat);
470 } else if (token.isDel('{')) {
471 parseBracketList('}', stat);
472 }
473
474 token = readToken();
475 }

486Statement* OpalParser::readStatement(TokenStream* is) const {
487 Statement* stat = 0;
488 Token token = is->readToken();
489
490 try {
491 if (token.isDel('{')) {
492 // Compound statement.
493 inputStack.back()->putBack(token);
494 stat = new CompoundStatement(*inputStack.back());
495 } else if (token.isKey("IF")) {
496 // IF statement.
497 inputStack.back()->putBack(token);
498 stat = new IfStatement(*this, *inputStack.back());
499 } else if (token.isKey("WHILE")) {
500 // WHILE statement.
501 inputStack.back()->putBack(token);
502 stat = new WhileStatement(*this, *inputStack.back());
503 } else if (token.isWord() || token.isString()) {
504 // Simple statement or MACRO statement.
505 stat = new SimpleStatement(token.getFile(), token.getLine());
506 stat->append(token);
507 token = is->readToken();
508
509 if (!token.isEOF()) {
510 if (token.isDel('(')) {
511 // Macro statement; statement already contains initial word.
512 stat->append(token);
513 parseBracketList(')', *stat);
514 token = is->readToken();
515
516 if (!token.isEOF() && token.isDel(':')) {
517 // Macro definition.
518 stat->append(token);
519 token = is->readToken();
520
521 if (!token.isEOF()) {
522 stat->append(token);
523 if (token.isKey("MACRO")) {
524 token = is->readToken();
525
526 if (!token.isEOF() && token.isDel('{')) {
527 stat->append(token);
528 parseBracketList('}', *stat);
529 } else {
530 throw ParseError(
531 "OpalParser::readStatement()",
532 "MACRO definition lacks \"{...}\".");
533 }
534 } else {
535 parseTokenList(*stat);
536 }
537 }
538 } else if (!token.isDel(';')) {
539 throw ParseError(
540 "OpalParser::readStatement()", "MACRO call is not terminated by ';'.");
541 }
542 } else if (!token.isDel(';')) {
543 stat->append(token);
544 parseTokenList(*stat);
545 }
546 }
547 stat->start();
548 } else if (token.isDel(';')) {
549 // Skip empty statement.
550 stat = readStatement(is);
551 } else if (token.isDel('?')) {
552 // Give help.
553 *gmsg << "\ntry typing \"HELP\" for help.\n" << endl;
554 stat = readStatement(is);
555 } else if (!token.isEOF()) {
556 stat = new SimpleStatement(token.getFile(), token.getLine());
557 stat->append(token);
558 parseTokenList(*stat);
559 stat->start();
560 throw ParseError("OpalParser::readStatement()", "Command should begin with a <name>.");
561 }
562 } catch (ParseError& ex) {
563 *ippl::Error << "\n*** Parse error detected by function \""
564 << "OpalParser::readStatement()"
565 << "\"\n";
567
568 std::string what = ex.what();
569 std::string::size_type pos = 0;
570 while ((pos = what.find("\n", pos)) != std::string::npos) {
571 what.replace(pos, 1, "\n ");
572 pos += 5; // length of "\n "
573 }
574
575 *ippl::Error << " " << *stat << " a" << what << '\n' << endl;
576
577 stat = readStatement(is);
578 exit(1);
579 }
580
581 return stat;
582}

584void OpalParser::run() const {
585 stopFlag = false;
586 while (Statement* stat = readStatement(&*inputStack.back())) {
587 try {
588 // The dispatch via Statement::execute() allows a special
589 // treatment of structured statements.
590 stat->execute(*this);
591 } catch (ParseError& ex) {
592 Inform errorMsg("Error", std::cerr);
593 errorMsg << "\n*** Parse error detected by function \"" << ex.where() << "\"\n";
594 stat->printWhere(errorMsg, true);
595 std::string what = ex.what();
596 size_t pos = what.find_first_of('\n');
597 do {
598 errorMsg << " " << what.substr(0, pos) << endl;
599 what = what.substr(pos + 1, std::string::npos);
600 pos = what.find_first_of('\n');
601 } while (pos != std::string::npos);
602 errorMsg << " " << what << endl;
603
604 MPI_Abort(MPI_COMM_WORLD, -100);
605 }
606
607 delete stat;
608 if (stopFlag)
609 break;
610 }
611}

623std::string OpalParser::getHint(const std::string& name, const std::string& type) {
624 auto owner = AttributeHandler::getOwner(name);
625 if (owner.empty()) {
626 return std::string();
627 }
628
629 std::string hint = "the " + type + " '" + name + "' could belong to\n";
630 {
631 std::string elements;
632 auto its = owner.equal_range(AttributeHandler::ELEMENT);
633 if (its.first != its.second) {
634 elements = (its.first)->second;
635 bool any = (its.first)->second == "Any";
636 for (auto it = std::next(its.first); it != its.second && !any; ++it) {
637 elements += ", " + it->second;
638 any = it->second == "Any";
639 }
640 if (any) {
641 hint += std::string(" - any element\n");
642 } else {
643 hint += std::string(" - the element")
644 + (std::distance(its.first, its.second) > 1 ? "s " : " ") + elements + "\n";
645 }
646 }
647 }
648 {
649 std::string commands;
650 auto its = owner.equal_range(AttributeHandler::COMMAND);
651 if (its.first != its.second) {
652 commands = (its.first)->second;
653 for (auto it = std::next(its.first); it != its.second; ++it) {
654 commands += ", " + it->second;
655 }
656 hint += std::string(" - the command")
657 + (std::distance(its.first, its.second) > 1 ? "s " : " ") + commands + "\n";
658 }
659 }
660 {
661 std::string sub_commands;
662 auto its = owner.equal_range(AttributeHandler::SUB_COMMAND);
663 if (its.first != its.second) {
664 sub_commands = (its.first)->second;
665 for (auto it = std::next(its.first); it != its.second; ++it) {
666 sub_commands += ", " + it->second;
667 }
668 hint += std::string(" - the sub-command")
669 + (std::distance(its.first, its.second) > 1 ? "s " : " ") + sub_commands + "\n";
670 }
671 }
672 {
673 std::string statements;
674 auto its = owner.equal_range(AttributeHandler::STATEMENT);
675 if (its.first != its.second) {
676 statements = (its.first)->second;
677 for (auto it = std::next(its.first); it != its.second; ++it) {
678 statements += ", " + it->second;
679 }
680 hint += std::string(" - the statement")
681 + (std::distance(its.first, its.second) > 1 ? "s " : " ") + statements + "\n";
682 }
683 }
684
685 hint += "but it's not present!";
686 return hint;
