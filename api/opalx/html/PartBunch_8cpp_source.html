344template <typename T, unsigned Dim>
345void PartBunch<T, Dim>::calcBeamParameters() {
346 Inform m("PartBunch::calcBeamParameters");
347 std::shared_ptr<ParticleContainer_t> pc = this->pcontainer_m;
348
349 using view_type = ippl::ParticleAttrib<Vector_t<double, 3>>::view_type;
350 view_type Rview = pc->R.getView();
351 view_type Pview = pc->P.getView();
352 this->getParticleContainer()->updateMoments();
353 m << level5 << "Moments updated." << endl;
354
358
359 using MomentsVec = ippl::Vector<double, 2 * Dim>;
360 using MomentsMat = ippl::Vector<MomentsVec, 2 * Dim>;
361
362 MomentsVec loc_centroid(0.0);
363 MomentsMat loc_moment(MomentsVec(0.0));
364
365 MomentsVec centroid(0.0);
366 MomentsMat moment(MomentsVec(0.0));
367
368 for (unsigned i = 0; i < 2 * Dim; ++i) {
369 const size_t nLocal = pc->getLocalNum();
370 Kokkos::parallel_reduce(
371 "calc moments of particle distr.", nLocal,
372 KOKKOS_LAMBDA(
373 const size_t k, double& cent, double& mom0, double& mom1, double& mom2,
374 double& mom3, double& mom4, double& mom5) {
375 double part[2 * Dim];
376 part[0] = Rview(k)[0];
377 part[1] = Pview(k)[0];
378 part[2] = Rview(k)[1];
379 part[3] = Pview(k)[1];
380 part[4] = Rview(k)[2];
381 part[5] = Pview(k)[2];
382
383 cent += part[i];
384 mom0 += part[i] * part[0];
385 mom1 += part[i] * part[1];
386 mom2 += part[i] * part[2];
387 mom3 += part[i] * part[3];
388 mom4 += part[i] * part[4];
389 mom5 += part[i] * part[5];
390 },
391 Kokkos::Sum<T>(loc_centroid[i]), Kokkos::Sum<T>(loc_moment[i][0]),
392 Kokkos::Sum<T>(loc_moment[i][1]), Kokkos::Sum<T>(loc_moment[i][2]),
393 Kokkos::Sum<T>(loc_moment[i][3]), Kokkos::Sum<T>(loc_moment[i][4]),
394 Kokkos::Sum<T>(loc_moment[i][5]));
395 Kokkos::fence();
396 }
397 m << level5 << "Local moments calculated." << endl;
398
399 moment = loc_moment;
400 centroid = loc_centroid;
401 ippl::Comm->allreduce(moment, 1, std::plus<MomentsMat>());
402 ippl::Comm->allreduce(centroid, 1, std::plus<MomentsVec>());
403 ippl::Comm->barrier();
404 m << level5 << "Global moments calculated." << endl;
405
406 ippl::Vector<double, Dim> rmax_loc(0.0);
407 ippl::Vector<double, Dim> rmin_loc(0.0);
408 ippl::Vector<double, Dim> rmax(0.0);
409 ippl::Vector<double, Dim> rmin(0.0);
410
411 // TODO: fuse min/max reductions with ippl::Vector reductions.
412 for (unsigned d = 0; d < Dim; ++d) {
413 Kokkos::parallel_reduce(
414 "rel max", pc->getLocalNum(),
415 KOKKOS_LAMBDA(const int i, double& mm) {
416 double tmp_vel = Rview(i)[d];
417 mm = tmp_vel > mm ? tmp_vel : mm;
418 },
419 Kokkos::Max<T>(rmax_loc[d]));
420
421 Kokkos::parallel_reduce(
422 "rel min", pc->getLocalNum(),
423 KOKKOS_LAMBDA(const int i, double& mm) {
424 double tmp_vel = Rview(i)[d];
425 mm = tmp_vel < mm ? tmp_vel : mm;
426 },
427 Kokkos::Min<T>(rmin_loc[d]));
428 }
429 m << level5 << "Local min/max calculated." << endl;
430 Kokkos::fence();
431 rmax = rmax_loc;
432 rmin = rmin_loc;
433 ippl::Comm->allreduce(rmax, 1, std::greater<ippl::Vector<double, Dim>>());
434 ippl::Comm->allreduce(rmin, 1, std::less<ippl::Vector<double, Dim>>());
435 ippl::Comm->barrier();
436 m << level5 << "Global min/max calculated." << endl;
437
438 rmax_m = rmax;
439 rmin_m = rmin;
440}

535 if (nr_m[Dim - 1] == nrZ) {
536 return;
537 }
538
539 Inform m("PartBunch::reinitializeGridZ");
540 m << level3 << "Resizing z grid: " << nr_m[Dim - 1] << " -> " << nrZ << " cells." << endl;
541
542 nr_m[Dim - 1] = nrZ;
543 domain_m[Dim - 1] = ippl::Index(nrZ);
544
545 const bool isAllPeriodic = this->getBCHandler()->isAll(BCHandler_t::PERIODIC);

555 this->fcontainer_m->getE().updateLayout(fl);
556 if (solver_m == "CG") {
557 this->fcontainer_m->getPhi().updateLayout(fl);
558 }
559 this->getTempEField()->updateLayout(fl);
560 this->getTempBField()->updateLayout(fl);

562 m << level3 << "Grid z reinit complete (nrZ=" << nrZ << ")." << endl;

600template <typename T, unsigned Dim>
601void PartBunch<T, Dim>::computeBoundsForFieldSolve(
602 Vector_t<double, Dim>& lower, Vector_t<double, Dim>& upper) {
603 Inform m("PartBunch::computeBoundsForFieldSolve");
604
605 const auto& containers = this->getParticleContainers();
606
607 bool hasNonEmptyContainer = false;
608 for (const auto& pc : containers) {
609 if (!pc || pc->getTotalNum() == 0) {
610 continue;
611 }
612
613 pc->computeMinMaxR();
614 const ippl::Vector<double, 3> minR = pc->getMinR();
615 const ippl::Vector<double, 3> maxR = pc->getMaxR();
616
617 if (!hasNonEmptyContainer) {
618 lower = minR;
619 upper = maxR;
620 hasNonEmptyContainer = true;
621 } else {
622 for (int i = 0; i < 3; ++i) {
623 lower[i] = std::min(lower[i], minR[i]);
624 upper[i] = std::max(upper[i], maxR[i]);
625 }
626 }
627 }
628
629 if (!hasNonEmptyContainer) {
630 if (containers.empty() || !containers[0]) {
631 throw OpalException(
632 "PartBunch::bunchUpdate",
633 "No valid particle container available for bunch update.");
634 }
635 containers[0]->computeMinMaxR();
636 lower = containers[0]->getMinR();
637 upper = containers[0]->getMaxR();
638 }
639
640 const BinnedFieldSolver_t* bsolver = this->getFieldSolver();
641
642 // Include mirrored particles in the domain envelope when image-charge mode is active for this
643 // step.
644 if (bsolver && bsolver->isImageChargeActiveForStep(this->getGlobalTrackStep())) {
645 const double planeZ = bsolver->getImageChargePlaneZ();
646 const double mirroredMinZ = 2.0 * planeZ - upper[2];
647 const double mirroredMaxZ = 2.0 * planeZ - lower[2];
648 lower[2] = std::min(lower[2], mirroredMinZ);
649 upper[2] = std::max(upper[2], mirroredMaxZ);
650 m << level4 << "Image-charge bounds enabled at zPlane=" << planeZ << endl;
651 }
652
653 Vector_t<double, Dim> span = upper - lower;
654 for (unsigned i = 0; i < Dim; ++i) {
655 if (span[i] < 1e-6) {
656 span[i] = 1e-6;
657 m << level3 << "Mesh spacing in dimension " << i << " too small. Set to 1e-6." << endl;
658 }
659 }
660
661 lower = lower - span * this->OPALFieldSolver_m->getBoxIncr() / 100.0;
662 upper = upper + span * this->OPALFieldSolver_m->getBoxIncr() / 100.0;
663}

667 const Vector_t<double, Dim>& lower, const Vector_t<double, Dim>& upper) {
668 Inform m("PartBunch::applyGridUpdate");

672

675
676 mesh->setMeshSpacing(hr_m);
677 mesh->setOrigin(lower);

688 const auto& containers = this->getParticleContainers();
689 for (size_t i = 0; i < containers.size(); ++i) {
690 const auto& pc = containers[i];
691 if (!pc) {
692 continue;
693 }
694 pc->getLayout().updateLayout(*FL, *mesh);

696 pc->markMomentsDirty(); // IPPL migration may have re-indexed R across ranks
