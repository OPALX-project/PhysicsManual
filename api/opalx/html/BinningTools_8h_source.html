33 template <typename bin_index_type>
34 HistoReductionMode determineHistoReductionMode(HistoReductionMode modePreference, bin_index_type binCount) {
35 // Overwrite standard mode if compiled with default host execution space!
36 if (std::is_same<Kokkos::DefaultExecutionSpace, Kokkos::DefaultHostExecutionSpace>::value)
37 return HistoReductionMode::HostOnly;
38
39#ifdef OPALX_DEVICE_COMPILATION
40 // In device builds HostOnly reductions are not supported; calling them is a logic error.
41 if (modePreference == HistoReductionMode::HostOnly) {
42 throw OpalException(
43 "ParticleBinning::determineHistoReductionMode",
44 "HistoReductionMode::HostOnly is not supported when OPALX_DEVICE_COMPILATION is "
45 "enabled.");
46 }
47#endif
48 // Otherwise choose automatically if Standard and respect preference if not on host and not standard!
49 if (modePreference == HistoReductionMode::Standard) {
50 return (binCount <= maxArrSize<bin_index_type>) ? HistoReductionMode::ParallelReduce : HistoReductionMode::TeamBased;
51 } else {
52 return modePreference;
53 }
54 }
