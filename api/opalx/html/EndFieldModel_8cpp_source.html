36 bool GreaterThan(std::vector<int> v1, std::vector<int> v2) {
37 size_t n1(v1.size()), n2(v2.size());
38 for (size_t i = 0; i < n1 && i < n2; ++i) {
39 if (v1[n1 - 1 - i] > v2[n2 - 1 - i]) return true;
40 if (v1[n1 - 1 - i] < v2[n2 - 1 - i]) return false;
41 }
42 return false;
43 }

45 std::vector<std::vector<int> > CompactVector(std::vector<std::vector<int> > vec) {
46 // first sort the list
47 std::sort(vec.begin(), vec.end(), GreaterThan);
48 // now look for n = n+1
49 for (size_t j = 0; j < vec.size() - 1; ++j) {
50 while (j < vec.size() - 1
51 && IterableEquality(
52 vec[j].begin() + 1, vec[j].end(), vec[j + 1].begin() + 1,
53 vec[j + 1].end())) {
54 vec[j][0] += vec[j + 1][0];
55 vec.erase(vec.begin() + j + 1);
56 }
57 }
58 return vec;
59 }
