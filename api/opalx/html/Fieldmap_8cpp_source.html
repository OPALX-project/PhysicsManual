154MapType Fieldmap::readHeader(std::string Filename) {
155 char magicnumber[5] = " ";
156 std::string buffer;
157 int lines_read_m = 0;
158
159 // Check for default map(s).
160 if (Filename == "1DPROFILE1-DEFAULT") return T1DProfile1;
161
162 if (Filename.empty())
163 throw GeneralOpalException("Fieldmap::readHeader()", "No field map file specified");
164
165 if (!fs::exists(Filename))
166 throw GeneralOpalException(
167 "Fieldmap::readHeader()", "File '" + Filename + "' doesn't exist");

169 std::ifstream File(Filename.c_str());

171 std::cerr << "could not open file " << Filename << std::endl;

214 if (std::strcmp(magicnumber, "1DMa") == 0) return T1DMagnetoStatic;

247 h5err = H5BlockGetFieldInfo(

264 if (std::strcmp(magicnumber, "Astr") == 0) {
265 char tmpString[3] = " ";

267 if (std::strcmp(tmpString, "aE") == 0) {

326void Fieldmap::checkMap(
327 unsigned int accuracy, double length, const std::vector<double>& zSampling,
328 const std::vector<double>& fourierCoefficients, gsl_spline* splineCoefficients,
329 gsl_interp_accel* splineAccelerator) {
330 double error = 0.0;
331 double maxDiff = 0.0;
332 double ezMax = 0.0;
333 double ezSquare = 0.0;
334 size_t lastDot = Filename_m.find_last_of(".");
335 size_t lastSlash = Filename_m.find_last_of("/");
336 lastSlash = (lastSlash == std::string::npos) ? 0 : lastSlash + 1;
337
338 auto opal = OpalData::getInstance();
339 std::ofstream out;
340 if (ippl::Comm->rank() == 0 && !opal->isOptimizerRun()) {
341 std::string fname = Util::combineFilePath(
342 {opal->getAuxiliaryOutputDirectory(),
343 Filename_m.substr(lastSlash, lastDot) + ".check"});
344 out.open(fname);
345 out << "# z original reproduced\n";
346 }
347 auto it = zSampling.begin();
348 auto end = zSampling.end();
349 for (; it != end; ++it) {
350 const double kz = Physics::two_pi * (*it / length + 0.5);
351 double onAxisFieldCheck = fourierCoefficients[0];
352 unsigned int n = 1;
353 for (unsigned int l = 1; l < accuracy; ++l, n += 2) {
354 double coskzl = std::cos(kz * l);
355 double sinkzl = std::sin(kz * l);
356
357 onAxisFieldCheck +=
358 (fourierCoefficients[n] * coskzl - fourierCoefficients[n + 1] * sinkzl);
359 }
360 double ez = gsl_spline_eval(splineCoefficients, *it, splineAccelerator);
361 double difference = std::abs(ez - onAxisFieldCheck);
362 maxDiff = difference > maxDiff ? difference : maxDiff;
363 ezMax = std::abs(ez) > ezMax ? std::abs(ez) : ezMax;
364 error += std::pow(difference, 2.0);
365 ezSquare += std::pow(ez, 2.0);
366
367 if (ippl::Comm->rank() == 0 && !opal->isOptimizerRun()) {
368 out << std::setw(16) << std::setprecision(8) << *it << std::setw(16)
369 << std::setprecision(8) << ez << std::setw(16) << std::setprecision(8)
370 << onAxisFieldCheck << std::endl;
371 }
372 }
373 out.close();
374
375 if (std::sqrt(error / ezSquare) > 1e-1 || maxDiff > 1e-1 * ezMax) {
376 lowResolutionWarning(std::sqrt(error / ezSquare), maxDiff / ezMax);
377
378 throw GeneralOpalException(
379 "Fieldmap::checkMap",
380 "Field map can't be reproduced properly with the given number of fourier "
381 "components");
382 }
383 if (std::sqrt(error / ezSquare) > 1e-2 || maxDiff > 1e-2 * ezMax) {
384 lowResolutionWarning(std::sqrt(error / ezSquare), maxDiff / ezMax);
385 }
386}

507std::string Fieldmap::typeset_msg(const std::string& msg, const std::string& title) {
508 static std::string frame(
509 "* ******************************************************************************\n");
510 static unsigned int frame_width = frame.length() - 5;
511 static std::string closure(
512 " *\n");
513
514 std::string return_string("\n" + frame);
515
516 int remaining_length = msg.length();
517 unsigned int current_position = 0;
518
519 unsigned int ii = 0;
520 for (; ii < title.length(); ++ii) {
521 char c = title[ii];
522 c = std::toupper(c);
523 return_string.replace(15 + 2 * ii, 1, " ");
524 return_string.replace(16 + 2 * ii, 1, &c, 1);
525 }
526 return_string.replace(15 + 2 * ii, 1, " ");
527
528 while (remaining_length > 0) {
529 size_t eol = msg.find("\n", current_position);
530 std::string next_to_process;
531 if (eol != std::string::npos) {
532 next_to_process = msg.substr(current_position, eol - current_position);
533 } else {
534 next_to_process = msg.substr(current_position);
535 eol = msg.length();
536 }
537
538 if (eol - current_position < frame_width) {
539 return_string += "* " + next_to_process + closure.substr(eol - current_position + 2);
540 } else {
541 unsigned int last_space = next_to_process.rfind(" ", frame_width);
542 if (last_space > 0) {
543 if (last_space < frame_width) {
544 return_string += "* " + next_to_process.substr(0, last_space)
545 + closure.substr(last_space + 2);
546 } else {
547 return_string += "* " + next_to_process.substr(0, last_space) + " *\n";
548 }
549 if (next_to_process.length() - last_space + 1 < frame_width) {
550 return_string += "* " + next_to_process.substr(last_space + 1)
551 + closure.substr(next_to_process.length() - last_space + 1);
552 } else {
553 return_string += "* " + next_to_process.substr(last_space + 1) + " *\n";
554 }
555 } else {
556 return_string += "* " + next_to_process + " *\n";
557 }
558 }
559
560 current_position = eol + 1;
561 remaining_length = msg.length() - current_position;
562 }
563 return_string += frame;
564
565 return return_string;
566}
