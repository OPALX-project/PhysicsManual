40// Does the user interface affect the correct magnet configuration
41TEST_F(TestOpalMultipoleT, UserInterface) {
42 // Make the UI
43 OpalMultipoleT ui;
44 // Set the attributes
45 Attributes::setReal(ui.itsAttr[OpalMultipoleT::LENGTH], 4.1);
46 Attributes::setRealArray(ui.itsAttr[OpalMultipoleT::TP], {0.2, 0.3});
47 Attributes::setReal(ui.itsAttr[OpalMultipoleT::LFRINGE], 0.5);
48 Attributes::setReal(ui.itsAttr[OpalMultipoleT::RFRINGE], 0.6);
49 Attributes::setReal(ui.itsAttr[OpalMultipoleT::HAPERT], 1.0);
50 Attributes::setReal(ui.itsAttr[OpalMultipoleT::VAPERT], 1.1);
51 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXFORDER], 4.0);
52 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.0);
53 Attributes::setReal(ui.itsAttr[OpalMultipoleT::EANGLE], 0.01);
54 Attributes::setReal(ui.itsAttr[OpalMultipoleT::BBLENGTH], 6.0);
55 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.628);
56 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXXORDER], 7.0);
57 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], false);
58 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 0.0);
59 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_H], 1.3);
60 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_V], 1.4);
61 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_S], 1.5);
62 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_ROLL], 1.6);
63 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_YAW], 1.7);
64 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MISALIGN_PITCH], 1.8);
65 // Update the magnet
66 EXPECT_NO_THROW(ui.update());
67 // Check the values
68 auto* myMagnet = dynamic_cast<MultipoleT*>(ui.getElement());
69 EXPECT_TRUE(myMagnet);
70 EXPECT_NEAR(myMagnet->getElementLength(), 4.1, 1e-6);
71 auto tp = myMagnet->getTransProfile();
72 EXPECT_EQ(tp.size(), MultipoleTConfig::NumPoles);
73 EXPECT_NEAR(tp[0], 0.2, 1e-6);
74 EXPECT_NEAR(tp[1], 0.3, 1e-6);
75 auto [s0, left, right] = myMagnet->getFringeField();
76 EXPECT_NEAR(s0, 4.1 / 2.0, 1e-6);
77 EXPECT_NEAR(left, 0.5, 1e-6);
78 EXPECT_NEAR(right, 0.6, 1e-6);
79 auto [vertical, horizontal] = myMagnet->getAperture();
80 EXPECT_NEAR(vertical, 1.1, 1e-6);
81 EXPECT_NEAR(horizontal, 1.0, 1e-6);
82 EXPECT_NEAR(myMagnet->getMaxFOrder(), 4.0, 1e-6);
83 EXPECT_NEAR(myMagnet->getRotation(), 0.0, 1e-6);
84 EXPECT_NEAR(myMagnet->getEntranceAngle(), 0.01, 1e-6);
85 EXPECT_NEAR(myMagnet->getBoundingBoxLength(), 6.0, 1e-6);
86 EXPECT_NEAR(myMagnet->getBendAngle(), 0.628, 1e-6);
87 EXPECT_NEAR(myMagnet->getMaxXOrder(), 7.0, 1e-6);
88 EXPECT_FALSE(myMagnet->getVariableRadius());
89 EXPECT_NEAR(myMagnet->getEntryOffset(), 0.0, 1e-6);
90 // Check rotation (only works for straight magnets)
91 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.0);
92 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.1);
93 EXPECT_NO_THROW(ui.update());
94 EXPECT_NEAR(myMagnet->getRotation(), 0.1, 1e-6);
95 // Check entry offset (only works for var radius magnets)
96 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.0);
97 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.5);
98 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 1.2);
99 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], true);
100 EXPECT_ANY_THROW(ui.update()); // Currently throws 'cos var radius not yet supported
101 // Check time dependency
102 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 0.0);
103 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], false);
104 Attributes::setUpperCaseString(ui.itsAttr[OpalMultipoleT::SCALING_MODEL], "Scaling");
105 EXPECT_NO_THROW(ui.update());
106 EXPECT_EQ(myMagnet->getScalingName(), "SCALING");
107}

164TEST_F(TestOpalMultipoleT, UserInterfaceSanityCheck) {
165 // Make the UI
166 OpalMultipoleT ui;
167 // Set the attributes
168 Attributes::setReal(ui.itsAttr[OpalMultipoleT::LENGTH], 4.1);
169 Attributes::setRealArray(ui.itsAttr[OpalMultipoleT::TP], {0.2, 0.3});
170 Attributes::setReal(ui.itsAttr[OpalMultipoleT::LFRINGE], 0.5);
171 Attributes::setReal(ui.itsAttr[OpalMultipoleT::RFRINGE], 0.6);
172 Attributes::setReal(ui.itsAttr[OpalMultipoleT::HAPERT], 1.0);
173 Attributes::setReal(ui.itsAttr[OpalMultipoleT::VAPERT], 1.1);
174 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.0);
175 Attributes::setReal(ui.itsAttr[OpalMultipoleT::EANGLE], 0.0);
176 Attributes::setReal(ui.itsAttr[OpalMultipoleT::BBLENGTH], 6.0);
177 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.0);
178 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXXORDER], 7.0);
179 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], false);
180 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 0.0);
181 // Try F order of zero
182 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXFORDER], 0.0);
183 EXPECT_ANY_THROW(ui.update());
184 // Try F order of 21
185 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXFORDER], 21.0);
186 EXPECT_ANY_THROW(ui.update());
187 // Try skew straight magnet
188 Attributes::setReal(ui.itsAttr[OpalMultipoleT::MAXFORDER], 4);
189 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.0);
190 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.5);
191 EXPECT_NO_THROW(ui.update());
192 // Try skew bent magnet
193 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.5);
194 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.5);
195 EXPECT_ANY_THROW(ui.update());
196 // Try the variable radius magnet
197 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ROTATION], 0.0);
198 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.5);
199 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], true);
200 EXPECT_ANY_THROW(ui.update());
201 // Try the entry offset with a straight magnet
202 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.0);
203 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], false);
204 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 1.0);
205 EXPECT_ANY_THROW(ui.update());
206 // Try the entry offset with a bent magnet
207 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.5);
208 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], false);
209 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 1.0);
210 EXPECT_ANY_THROW(ui.update());
211 // Try the entry offset with a variable radius bent magnet
212 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ANGLE], 0.5);
213 Attributes::setBool(ui.itsAttr[OpalMultipoleT::VARRADIUS], true);
214 Attributes::setReal(ui.itsAttr[OpalMultipoleT::ENTRYOFFSET], 1.0);
215 EXPECT_ANY_THROW(ui.update());
216}
