396void Option::execute() {
397 // Store the option flags.
398 echo = Attributes::getBool(itsAttr[ECHO]);
399 info = Attributes::getBool(itsAttr[INFO]);
400 mtrace = Attributes::getBool(itsAttr[TRACE]);
401 warn = Attributes::getBool(itsAttr[WARN]);
402 psDumpEachTurn = Attributes::getBool(itsAttr[PSDUMPEACHTURN]);
403 remotePartDel = Attributes::getReal(itsAttr[REMOTEPARTDEL]);
404 rhoDump = Attributes::getBool(itsAttr[RHODUMP]);
405 ebDump = Attributes::getBool(itsAttr[EBDUMP]);
406 csrDump = Attributes::getBool(itsAttr[CSRDUMP]);
407 enableHDF5 = Attributes::getBool(itsAttr[ENABLEHDF5]);
408 enableVTK = Attributes::getBool(itsAttr[ENABLEVTK]);
409 idealized = Attributes::getBool(itsAttr[IDEALIZED]);
410 asciidump = Attributes::getBool(itsAttr[ASCIIDUMP]);
411 version = Attributes::getReal(itsAttr[VERSION]);
412 seed = Attributes::getReal(itsAttr[SEED]);
413 writeBendTrajectories = Attributes::getBool(itsAttr[LOGBENDTRAJECTORY]);
414
415 haloShift = Attributes::getReal(itsAttr[HALOSHIFT]);
416 delPartFreq = Attributes::getReal(itsAttr[DELPARTFREQ]);
417 computePercentiles = Attributes::getBool(itsAttr[COMPUTEPERCENTILES]);
418 const std::string qmMode = Attributes::getString(itsAttr[QM_MODE]);
419 if (qmMode == "ATTRIBUTES") {
420 useQMAttributes = true;
421 } else if (qmMode == "SINGLE") {
422 useQMAttributes = false;
423 } else {
424 throw OpalException(
425 "Option::execute",
426 "Unsupported QM_MODE '" + qmMode + "'. Use \"SINGLE\" or \"ATTRIBUTES\".");
427 }
428
429 aggressiveStateSync = Attributes::getBool(itsAttr[AGGRESSIVE_STATE_SYNC]);
430
433
434 if (Options::seed == -1)
435 rangen.init55(time(0));
436 else
437 rangen.init55(seed);
438
440 /*
441 IpplInfo::Info->on(info);
442 IpplInfo::Warn->on(warn);
443 */
444 handlePsDumpFrame(Attributes::getString(itsAttr[PSDUMPFRAME]));
445
448 if (itsAttr[SEED]) {
449 seed = int(Attributes::getReal(itsAttr[SEED]));
450 if (seed == -1)
451 rangen.init55(time(0));
452 else
453 rangen.init55(seed);
454 }
455
456 if (itsAttr[PSDUMPFREQ]) {
457 psDumpFreq = int(Attributes::getReal(itsAttr[PSDUMPFREQ]));
458 if (psDumpFreq == 0) psDumpFreq = std::numeric_limits<int>::max();
459 }
460
461 if (itsAttr[STATDUMPFREQ]) {
462 statDumpFreq = int(Attributes::getReal(itsAttr[STATDUMPFREQ]));
463 if (statDumpFreq == 0) statDumpFreq = std::numeric_limits<int>::max();
464 }
465
466 if (itsAttr[SPTDUMPFREQ]) {
467 sptDumpFreq = int(Attributes::getReal(itsAttr[SPTDUMPFREQ]));
468 if (sptDumpFreq == 0) sptDumpFreq = std::numeric_limits<int>::max();
469 }
470
471 if (itsAttr[SCSOLVEFREQ]) {
472 scSolveFreq = int(Attributes::getReal(itsAttr[SCSOLVEFREQ]));
473 scSolveFreq = (scSolveFreq < 1) ? 1 : scSolveFreq;
474 }
475
476 if (itsAttr[MTSSUBSTEPS]) {
477 mtsSubsteps = int(Attributes::getReal(itsAttr[MTSSUBSTEPS]));
478 }
479
480 if (itsAttr[REPARTFREQ]) {
481 repartFreq = int(Attributes::getReal(itsAttr[REPARTFREQ]));
482 }
483
484 if (itsAttr[MINBINEMITTED]) {
485 minBinEmitted = int(Attributes::getReal(itsAttr[MINBINEMITTED]));
486 }
487
488 if (itsAttr[MINSTEPFORREBIN]) {
489 minStepForRebin = int(Attributes::getReal(itsAttr[MINSTEPFORREBIN]));
490 }
491
492 if (itsAttr[REBINFREQ]) {
493 rebinFreq = int(Attributes::getReal(itsAttr[REBINFREQ]));
494 }
495
496 if (itsAttr[AUTOPHASE]) {
497 autoPhase = int(Attributes::getReal(itsAttr[AUTOPHASE]));
498 }
499
500 if (itsAttr[NUMBLOCKS]) {
501 numBlocks = int(Attributes::getReal(itsAttr[NUMBLOCKS]));
502 }
503
504 if (itsAttr[RECYCLEBLOCKS]) {
505 recycleBlocks = int(Attributes::getReal(itsAttr[RECYCLEBLOCKS]));
506 }
507
508 if (itsAttr[NLHS]) {
509 nLHS = int(Attributes::getReal(itsAttr[NLHS]));
510 }
511
512 if (itsAttr[BOUNDPDESTROY]) {
513 /*
514 * Historically, BOUNDPDESTROYFQ was used as a positive frequency and
515 * clamped to values >= 1 here. In OPAL-X, the same parameter now also
516 * encodes the N-sigma boundary used in ParticleContainer::
517 * deleteParticlesOutside(N). A value <= 0 is treated as "disabled"
518 * by deleteParticlesOutside, so we must *not* clamp it to 1 anymore.
519 */
520 boundpDestroy = Attributes::getReal(itsAttr[BOUNDPDESTROY]);
521 }
522
523 if (itsAttr[CZERO]) {
524 cZero = bool(Attributes::getBool(itsAttr[CZERO]));
525 }
526
527 if (itsAttr[RNGTYPE]) {
528 rngtype = std::string(Attributes::getString(itsAttr[RNGTYPE]));
529 } else {
530 rngtype = std::string("RANDOM");
531 }
532
533 if (itsAttr[BEAMHALOBOUNDARY]) {
534 beamHaloBoundary = Attributes::getReal(itsAttr[BEAMHALOBOUNDARY]);
535 } else {
536 beamHaloBoundary = 0;
537 }
538 if (itsAttr[CLOTUNEONLY]) {
539 cloTuneOnly = bool(Attributes::getBool(itsAttr[CLOTUNEONLY]));
540 } else {
541 cloTuneOnly = false;
542 }
543
544 // Set message flags.
545 FileStream::setEcho(echo);
546
547 if (Attributes::getBool(itsAttr[TELL])) {
548 *gmsg << "\nCurrent settings of options:\n" << *this << endl;
549 }
550
551 Option* main = dynamic_cast<Option*>(OpalData::getInstance()->find("OPTION"));
552 if (main) {
553 main->update(itsAttr);
554 }
555}
