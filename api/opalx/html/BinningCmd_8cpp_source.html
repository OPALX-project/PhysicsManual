91void BinningCmd::update() {
92 if (itsAttr[BINNING::PARAMETER]) {
93 parameterName_m = Attributes::getString(itsAttr[BINNING::PARAMETER]);
94 } else {
95 parameterName_m = "VELOCITYZ";
96 }
97
98 // Sanitize dump-to-file configuration if present.
99 if (itsAttr[BINNING::DUMPBINSFILE]) {
100 std::string filename = Attributes::getString(itsAttr[BINNING::DUMPBINSFILE]);
101
102 // Treat empty or "NONE" as "no dumping".
103 if (!filename.empty() && filename != "NONE") {
104 // Ensure the file name ends with ".json".
105 const std::string extension = ".json";
106 if (filename.size() < extension.size()
107 || filename.substr(filename.size() - extension.size()) != extension) {
108 filename += extension;
109 Attributes::setString(itsAttr[BINNING::DUMPBINSFILE], filename);
110 }
111
112 // Validate frequency when dumping is enabled.
113 int freq = static_cast<int>(Attributes::getReal(itsAttr[BINNING::DUMPBINSFREQ]));
114 if (freq < 1) {
115 throw OpalException(
116 "BinningCmd::update",
117 "DUMPBINSFREQ must be >= 1 when DUMPBINSFILE is set.");
118 }
119 }
120 }
121
122 // Validate console table print frequency (used in binned mode).
123 {
124 const int freq =
125 static_cast<int>(Attributes::getReal(itsAttr[BINNING::TABLEPRINTFREQ]));
126 if (freq < 0) {
127 throw OpalException("BinningCmd::update",
128 "TABLEPRINTFREQ must be >= 0.");
129 }
130 }
131}

133void BinningCmd::setParameterType() {
134 // Ensure we work from the current attribute value.
135 parameterName_m = getParameter();
136
137 static const std::map<std::string, BinningParameter> stringToParam = {
138 {"VELOCITYZ", BinningParameter::VELOCITYZ},
139 {"POSITIONZ", BinningParameter::POSITIONZ},
140 {"PZ", BinningParameter::PZ},
141 {"GAMMAZ", BinningParameter::GAMMAZ}
142 };
143
144 auto it = stringToParam.find(parameterName_m);
145 if (it == stringToParam.end()) {
146 throw OpalException("BinningCmd::setParameterType",
147 "Unknown binning PARAMETER \"" + parameterName_m + "\"");
148 }
149
150 parameterType_m = it->second;
151}
