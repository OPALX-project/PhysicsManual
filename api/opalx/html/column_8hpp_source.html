79 bool parse(const std::string& input, size_t& pos)
80 {
81 parser::ValueParser parser(input, pos);
82 switch(*this->type_m) {
83 case ast::FLOAT:
84 {
85 float f = 0.0f;
86 if (parser.parseFloat(f)) {
87 this->values_m.push_back(f);
88 pos = parser.getPosition();
89 return true;
90 }
91 break;
92 }
93 case ast::DOUBLE:
94 {
95 double d = 0.0;
96 if (parser.parseDouble(d)) {
97 this->values_m.push_back(d);
98 pos = parser.getPosition();
99 return true;
100 }
101 break;
102 }
103 case ast::SHORT:
104 {
105 short s = 0;
106 if (parser.parseShort(s)) {
107 this->values_m.push_back(s);
108 pos = parser.getPosition();
109 return true;
110 }
111 break;
112 }
113 case ast::LONG:
114 {
115 long l = 0;
116 if (parser.parseLong(l)) {
117 this->values_m.push_back(l);
118 pos = parser.getPosition();
119 return true;
120 }
121 break;
122 }
123 case ast::CHARACTER:
124 {
125 char c = '\0';
126 if (parser.parseChar(c)) {
127 this->values_m.push_back(c);
128 pos = parser.getPosition();
129 return true;
130 }
131 break;
132 }
133 case ast::STRING:
134 {
135 std::string s;
136 if (parser.parseQuotedString(s)) {
137 this->values_m.push_back(s);
138 pos = parser.getPosition();
139 return true;
140 }
141 break;
142 }
143 }
144 return false;
145 }
