329void LossDataSink::save(unsigned int numSets, OpalData::OpenMode openMode) {
330 if (outputName_m.empty()) return;
331 if (hasNoParticlesToDump()) return;
332
333 if (openMode == OpalData::OpenMode::UNDEFINED) {
334 openMode = OpalData::getInstance()->getOpenMode();
335 }
336
337 namespace fs = std::filesystem;
338 if (h5hut_mode_m) {
339 fileName_m = outputName_m + std::string(".h5");
340 if (openMode == OpalData::OpenMode::WRITE || !fs::exists(fileName_m)) {
341 openH5();
342 writeHeaderH5();
343 } else {
344 openH5(H5_O_APPENDONLY);
345 GET_NUM_STEPS();
346 }
347
348 for (unsigned int i = 0; i < numSets; ++i) {
349 saveH5(i);
350 }
351 CLOSE_FILE();
352 H5file_m = 0;
353
354 } else {
355 fileName_m = outputName_m + std::string(".loss");
356 if (openMode == OpalData::OpenMode::WRITE || !fs::exists(fileName_m)) {
357 openASCII();
358 writeHeaderASCII();
359 } else {
360 appendASCII();
361 }
362 saveASCII();
363 closeASCII();
364 }
365 *gmsg << level2 << "Save '" << fileName_m << "'" << endl;
366
367 ippl::Comm->barrier();
368
369 particles_m.clear();
370 turnNumber_m.clear();
371 bunchNumber_m.clear();
372 spos_m.clear();
373 refTime_m.clear();
374 RefPartR_m.clear();
375 RefPartP_m.clear();
376 globalTrackStep_m.clear();
377}
378
379// Note: This was changed to calculate the global number of dumped particles
380// because there are two cases to be considered:
381// 1. ALL nodes have 0 lost particles -> nothing to be done.
382// 2. Some nodes have 0 lost particles, some not -> H5 can handle that but all
383// nodes HAVE to participate, otherwise H5 waits endlessly for a response from
384// the nodes that didn't enter the saveH5 function. -DW
385bool LossDataSink::hasNoParticlesToDump() const {
386 size_t nLoc = particles_m.size();
387 ippl::Comm->reduce(nLoc, nLoc, 1, std::plus<size_t>());
388 return nLoc == 0;
389}

391bool LossDataSink::hasTurnInformations() const {
392 bool hasTurnInformation = !turnNumber_m.empty();
393
394 ippl::Comm->allreduce(hasTurnInformation, 1, std::logical_or<bool>());
395
396 return hasTurnInformation > 0;
397}

399void LossDataSink::saveH5(unsigned int setIdx) {
400 size_t nLoc = particles_m.size();
401 size_t startIdx = 0, endIdx = nLoc;
402 if (setIdx + 1 < startSet_m.size()) {
403 startIdx = startSet_m[setIdx];
404 endIdx = startSet_m[setIdx + 1];
405 nLoc = endIdx - startIdx;
406 }
407
408 std::unique_ptr<size_t[]> locN(new size_t[ippl::Comm->size()]);
409 std::unique_ptr<size_t[]> globN(new size_t[ippl::Comm->size()]);
410
411 for (int i = 0; i < ippl::Comm->size(); i++) {
412 globN[i] = locN[i] = 0;
413 }
414
415 locN[ippl::Comm->rank()] = nLoc;
416
417 reduce(locN.get(), globN.get(), ippl::Comm->size(), std::plus<size_t>());
418
419 DistributionMoments engine;
420 engine.compute(particles_m.begin() + startIdx, particles_m.begin() + endIdx);
421
423 SET_STEP();
424 SET_NUM_PARTICLES(nLoc);
425
426 if (setIdx < spos_m.size()) {
427 WRITE_STEPATTRIB_FLOAT64("SPOS", &(spos_m[setIdx]), 1);
428 WRITE_STEPATTRIB_FLOAT64("TIME", &(refTime_m[setIdx]), 1);
429 WRITE_STEPATTRIB_FLOAT64("RefPartR", (h5_float64_t*)&(RefPartR_m[setIdx]), 3);
430 WRITE_STEPATTRIB_FLOAT64("RefPartP", (h5_float64_t*)&(RefPartP_m[setIdx]), 3);
431 WRITE_STEPATTRIB_INT64("GlobalTrackStep", &(globalTrackStep_m[setIdx]), 1);
432 }
433
434 Vector_t<double, 3> tmpVector;
435 double tmpDouble;
436 WRITE_STEPATTRIB_FLOAT64("centroid", (tmpVector = engine.getMeanPosition(), &tmpVector[0]), 3);
437 WRITE_STEPATTRIB_FLOAT64(
438 "RMSX", (tmpVector = engine.getStandardDeviationPosition(), &tmpVector[0]), 3);
439 WRITE_STEPATTRIB_FLOAT64("MEANP", (tmpVector = engine.getMeanMomentum(), &tmpVector[0]), 3);
440 WRITE_STEPATTRIB_FLOAT64(
441 "RMSP", (tmpVector = engine.getStandardDeviationMomentum(), &tmpVector[0]), 3);
442 WRITE_STEPATTRIB_FLOAT64(
443 "#varepsilon", (tmpVector = engine.getNormalizedEmittance(), &tmpVector[0]), 3);
444 WRITE_STEPATTRIB_FLOAT64(
445 "#varepsilon-geom", (tmpVector = engine.getGeometricEmittance(), &tmpVector[0]), 3);
446 WRITE_STEPATTRIB_FLOAT64("ENERGY", (tmpDouble = engine.getMeanKineticEnergy(), &tmpDouble), 1);
447 WRITE_STEPATTRIB_FLOAT64("dE", (tmpDouble = engine.getStdKineticEnergy(), &tmpDouble), 1);
448 WRITE_STEPATTRIB_FLOAT64("TotalCharge", (tmpDouble = engine.getTotalCharge(), &tmpDouble), 1);
449 WRITE_STEPATTRIB_FLOAT64("TotalMass", (tmpDouble = engine.getTotalMass(), &tmpDouble), 1);
450 WRITE_STEPATTRIB_FLOAT64("meanTime", (tmpDouble = engine.getMeanTime(), &tmpDouble), 1);
451 WRITE_STEPATTRIB_FLOAT64("rmsTime", (tmpDouble = engine.getStdTime(), &tmpDouble), 1);
452 if (Options::computePercentiles) {
453 WRITE_STEPATTRIB_FLOAT64(
454 "68-percentile", (tmpVector = engine.get68Percentile(), &tmpVector[0]), 3);
455 WRITE_STEPATTRIB_FLOAT64(
456 "95-percentile", (tmpVector = engine.get95Percentile(), &tmpVector[0]), 3);
457 WRITE_STEPATTRIB_FLOAT64(
458 "99-percentile", (tmpVector = engine.get99Percentile(), &tmpVector[0]), 3);
459 WRITE_STEPATTRIB_FLOAT64(
460 "99_99-percentile", (tmpVector = engine.get99_99Percentile(), &tmpVector[0]), 3);
461
462 WRITE_STEPATTRIB_FLOAT64(
463 "normalizedEps68Percentile",
464 (tmpVector = engine.getNormalizedEmittance68Percentile(), &tmpVector[0]), 3);
465 WRITE_STEPATTRIB_FLOAT64(
466 "normalizedEps95Percentile",
467 (tmpVector = engine.getNormalizedEmittance95Percentile(), &tmpVector[0]), 3);
468 WRITE_STEPATTRIB_FLOAT64(
469 "normalizedEps99Percentile",
470 (tmpVector = engine.getNormalizedEmittance99Percentile(), &tmpVector[0]), 3);
471 WRITE_STEPATTRIB_FLOAT64(
472 "normalizedEps99_99Percentile",
473 (tmpVector = engine.getNormalizedEmittance99_99Percentile(), &tmpVector[0]), 3);
474 }
475
476 WRITE_STEPATTRIB_FLOAT64("maxR", (tmpVector = engine.getMaxR(), &tmpVector[0]), 3);
477
478 // Write all data
479 std::vector<char> buffer(nLoc * sizeof(h5_float64_t));
480 h5_float64_t* f64buffer = nLoc > 0 ? reinterpret_cast<h5_float64_t*>(&buffer[0]) : nullptr;
481 h5_int64_t* i64buffer = nLoc > 0 ? reinterpret_cast<h5_int64_t*>(&buffer[0]) : nullptr;
482
483 ::i64transform(particles_m, startIdx, nLoc, i64buffer, [](const OpalParticle& particle) {
484 return particle.getId();
485 });
486 WRITE_DATA_INT64("id", i64buffer);
487 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
488 return particle.getX();
489 });
490 WRITE_DATA_FLOAT64("x", f64buffer);
491 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
492 return particle.getY();
493 });
494 WRITE_DATA_FLOAT64("y", f64buffer);
495 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
496 return particle.getZ();
497 });
498 WRITE_DATA_FLOAT64("z", f64buffer);
499 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
500 return particle.getPx();
501 });
502 WRITE_DATA_FLOAT64("px", f64buffer);
503 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
504 return particle.getPy();
505 });
506 WRITE_DATA_FLOAT64("py", f64buffer);
507 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
508 return particle.getPz();
509 });
510 WRITE_DATA_FLOAT64("pz", f64buffer);
511 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
512 return particle.getCharge();
513 });
514 WRITE_DATA_FLOAT64("q", f64buffer);
515 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
516 return particle.getMass();
517 });
518 WRITE_DATA_FLOAT64("m", f64buffer);
519
520 if (hasTurnInformations()) {
521 std::copy(
522 turnNumber_m.begin() + startIdx, turnNumber_m.begin() + startIdx + nLoc, i64buffer);
523 WRITE_DATA_INT64("turn", i64buffer);
524
525 std::copy(
526 bunchNumber_m.begin() + startIdx, bunchNumber_m.begin() + startIdx + nLoc,
527 i64buffer);
528 WRITE_DATA_INT64("bunchNumber", i64buffer);
529 }
530
531 ::f64transform(particles_m, startIdx, nLoc, f64buffer, [](const OpalParticle& particle) {
532 return particle.getTime();
533 });
534 WRITE_DATA_FLOAT64("time", f64buffer);
535
536 ++H5call_m;
537}

539void LossDataSink::saveASCII() {
540 /*
541 ASCII output
542
543 int tag = Ippl::Comm->next_tag(IPPL_APP_TAG3, IPPL_APP_CYCLE);
544 bool hasTurn = hasTurnInformations();
545 if (ippl::Comm->rank == 0) {
546 const unsigned partCount = particles_m.size();
547
548 for (unsigned i = 0; i < partCount; i++) {
549 const OpalParticle& particle = particles_m[i];
550 os_m << particle.getX() << " ";
551 os_m << particle.getY() << " ";
552 os_m << particle.getZ() << " ";
553 os_m << particle.getPx() << " ";
554 os_m << particle.getPy() << " ";
555 os_m << particle.getPz() << " ";
556 os_m << particle.getId() << " ";
557 if (hasTurn) {
558 os_m << turnNumber_m[i] << " ";
559 os_m << bunchNumber_m[i] << " ";
560 }
561 os_m << particle.getTime() << std::endl;
562 }
563
564 int notReceived = ippl::Comm->size() - 1;
565 while (notReceived > 0) {
566 unsigned dataBlocks = 0;
567 int node = COMM_ANY_NODE;
568 Message* rmsg = Ippl::Comm->receive_block(node, tag);
569 if (rmsg == 0) {
570 *ippl::Error << "LossDataSink: Could not receive from client nodes output." << endl;
571 }
572 notReceived--;
573 rmsg->get(&dataBlocks);
574 rmsg->get(&hasTurn);
575 for (unsigned i = 0; i < dataBlocks; i++) {

642void LossDataSink::splitSets(unsigned int numSets) {
643 if (numSets <= 1 || particles_m.size() == 0) return;
644
645 const size_t nLoc = particles_m.size();
646 size_t avgNumPerSet = nLoc / numSets;
647 std::vector<size_t> numPartsInSet(numSets, avgNumPerSet);
648 for (unsigned int j = 0; j < (nLoc - numSets * avgNumPerSet); ++j) {
649 ++numPartsInSet[j];
650 }
651 startSet_m.resize(numSets + 1, 0);
652
653 std::vector<double> data(2 * numSets, 0.0);
654 double* meanT = &data[0];
655 double* numParticles = &data[numSets];
656 std::vector<double> timeRange(numSets, 0.0);
657 double maxT = particles_m[0].getTime();
658
659 for (unsigned int iteration = 0; iteration < 2; ++iteration) {
660 size_t partIdx = 0;
661 for (unsigned int j = 0; j < numSets; ++j) {
662 const size_t& numThisSet = numPartsInSet[j];
663 for (size_t k = 0; k < numThisSet; ++k, ++partIdx) {
664 meanT[j] += particles_m[partIdx].getTime();
665 maxT = std::max(maxT, particles_m[partIdx].getTime());
666 }
667 numParticles[j] = numThisSet;
668 }
669
670 ippl::Comm->allreduce(&(data[0]), 2 * numSets, std::plus<double>());
671
672 for (unsigned int j = 0; j < numSets; ++j) {
673 meanT[j] /= numParticles[j];
674 }
675
676 for (unsigned int j = 0; j < numSets - 1; ++j) {
677 timeRange[j] = 0.5 * (meanT[j] + meanT[j + 1]);
678 }
679 timeRange[numSets - 1] = maxT;
680
681 std::fill(numPartsInSet.begin(), numPartsInSet.end(), 0);
682
683 size_t setNum = 0;
684 size_t idxPrior = 0;
685 for (size_t idx = 0; idx < nLoc; ++idx) {
686 if (particles_m[idx].getTime() > timeRange[setNum]) {
687 numPartsInSet[setNum] = idx - idxPrior;
688 idxPrior = idx;
689 ++setNum;
690 }
691 }
692 numPartsInSet[numSets - 1] = nLoc - idxPrior;
693 }
694
695 for (unsigned int i = 0; i < numSets; ++i) {
696 startSet_m[i + 1] = startSet_m[i] + numPartsInSet[i];
697 }
698}

700SetStatistics LossDataSink::computeSetStatistics(unsigned int setIdx) {
701 SetStatistics stat;
702 double part[6];
703
704 const unsigned int totalSize = 45;
705 double plainData[totalSize];
706 double rminmax[6] = {};
707
708 Util::KahanAccumulation data[totalSize];
709 Util::KahanAccumulation* localCentroid = data + 1;
710 Util::KahanAccumulation* localMoments = data + 7;
711 Util::KahanAccumulation* localOthers = data + 43;
712
713 size_t startIdx = 0;
714 size_t nLoc = particles_m.size();
715 if (setIdx + 1 < startSet_m.size()) {
716 startIdx = startSet_m[setIdx];
717 nLoc = startSet_m[setIdx + 1] - startSet_m[setIdx];
718 }
719
720 data[0].sum = nLoc;
721
722 unsigned int idx = startIdx;
723 for (unsigned long k = 0; k < nLoc; ++k, ++idx) {
724 const OpalParticle& particle = particles_m[idx];
725
726 part[0] = particle.getX();
727 part[1] = particle.getPx();
728 part[2] = particle.getY();
729 part[3] = particle.getPy();
730 part[4] = particle.getZ();
731 part[5] = particle.getPz();
732
733 for (int i = 0; i < 6; i++) {
734 localCentroid[i] += part[i];
735 for (int j = 0; j <= i; j++) {
736 localMoments[i * 6 + j] += part[i] * part[j];
737 }
738 }
739 localOthers[0] += particle.getTime();
740 localOthers[1] += std::pow(particle.getTime(), 2);
741
742 ::cminmax(rminmax[0], rminmax[1], particle.getX());
743 ::cminmax(rminmax[2], rminmax[3], particle.getY());
744 ::cminmax(rminmax[4], rminmax[5], particle.getZ());
745 }
746
747 for (int i = 0; i < 6; i++) {
748 for (int j = 0; j < i; j++) {
749 localMoments[j * 6 + i] = localMoments[i * 6 + j];
750 }
751 }
752
753 for (unsigned int i = 0; i < totalSize; ++i) {
754 plainData[i] = data[i].sum;
755 }
756
757 ippl::Comm->allreduce(plainData, totalSize, std::plus<double>());
758 ippl::Comm->allreduce(rminmax, 6, std::greater<double>());
759
760 if (plainData[0] == 0.0) return stat;
761
762 double* centroid = plainData + 1;
763 double* moments = plainData + 7;
764 double* others = plainData + 43;
765
766 stat.outputName_m = outputName_m;
767 stat.spos_m = spos_m[setIdx];
768 stat.refTime_m = refTime_m[setIdx];
769 stat.RefPartR_m = RefPartR_m[setIdx];
770 stat.RefPartP_m = RefPartP_m[setIdx];
771 stat.nTotal_m = (unsigned long)std::round(plainData[0]);
772
773 for (unsigned int i = 0; i < 3u; i++) {
774 stat.rmean_m(i) = centroid[2 * i] / stat.nTotal_m;
775 stat.pmean_m(i) = centroid[(2 * i) + 1] / stat.nTotal_m;
776 stat.rsqsum_m(i) =
777 (moments[2 * i * 6 + 2 * i] - stat.nTotal_m * std::pow(stat.rmean_m(i), 2));
778 stat.psqsum_m(i) = std::max(
779 0.0, moments[(2 * i + 1) * 6 + (2 * i) + 1]
780 - stat.nTotal_m * std::pow(stat.pmean_m(i), 2));
781 stat.rpsum_m(i) =
782 (moments[(2 * i) * 6 + (2 * i) + 1]
783 - stat.nTotal_m * stat.rmean_m(i) * stat.pmean_m(i));
784 }
785 stat.tmean_m = others[0] / stat.nTotal_m;
786 stat.trms_m = std::sqrt(std::max(0.0, (others[1] / stat.nTotal_m - std::pow(stat.tmean_m, 2))));
787
788 stat.eps2_m =
789 ((stat.rsqsum_m * stat.psqsum_m - stat.rpsum_m * stat.rpsum_m)
790 / (1.0 * stat.nTotal_m * stat.nTotal_m));
791
792 stat.rpsum_m /= stat.nTotal_m;
793
794 for (unsigned int i = 0; i < 3u; i++) {
795 stat.rrms_m(i) = std::sqrt(std::max(0.0, stat.rsqsum_m(i)) / stat.nTotal_m);
796 stat.prms_m(i) = std::sqrt(std::max(0.0, stat.psqsum_m(i)) / stat.nTotal_m);
797 stat.eps_norm_m(i) = std::sqrt(std::max(0.0, stat.eps2_m(i)));
798 double tmp = stat.rrms_m(i) * stat.prms_m(i);
799 stat.fac_m(i) = (tmp == 0) ? 0.0 : 1.0 / tmp;
800 stat.rmin_m(i) = -rminmax[2 * i];
801 stat.rmax_m(i) = rminmax[2 * i + 1];
802 }
803
804 stat.rprms_m = stat.rpsum_m * stat.fac_m;
805
806 return stat;
807}
