128void OpalFilter::initOpalFilter() {
129 if (filter_m == 0) {
130 *gmsg << "* ************* F I L T E R "
131 "************************************************************"
132 << endl;
133 *gmsg << "OpalFilter::initOpalFilterfunction " << endl;
134 *gmsg << "* "
135 "*********************************************************************************"
136 "*"
137 << endl;
138
139 static const std::unordered_map<std::string, FilterType> stringFilterType_s = {
140 {"SAVITZKY-GOLAY", FilterType::SAVITZKYGOLAY},
141 {"FIXEDFFTLOWPASS", FilterType::FIXEDFFTLOWPASS},
142 {"RELATIVEFFTLOWPASS", FilterType::RELATIVEFFTLOWPASS},
143 {"STENCIL", FilterType::STENCIL}};
144
145 FilterType type = stringFilterType_s.at(Attributes::getString(itsAttr[TYPE]));
146 switch (type) {
147 case FilterType::SAVITZKYGOLAY: {
148 int num_points = (int)(Attributes::getReal(itsAttr[NPOINTS]));
149 int num_points_left = (int)(Attributes::getReal(itsAttr[NLEFT]));
150 int num_points_right = (int)(Attributes::getReal(itsAttr[NRIGHT]));
151 int polynomial_order = std::abs((int)(Attributes::getReal(itsAttr[POLYORDER])));
152
153 Inform svg("Savitzky-Golay: ");
154 if (num_points_left < 0) {
155 svg << "Number of points to the left negative; using default (" << NLEFT_DEFAULT
156 << ");" << endl;
157 num_points_left = NLEFT_DEFAULT;
158 }
159 if (num_points_right < 0) {
160 svg << "Number of points to the right negative; using default ("
161 << NRIGHT_DEFAULT << ");" << endl;
162 num_points_right = NRIGHT_DEFAULT;
163 }
164 if (num_points < num_points_left + num_points_right) {
165 svg << "Total number of points small than sum of the ones to the left and to "
166 "the right plus 1; using default (NLEFT + NRIGHT + 1);"
167 << endl;
168 num_points = num_points_left + num_points_right + 1;
169 }
170 if (polynomial_order > num_points_left + num_points_right) {
171 svg << "Polynomial order bigger than sum of points to the left and to the "
172 "right; using default (NLEFT + NRIGHT);"
173 << endl;
174 polynomial_order = num_points_left + num_points_right;
175 }
176
177 filter_m = new SavitzkyGolayFilter(
178 num_points, num_points_left, num_points_right, polynomial_order);
179 break;
180 }
181 case FilterType::FIXEDFFTLOWPASS: {
182 filter_m = new FixedFFTLowPassFilter(
183 std::abs((int)(Attributes::getReal(itsAttr[NFREQ]))));
184 break;
185 }
186 case FilterType::RELATIVEFFTLOWPASS: {
187 filter_m = new RelativeFFTLowPassFilter(
188 std::abs(Attributes::getReal(itsAttr[THRESHOLD])));
189 break;
190 }
191 case FilterType::STENCIL: {
192 filter_m = new StencilFilter();
193 break;
194 }
195 default: {
196 filter_m = 0;
197 *ippl::Info << "No filter attached" << endl;
198 break;
199 }
200 }
201 }
202}
