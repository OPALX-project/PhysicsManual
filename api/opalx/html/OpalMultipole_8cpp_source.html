74void OpalMultipole::update() {
75 OpalElement::update();
76
77 // Get pointer to MultipoleRep object to set length
78 MultipoleRep *mult = dynamic_cast<MultipoleRep *>(getElement());
79 double length = getLength();
80 mult->setElementLength(length);
81
82 // Multipole field for MultipoleRep
83 BMultipoleField field;
84
85 // Get the vector with the multipole expansion components
86 const std::vector<double> norm = Attributes::getRealArray(itsAttr[KN]);
87 std::vector<double> normErrors = Attributes::getRealArray(itsAttr[DKN]);
88 const std::vector<double> skew = Attributes::getRealArray(itsAttr[KS]);
89 std::vector<double> skewErrors = Attributes::getRealArray(itsAttr[DKS]);
90
91 // Resize error arrays
92 unsigned int normSize = norm.size();
93 unsigned int skewSize = skew.size();
94 normErrors.resize(normSize, 0.0);
95 skewErrors.resize(skewSize, 0.0);
96
97 double factor = OpalData::getInstance()->getP0() / Physics::c;
98 unsigned int top = (normSize > skewSize) ? normSize : skewSize;
99
100 // Loop over components (0=Dipole, 1=Quadrupole, ...)
101 for(unsigned int comp = 0; comp < top; ++comp) {
102 factor /= double(comp+1);
103 if(comp < normSize) {
104 field.setNormalComponent(comp, norm[comp] * factor);
105 mult->setNormalComponent(comp, norm[comp], normErrors[comp]);
106 }
107 if(comp < skewSize) {
108 field.setSkewComponent(comp, skew[comp] * factor);
109 mult->setSkewComponent(comp, skew[comp], skewErrors[comp]);
110 }
111 }
112
113 mult->setField(field);
114
115 // Transmit "unknown" attributes.
116 OpalElement::updateUnknown(mult);
117}
