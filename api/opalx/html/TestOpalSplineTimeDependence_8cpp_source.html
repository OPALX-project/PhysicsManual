35TEST(TestOpalSplineTimeDependence, UserInterface) {
36 // Make the UI
37 OpalSplineTimeDependence ui;
38 // Set the attributes
39 Attributes::setReal(ui.itsAttr[OpalSplineTimeDependence::ORDER], 3);
40 Attributes::setRealArray(ui.itsAttr[OpalSplineTimeDependence::TIMES], {1, 2, 3, 4});
41 Attributes::setRealArray(ui.itsAttr[OpalSplineTimeDependence::VALUES], {5, 6, 7, 8});
42 // Update the object
43 EXPECT_NO_THROW(ui.update());
44 // Check the values
