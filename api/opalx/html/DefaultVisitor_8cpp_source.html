44DefaultVisitor::DefaultVisitor(const Beamline& beamline, bool backBeam, bool backTrack)
45 : itsLine(beamline), back_beam(backBeam), back_track(backTrack) {
46 local_flip = back_path = (back_beam && !back_track) || (back_track && !back_beam);
47 flip_B = back_beam ? -1.0 : 1.0;
48 flip_s = back_path ? -1.0 : 1.0;
49}
50
51void DefaultVisitor::execute() {
52 local_flip = (back_beam && !back_track) || (back_track && !back_beam);
53 itsLine.accept(*this);
54}

116void DefaultVisitor::visitBeamline(const Beamline& bl) {
117 // Default behaviour: Apply algorithm to all beamline members.
118 // If flip_local is true, track from right to left.
119 bl.iterate(*this, local_flip);
120}

122void DefaultVisitor::visitFlaggedElmPtr(const FlaggedElmPtr& fep) {
123 if (fep.getReflectionFlag()) {
124 local_flip = !local_flip;
125 fep.getElement()->accept(*this);
126 local_flip = !local_flip;
127 } else {
128 fep.getElement()->accept(*this);
129 }
