30 // Cooley-Tukey FFT
31 void fft(std::vector<std::complex<double>>& x) {
32 size_t n = x.size();
33 if (n <= 1) return;
34
35 // Divide
36 std::vector<std::complex<double>> even(n / 2);
37 std::vector<std::complex<double>> odd(n / 2);
38 for (size_t i = 0; i < n / 2; ++i) {
39 even[i] = x[2 * i];
40 odd[i] = x[2 * i + 1];
41 }
42
43 // Conquer
44 fft(even);
45 fft(odd);
46
47 // Combine
48 for (size_t i = 0; i < n / 2; ++i) {
49 std::complex<double> t = std::polar(1.0, -2.0 * M_PI * i / n) * odd[i];
50 x[i] = even[i] + t;
51 x[i + n / 2] = even[i] - t;
52 }
53 }
54
55 // Inverse FFT
56 void ifft(std::vector<std::complex<double>>& x) {
57 size_t n = x.size();
58 if (n <= 1) return;
59
60 // Conjugate input
61 for (auto& val : x) {
62 val = std::conj(val);
63 }
64
65 // Forward FFT
66 fft(x);
67
68 // Conjugate and scale
69 for (auto& val : x) {
70 val = std::conj(val) / static_cast<double>(n);
71 }
72 }
73}

85 inline void fft_real_transform(double* data, size_t stride, size_t n) {
86 if (n == 0) return;
87
88 // Convert real data to complex
89 std::vector<std::complex<double>> complex_data(n);
90 for (size_t i = 0; i < n; ++i) {
91 complex_data[i] = std::complex<double>(data[i * stride], 0.0);
92 }
93
94 // Perform FFT
95 fft(complex_data);
96
97 // Convert back to real (packed format like GSL)
98 // GSL packs the result: [r0, r1, i1, r2, i2, ..., r(n/2), i(n/2)]
99 data[0] = complex_data[0].real();
100 if (n > 1) {
101 size_t half = n / 2;
102 for (size_t i = 1; i < half; ++i) {
103 data[2 * i - 1] = complex_data[i].real();
104 data[2 * i] = complex_data[i].imag();
105 }
106 if (n % 2 == 0) {
107 data[n - 1] = complex_data[half].real();
108 }
109 }
110 }
111}

207inline void gsl_fft_halfcomplex_transform(double* data, size_t stride, size_t n,
208 gsl_fft_halfcomplex_wavetable* /* wavetable */,
209 gsl_fft_halfcomplex_workspace* /* workspace */) {
210 // Inverse FFT for halfcomplex data (packed format)
211 // Convert halfcomplex to complex, perform inverse FFT, convert back
212 std::vector<std::complex<double>> complex_data(n);
213 complex_data[0] = std::complex<double>(data[0], 0.0);
214
215 size_t half = n / 2;
216 for (size_t i = 1; i < half; ++i) {
217 complex_data[i] = std::complex<double>(data[2 * i - 1], data[2 * i]);
218 complex_data[n - i] = std::conj(complex_data[i]);
219 }
220 if (n % 2 == 0) {
221 complex_data[half] = std::complex<double>(data[n - 1], 0.0);
222 }
223
224 // Inverse FFT (GSL convention: scale by n)
225 // Our ifft scales by 1/n, so we multiply by n to get n scaling
226 ifft(complex_data);
227 double scale = static_cast<double>(n);
228
229 // Convert back to real
230 for (size_t i = 0; i < n; ++i) {
231 data[i * stride] = complex_data[i].real() * scale;
232 }
233}

314inline void gsl_fft_complex_forward(double* data, size_t stride, size_t n,
315 gsl_fft_complex_wavetable* /* wavetable */,
316 gsl_fft_complex_workspace* /* workspace */) {
317 std::vector<std::complex<double>> complex_data(n);
318 for (size_t i = 0; i < n; ++i) {
319 complex_data[i] = std::complex<double>(data[2 * i * stride], data[(2 * i + 1) * stride]);
320 }
321 fft(complex_data);
322 for (size_t i = 0; i < n; ++i) {
323 data[2 * i * stride] = complex_data[i].real();
324 data[(2 * i + 1) * stride] = complex_data[i].imag();
325 }
326}

335inline void gsl_fft_complex_inverse(double* data, size_t stride, size_t n,
336 gsl_fft_complex_wavetable* /* wavetable */,
337 gsl_fft_complex_workspace* /* workspace */) {
338 std::vector<std::complex<double>> complex_data(n);
339 for (size_t i = 0; i < n; ++i) {
340 complex_data[i] = std::complex<double>(data[2 * i * stride], data[(2 * i + 1) * stride]);
341 }
342 // GSL convention: inverse FFT scales by n (not 1/n like standard FFT)
343 // Our ifft scales by 1/n, so we multiply by n to get n scaling
344 ifft(complex_data);
345 double scale = static_cast<double>(n);
346 for (size_t i = 0; i < n; ++i) {
347 data[2 * i * stride] = complex_data[i].real() * scale;
348 data[(2 * i + 1) * stride] = complex_data[i].imag() * scale;
349 }
350}
