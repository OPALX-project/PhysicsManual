379 void ThreeDGrid::xLowerBound(const double& x, int& xIndex) const {
380 if (constantSpacing_m)
381 xIndex = static_cast<int>(std::floor((x - x_m[0]) / (x_m[1] - x_m[0])));
382 else {
383 vectorLowerBound(x_m, x, xIndex);
384 }
385 }
