29#include <gtest/gtest.h>
30#include <vector>
31#include "Utilities/LinearSpline.h"

33class LinearSplineTest : public testing::Test {
34protected:
35 void SetUp() override {
36 // Test data: y = x^2
37 x_data = {0.0, 1.0, 2.0, 3.0, 4.0};
38 y_data = {0.0, 1.0, 4.0, 9.0, 16.0};
39 }
40
41 std::vector<double> x_data;
42 std::vector<double> y_data;
43};
44
45TEST_F(LinearSplineTest, BasicInterpolation) {
46 const LinearSpline spline(x_data, y_data);
47 LinearSpline::Accelerator acc;
48 // Test interpolation at data points
49 EXPECT_NEAR(spline.eval(0.0, acc), 0.0, 1e-10);
50 EXPECT_NEAR(spline.eval(1.0, acc), 1.0, 1e-10);
51 EXPECT_NEAR(spline.eval(2.0, acc), 4.0, 1e-10);
52 EXPECT_NEAR(spline.eval(3.0, acc), 9.0, 1e-10);
53 EXPECT_NEAR(spline.eval(4.0, acc), 16.0, 1e-10);
54
55 // Test interpolation between points
56 EXPECT_NEAR(spline.eval(1.5, acc), 2.5, 1e-10);
57}
58
59TEST_F(LinearSplineTest, Uninitialised) {
60 // Uninitialised
61 LinearSpline::Accelerator accel;
62 LinearSpline spline_uninit;
63 EXPECT_THROW(spline_uninit.eval(0.0, accel), std::runtime_error);
64}
65
66TEST_F(LinearSplineTest, Extrapolation) {
67 const LinearSpline spline(x_data, y_data);
68 LinearSpline::Accelerator acc;
69
70 // Test left extrapolation
71 EXPECT_NEAR(spline.eval(-1.0, acc), -1.0, 1e-10);
72 EXPECT_NEAR(spline.eval(-0.5, acc), -0.5, 1e-10);
73
74 // Test right extrapolation
75 EXPECT_NEAR(spline.eval(5.0, acc), 23.0, 1e-10);
76 EXPECT_NEAR(spline.eval(5.5, acc), 26.5, 1e-10);
77}
78
79TEST_F(LinearSplineTest, Accelerator) {
80 const LinearSpline spline(x_data, y_data);
81 LinearSpline::Accelerator acc;
82
83 // First evaluation
84 EXPECT_NEAR(spline.eval(1.5, acc), 2.5, 1e-10);
85
86 // Second evaluation in the same interval (should use cached index)
87 EXPECT_NEAR(spline.eval(1.5, acc), 2.5, 1e-10);
88
89 // Evaluation in a different interval
90 EXPECT_NEAR(spline.eval(2.5, acc), 6.5, 1e-10);
91}

93TEST_F(LinearSplineTest, InvalidInput) {
94 // Too few points
95 EXPECT_THROW(LinearSpline spline({0.0}, {0.0}), std::invalid_argument);
96 // Differing lengths

103TEST_F(LinearSplineTest, Integration) {
104 LinearSpline spline(x_data, y_data);
105 LinearSpline::Accelerator accel;
106
107 // Whole pre-computed intervals
108 EXPECT_NEAR( spline.evalIntegral(0, 1, accel), 0.5, 1e-10);
109 EXPECT_NEAR( spline.evalIntegral(1, 2, accel), 2.5, 1e-10);
110 EXPECT_NEAR( spline.evalIntegral(2, 3, accel), 6.5, 1e-10);
111 EXPECT_NEAR( spline.evalIntegral(3, 4, accel), 12.5, 1e-10);
112 EXPECT_NEAR( spline.evalIntegral(0, 4, accel), 22.0, 1e-10);
113
114 // Partial first and last intervals
115 EXPECT_NEAR( spline.evalIntegral(0.6, 3.2, accel), 11.26, 1e-10);
116
117 // Asking again should use the cache
118 EXPECT_NEAR( spline.evalIntegral(0.6, 3.2, accel), 11.26, 1e-10);
119
120 // Extrapolation below the first interval
121 EXPECT_NEAR( spline.evalIntegral(-1, 1.5, accel), 0.875, 1e-10);
122
123 // Extrapolation above the last interval
124 EXPECT_NEAR( spline.evalIntegral(3, 4.5, accel), 21.375, 1e-10);
125
126 // Swapping integral bounds
127 EXPECT_NEAR( spline.evalIntegral(2, 1, accel), 2.5, 1e-10);
128
129 // Uninitialised spline
130 LinearSpline spline_uninit;
131 EXPECT_THROW(spline_uninit.evalIntegral(2, 1, accel), std::runtime_error);
132
133 // Re-initialise the spline to something different
134 spline.init({0.0, 1.0, 2.0, 3.0, 4.0}, {0.0, 2.0, 4.0, 6.0, 8.0});
135 EXPECT_NEAR( spline.evalIntegral(0, 4, accel), 16.0, 1e-10);
136}
