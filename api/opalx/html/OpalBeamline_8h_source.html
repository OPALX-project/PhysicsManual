113template <class T>
114inline void OpalBeamline::visit(
115 const T& element, BeamlineVisitor&, PartBunch_t& bunch) {
116 Inform msg("OPAL ");
117 double startField = 0.0;
118 double endField = 0.0;
119 std::shared_ptr<T> elptr(dynamic_cast<T*>(element.clone()));
120
121 positionElementRelative(elptr);
122
123 if (elptr->isElementPositionSet())
124 startField = elptr->getElementPosition();
125
126 elptr->initialise(&bunch, startField, endField);
127 elements_m.push_back(BeamlineFieldElement(elptr, startField, endField));
128}
