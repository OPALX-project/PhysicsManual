317 void sync() {
318 IpplTimings::startTimer(bDeviceSyncronizationT);
319 if constexpr (UseDualView) {
320 if (histogram_m.need_sync_host() && histogram_m.need_sync_device()) {
321 std::cerr << "Warning: Histogram was modified on host AND device -- overwriting changes on host." << std::endl;
322 }
323 if (histogram_m.need_sync_host()) {
324 histogram_m.sync_host();
325 } else if (histogram_m.need_sync_device()) {
326 histogram_m.sync_device();
327 } // else do nothing
328 }
329 IpplTimings::stopTimer(bDeviceSyncronizationT);
330 }
