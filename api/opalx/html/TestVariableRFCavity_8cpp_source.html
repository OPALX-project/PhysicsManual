257TEST_F(TestVariableRFCavity, TestApplyField) {
258 VariableRFCavity cav1;
259 const std::shared_ptr<AbstractTimeDependence> poly1(new PolynomialTimeDependence({1.0, 2.0}));
260 const std::shared_ptr<AbstractTimeDependence> poly2(new PolynomialTimeDependence({3.0, 4.0}));
261 const std::shared_ptr<AbstractTimeDependence> poly3(new PolynomialTimeDependence({5.0, 6.0}));
262 cav1.setAmplitudeModel(poly1);
263 cav1.setFrequencyModel(poly2);
264 cav1.setPhaseModel(poly3);
265 cav1.setLength(2.);
266 cav1.setWidth(3.);
267 cav1.setHeight(4.);
268 const Vector_t<double, 3> R({1., 1., 1.});
269 for (double t = 0.0; t < 10.0e-9; t += 1.0e-9) {
270 Vector_t<double, 3> B({0., 0., 0.});
271 Vector_t<double, 3> E({0., 0., 0.});
272 const double phase = poly3->getValue(t);
273 const double amplitude = poly1->getValue(t);
274 const double integralF = poly2->getIntegral(t) * Units::MHz2Hz;
275 const double e_test =
276 amplitude * sin(Physics::two_pi * integralF + phase) * Units::MVpm2Vpm;
277 EXPECT_FALSE(cav1.apply(R, Vector_t<double, 3>(0.0), t, E, B));
278 EXPECT_NEAR(0., E[0], 1.e-6);
279 EXPECT_NEAR(0., E[1], 1.e-6);
280 EXPECT_NEAR(e_test, E[2], 1.e-6);
281 EXPECT_NEAR(0., B[0], 1.e-6);
282 EXPECT_NEAR(0., B[1], 1.e-6);
283 EXPECT_NEAR(0., B[2], 1.e-6);
284 }
285}

326TEST_F(TestVariableRFCavity, BunchFields) {
327 // Set up the cavity
328 const auto amplPoly = std::make_shared<PolynomialTimeDependence>(std::vector{1.0});
329 const auto freqPoly = std::make_shared<PolynomialTimeDependence>(std::vector{1.0});
330 const auto phasePoly =
331 std::make_shared<PolynomialTimeDependence>(std::vector{Physics::pi / 2.0});
332 setAmplitudeModel(amplPoly);
333 setFrequencyModel(freqPoly);
334 setPhaseModel(phasePoly);
335 constexpr double width = 1.0;
336 constexpr double length = 1.0;
337 setLength(length);
338 setHeight(2 * width);
339 setWidth(2 * width);
340 // Make the bunch
341 std::vector<double> line(11);
342 const auto bunch = makeBunch(line.size());
343 const auto pc = bunch->getParticleContainer();
344 // Create the local views and data
345 std::vector<Vector_t<double, 3>> localR(line.size());
346 const auto hostR = Kokkos::create_mirror_view(pc->R.getView());
347 const auto hostE = Kokkos::create_mirror_view(pc->E.getView());
348 // Set the particle positions
349 const double stepSize = width / static_cast<double>(line.size() - 1);
350 for (size_t i = 0; i < line.size(); ++i) {
351 localR[i] = {static_cast<double>(i) * stepSize - width / 2.0, 0.0, length / 2.0};
352 hostR(i) = {localR[i][0], localR[i][1], localR[i][2] + length / 2.0};
353 }
354 Kokkos::deep_copy(pc->R.getView(), hostR);
355 pc->setQ(pc->getChargePerParticle());
356 ippl::Comm->barrier();
357 Kokkos::fence();
358 // Register the bunch with the element
359 bunch->setT(0.0);
360 double startField, endField;
361 initialise(bunch.get(), startField, endField);
362 EXPECT_NE(RefPartBunch_m, nullptr);
363 // Get the fields for all particles
364 apply(pc);
365 // Extract the fields from the GPU
366 Kokkos::deep_copy(hostE, pc->E.getView());
367 Kokkos::fence();
368 for (size_t i = 0; i < line.size(); ++i) {
369 line[i] = std::hypot(hostE(i)[0], hostE(i)[1], hostE(i)[2]);
370 }
371 for (size_t i = 0; i < line.size(); ++i) {
372 EXPECT_DOUBLE_EQ(line[i], 1.0 * Units::MVpm2Vpm);
373 }
374 // Get the field for one of the particles
375 Vector_t<double, 3> singleE{}, singleB{};
376 EXPECT_FALSE(apply(0, 0.0, singleE, singleB));
377 EXPECT_DOUBLE_EQ(singleE[2], 1.0 * Units::MVpm2Vpm);
378 // Done
379 finalise();
380 EXPECT_EQ(RefPartBunch_m, nullptr);
381}
