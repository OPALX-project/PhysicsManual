40
41std::shared_ptr<Material> Material::addMaterial(const std::string& name,
42 std::shared_ptr<Material> mat_ptr) {
43 std::string nameUp = Util::toUpper(name);
44 if (protoTable_sm.find(nameUp) != protoTable_sm.end())
45 return protoTable_sm[nameUp];
46
47 protoTable_sm.insert(std::make_pair(nameUp, mat_ptr));
48
49 return mat_ptr;
50}
51
52std::shared_ptr<Material> Material::getMaterial(const std::string& name) {
53 std::string nameUp = Util::toUpper(name);
54 return protoTable_sm[nameUp];
55}
