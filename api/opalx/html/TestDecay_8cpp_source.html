45 std::shared_ptr<PC_t> makeContainer(ParticleType species = ParticleType::UNNAMED) {
46 ippl::Vector<int, 3> nr = 8;
47 ippl::Vector<double, 3> rmin = -4.0;
48 ippl::Vector<double, 3> rmax = 4.0;
49 ippl::Vector<double, 3> origin = rmin;
50 ippl::Vector<double, 3> hr = (rmax - rmin) / ippl::Vector<double, 3>(nr);
51 std::array<bool, 3> decomp = {true, true, true};
52
53 ippl::NDIndex<3> domain;
54 for (unsigned i = 0; i < 3; ++i) {
55 domain[i] = ippl::Index(nr[i]);
56 }
57
58 Mesh_t<3> mesh(domain, hr, origin);
59 FieldLayout_t<3> fl(MPI_COMM_WORLD, domain, decomp, true);
60 auto pc = std::make_shared<PC_t>(mesh, fl);
61 pc->Sp = static_cast<short>(species);
62 return pc;
63 }

269 auto electrons = makeContainer(ParticleType::ELECTRON);
270 muons->setM(Physics::m_mu);
271 electrons->setM(Physics::m_e);

275 MuonDecay decay(1.0e-12, 0, Physics::m_mu);

287 for (size_t i = 0; i < nPart; ++i) {

299 const double p2 = bg2 * Physics::m_e * Physics::m_e;

415 constexpr size_t nPart = 512;
416 auto pions = makeContainer();
417 auto muons = makeContainer(ParticleType::MUON);
418 pions->setM(Physics::m_pi);
419 muons->setM(Physics::m_mu);

432
433 // Expected fixed momentum in pion rest frame: p = (m_pi^2 - m_mu^2) / (2 m_pi)
434 const double pExpected = (Physics::m_pi * Physics::m_pi - Physics::m_mu * Physics::m_mu)

446 }
447
448 // Verify directions are not all identical (isotropic).
