89 inline std::vector<T> AUnary<T, U>::evaluate() const {
90 errno = 0;
91 std::vector<U> arg = opr->evaluate();
92 std::vector<T> result;
93
94 for (typename std::vector<U>::size_type i = 0; i < arg.size(); ++i) {
95 result.push_back((*fun.function)(arg[i]));
96 }
