575void ParallelTracker::computeSpaceChargeFields(unsigned long long step) {
576 Inform m("ParallelTracker::computeSpaceChargeFields");
577 // Current limitation: space-charge transform/scatter/gather is applied via the primary
578 // container path only. Keep this behavior until the dedicated multi-container SC refactor.
579 if (!itsBunch_m->hasFieldSolver()) {
580 /*
581 This should not happen, so when we do not have a field solve, we can
582 throw an exception. If we have "no solver" and want to run it, we would
583 choose the null solver.
584 */
585 *gmsg << level1 << "no solver available!" << endl;
586 throw OpalException(
587 "ParallelTracker::computeSpaceChargeFields",
588 "Bunch has no field solver assigned! If you want to run without "
589 "space charge effects, please use TYPE=NONE for the field solver.");
590 }
591
592 itsBunch_m->calcBeamParameters();
593 m << level4 << "Calculate beam parameters done." << endl;
594
595 // Use mean momentum for beam-frame alignment; with 0 or 1 particle get_pmean() can be
596 // zero or negligible (e.g. rank with no particles), which would make getQuaternion throw.
597 const double pmean_tol = 1e-12;
598 Vector_t<double, 3> pmean = itsBunch_m->getParticleContainer()->getMeanP();
599 double pmean_len2 = dot(pmean, pmean);
600 if (pmean_len2 < pmean_tol * pmean_tol) {
601 pmean = itsBunch_m->getParticleContainer()->getRefPartP();
602 pmean_len2 = dot(pmean, pmean);
603 }
604 if (pmean_len2 < pmean_tol * pmean_tol) {
605 pmean = Vector_t<double, 3>(0, 0, 1);
606 }
607 Quaternion alignment = getQuaternion(pmean, Vector_t<double, 3>(0, 0, 1));
608
609 CoordinateSystemTrafo beamToReferenceCSTrafo(
610 Vector_t<double, 3>(0, 0, itsBunch_m->getParticleContainer()->get_sPos()),
611 alignment.conjugate());
612
613 CoordinateSystemTrafo referenceToBeamCSTrafo = beamToReferenceCSTrafo.inverted();
614
615 // Transform particle positions to the beam frame.
616 referenceToBeamCSTrafo.transformBunchTo(
617 itsBunch_m->getParticleContainer()->R.getView(),
618 itsBunch_m->getParticleContainer()->getLocalNum());
619 m << level4 << "Transform particle positions to beam coordinate system done." << endl;
620 itsBunch_m->bunchUpdate();
621 m << level5 << "Bunch updated for positions in beam coordinate system." << endl;
622
623 // TODO: itsBunch_m->boundp() not implemented yet.
624 // itsBunch_m->boundp();
625
626 if (step % repartFreq_m + 1 == repartFreq_m) {
627 doBinaryRepartition();
628 m << level4 << "Binary repartition done." << endl;
629 }
630
631 // itsBunch_m->setGlobalMeanR(itsBunch_m->get_centroid());
632
633 itsBunch_m->computeSelfFields();
634 m << level3 << "Compute self fields done." << endl;
635
636 // Transform positions back to the reference frame.
637 const size_t nLocRef = itsBunch_m->getParticleContainer()->getLocalNum();
638 beamToReferenceCSTrafo.transformBunchTo(
639 itsBunch_m->getParticleContainer()->R.getView(), nLocRef);
640 m << level5 << "Transform particle positions back to reference coordinate system done." << endl;
641 // Rotate E and B back to the reference frame.
642 beamToReferenceCSTrafo.rotateBunchTo(itsBunch_m->getParticleContainer()->E.getView(), nLocRef);
643 m << level5 << "Rotate E fields back to reference coordinate system done." << endl;
644 beamToReferenceCSTrafo.rotateBunchTo(itsBunch_m->getParticleContainer()->B.getView(), nLocRef);
645 m << level5
646 << "Rotate B fields back to reference coordinate system done. ComputeSelfFields done."

648 // Rebuild mesh from reference-frame R (computeSelfFields used beam-frame bunchUpdate).
649 itsBunch_m->bunchUpdate();
650 m << level5 << "Bunch updated for positions in reference coordinate system." << endl;
651}

656void ParallelTracker::computeExternalFields(OrbitThreader& oth) {
657 IpplTimings::startTimer(fieldEvaluationTimer_m);
658 Inform msg("ParallelTracker ", *gmsg);
659
660 const size_t nContainers = itsBunch_m->getNumParticleContainers();
661 for (size_t ci = 0; ci < nContainers; ++ci) {
662 if (!itsBunch_m->isPcActive(ci)) {
663 continue;
664 }
665 auto pc = itsBunch_m->getParticleContainer(ci);
666 if (!pc) {
667 continue;
668 }
669
670 // Bunch bounds for this container.
671 Vector_t<double, 3> rmin(0.0), rmax(0.0);
672 if (pc->getTotalNum() > 0) {
673 pc->computeMinMaxR();
674 rmin = pc->getMinR();
675 rmax = pc->getMaxR();
676
677 // get_bounds returns cached mesh extents (rmin_m/rmax_m). These are
678 // sentinel values (DBL_MAX / DBL_MIN) when the mesh hasn't been
679 // computed yet — e.g. immediately after the first particles are emitted
680 // before calcBeamParameters() has run. Fall back to zero so the query
681 // uses pathLength_m as the centre with zero half-width.
682 if (!std::isfinite(rmin(2)) || !std::isfinite(rmax(2))) {
683 rmin = rmax = 0.0;
684 }
685 }
686
687 // Get elements at bunch position.
688 IndexMap::value_t elements;
689 try {
690 elements = oth.query(pc->get_sPos() + 0.5 * (rmax(2) + rmin(2)), rmax(2) - rmin(2));
691 } catch (IndexMap::OutOfBounds& e) {
692 globalEOL_m = true;
693 continue;
694 }
695
696 // Iterate over all elements.
697 IndexMap::value_t::const_iterator it = elements.begin();
698 const IndexMap::value_t::const_iterator end = elements.end();
699 for (; it != end; ++it) {
700 CoordinateSystemTrafo refToLocalCSTrafo =
701 (itsOpalBeamline_m.getMisalignment((*it))
702 * (itsOpalBeamline_m.getCSTrafoLab2Local((*it)) * pc->getToLabTrafo()));
703
704 CoordinateSystemTrafo localToRefCSTrafo = refToLocalCSTrafo.inverted();
705
706 (*it)->setCurrentSCoordinate(pc->get_sPos() + rmin(2));
707
708 pc->transformBunch(refToLocalCSTrafo);
709
710 // Apply element to this iteration's particle container.
711 (*it)->apply(pc);
712
713 pc->transformBunch(localToRefCSTrafo);
714 }
715 }
716
717 IpplTimings::stopTimer(fieldEvaluationTimer_m);
718}

723void ParallelTracker::emitFromEmissionSources(double t, double dt) {
724 const auto& containers = itsBunch_m->getParticleContainers();
725 for (size_t ci = 0; ci < containers.size(); ++ci) {
726 const auto& pc = containers[ci];
727 if (!pc) {
728 continue;
729 }
730 if (itsBunch_m->pcAtZStop(ci)) {
731 continue;
732 }
733
734 // Record the extent of the position array and the current local particle count
735 // before emission. If an internal resize (Kokkos::realloc) happens during
736 // emission, the extent of R will change and we can flag this as an error.
737 const size_t extentBeforeEmission = pc->R.getView().extent(0);
738
739 CoordinateSystemTrafo refToGun =
740 itsOpalBeamline_m.getCSTrafoLab2Local() * pc->getToLabTrafo();
741 pc->transformBunch(refToGun);
742 if (ci < emittingSamplers_m.size()) {
743 for (const auto& sampler : emittingSamplers_m[ci]) {
744 if (sampler) {
745 sampler->emitParticles(t, dt);
746 }
747 }
748 }
749 pc->transformBunch(refToGun.inverted());
750
751 pc->setM(pc->getMassPerParticle());
752 pc->setQ(pc->getChargePerParticle());
753 // itsBunch_m->updateNumTotal(); // handled internally by ippl
754 // itsBunch_m->bunchUpdate();
755
756 // Sanity guard: the total number of macroparticles in the bunch must
757 // never exceed the globally configured BEAM::NALLOC value. Overshooting
758 // this limit would trigger internal reallocations in the particle
759 // container and silently drop already-tracked particles/delete their data in the particle
760 // attributes. This is only a check for the number of local particles.
761 const size_t extentAfterEmission = pc->R.getView().extent(0);
762 if (extentAfterEmission != extentBeforeEmission) {
763 throw OpalException(
764 "ParallelTracker::emitFromEmissionSources",
765 "Local particle storage was resized during emission (likely due to "
766 "over-emission causing a Kokkos::realloc). This leads to loss of "
767 "previously tracked particles. Please increase the total number of "
768 "macroparticles or adjust NPARTDIST / the emission profile. If you are "
769 "using emission sources, please check the emission profile and adjust "
770 "the number of particles emitted.");
771 }
772 }
773 itsBunch_m->refreshPcActiveAfterEmit();
774}

820void ParallelTracker::pushParticles(
821 const BorisPusher& pusher, PartBunch_t::ParticleContainer_t& pc) {
822 // Per-particle dt is used so that newly emitted particles with fractional dt
823 // (sampled during emission) are pushed proportionally to their sub-timestep fraction.
824 pc.switchToUnitlessPositions();
825
826 auto Rview = pc.R.getView();
827 auto Pview = pc.P.getView();
828 // auto dtview = pc.dt.getView();
829
830 Kokkos::parallel_for(
831 "pushParticles", pc.getLocalNum(), KOKKOS_LAMBDA(const size_t i) {
832 // Half drift: x_{n+1/2} = x_n + (dt/2) * v; unitless form via pusher.push(..., 0).
833 // TODO: verify sign convention for half-step push.
834 Vector_t<double, 3> x = Rview(i);
835 pusher.push(
836 x, Pview(i),
837 0); // this 0 is "dt" that is not used with unitless positions!
838 Rview(i) = x;
839 });
840
841 pc.switchOffUnitlessPositions();
842 // TODO: pc->update() changes results on single rank; keep disabled until investigated.
843 // itsBunch_m->getParticleContainer()->update();
844 Kokkos::fence();
845 ippl::Comm->barrier();
846 pc.markMomentsDirty();
847 // itsBunch_m->bunchUpdate();

855void ParallelTracker::kickParticles(
856 const BorisPusher& pusher, PartBunch_t::ParticleContainer_t& pc) {
857 Inform m("ParallelTracker::kickParticles");
858
859 // auto Rview = pc.R.getView();
860 auto Pview = pc.P.getView();
861 auto dtview = pc.dt.getView();
862 auto Efview = pc.E.getView();
863 auto Bfview = pc.B.getView();
864 m << level5 << "Got particle views for kick operation." << endl;
865 // Mass (eV) and charge (proton charges) from this container's reference particle,
866 // passed explicitly into BorisPusher::kick for GPU-safe kernels.
867
868 const PartData& ref = *pc.getReference();
869 const double mass = ref.getM();
870 const double charge = ref.getQ();
871 Kokkos::parallel_for(
872 "kickParticles", pc.getLocalNum(), KOKKOS_LAMBDA(const size_t i) {
873 // Boris kick: Birdsall & Langdon (1985), ch. 4-4 (non-relativistic) and ch. 15-4
874 // (relativistic); scaled P = (v/c)*gamma, mass in rest energy units.
875 Vector_t<double, 3> p = Pview(i);
876 // TODO: consider dropping unused R/dt arguments from kick when API allows.
877 pusher.kick(0, p, Efview(i), Bfview(i), dtview(i), mass, charge);
878 Pview(i) = p;
879 });
880 /*
881 Wait until everyone completed the kick operation before proceeding. For now,
882 this is just a precaution and could be removed for a small performance gain
883 (technically, these shouldn't be necessary!).
884 */
885 Kokkos::fence();
886 ippl::Comm->barrier();
887 pc.markMomentsDirty();
888
889 m << level5 << "Completed parallel kick operation." << endl;
890}

974 const auto& particleContainers = itsBunch_m->getParticleContainers();
975 bool hasNonEmpty = false;

977

979 if (!pc || pc->getTotalNum() == 0) {
980 continue;

982 pc->computeMinMaxR();

984 const ippl::Vector<double, 3> mx = pc->getMaxR();
985 if (!hasNonEmpty) {
986 rminLoc = mn;
987 rmaxLoc = mx;
988 hasNonEmpty = true;
989 } else {
990 for (int i = 0; i < 3; ++i) {
991 rminLoc[i] = std::min(rminLoc[i], mn[i]);
992 rmaxLoc[i] = std::max(rmaxLoc[i], mx[i]);
993 }
994 }
995 }
996
997 if (!hasNonEmpty) {
998 if (particleContainers.empty() || !particleContainers[0]) {
999 throw OpalException(
1000 "ParallelTracker::computeInitialBounds",
1001 "No valid particle container for initial bounds.");
1002 }
1003 particleContainers[0]->computeMinMaxR();
1004 rminLoc = particleContainers[0]->getMinR();
1005 rmaxLoc = particleContainers[0]->getMaxR();
1006 }
1007
1008 rmax = rmaxLoc;
1009 rmin = rminLoc;
1010 ippl::Comm->allreduce(rmax, 1, std::greater<ippl::Vector<double, 3>>());
1011 ippl::Comm->allreduce(rmin, 1, std::less<ippl::Vector<double, 3>>());
1012 ippl::Comm->barrier();

1053void ParallelTracker::updateReferenceParticles(const BorisPusher& pusher) {
1054 const double dt = std::min(itsBunch_m->getT(), itsBunch_m->getdT());
1055 const double scaleFactor = Physics::c * dt;
1056
1057 const size_t n = itsBunch_m->getNumParticleContainers();
1058 for (size_t i = 0; i < n; ++i) {
1059 if (!itsBunch_m->isPcActive(i)) {
1060 continue;
1061 }
1062 auto pcPtr = itsBunch_m->getParticleContainer(i);
1063 if (!pcPtr) {
1064 continue;
1065 }
1066 auto& pc = *pcPtr;
1067 const PartData& refKick = *pc.getReference();
1068 Vector_t<double, 3> Ef(0.0), Bf(0.0);
1069
1070 pc.getRefPartR() /= scaleFactor;
1071 pusher.push(pc.getRefPartR(), pc.getRefPartP(), dt);
1072 pc.getRefPartR() *= scaleFactor;
1073
1074 IndexMap::value_t elements = itsOpalBeamline_m.getElements(pc.getRefPartR());
1075 IndexMap::value_t::const_iterator it = elements.begin();
1076 const IndexMap::value_t::const_iterator end = elements.end();
1077
1078 for (; it != end; ++it) {
1079 const CoordinateSystemTrafo& refToLocalCSTrafo =
1080 itsOpalBeamline_m.getCSTrafoLab2Local((*it));
1081
1082 Vector_t<double, 3> localR = refToLocalCSTrafo.transformTo(pc.getRefPartR());
1083 Vector_t<double, 3> localP = refToLocalCSTrafo.rotateTo(pc.getRefPartP());
1084 Vector_t<double, 3> localE(0.0), localB(0.0);
1085
1086 if ((*it)->applyToReferenceParticle(
1087 localR, localP, itsBunch_m->getT() - 0.5 * dt, localE, localB)) {
1088 *gmsg << level1 << "The reference particle hit an element" << endl;
1089 globalEOL_m = true;
1090 }
1091
1092 Ef += refToLocalCSTrafo.rotateFrom(localE);
1093 Bf += refToLocalCSTrafo.rotateFrom(localB);
1094 }
1095
1096 pusher.kick(pc.getRefPartR(), pc.getRefPartP(), Ef, Bf, dt, refKick.getM(), refKick.getQ());
1097
1098 pc.getRefPartR() /= scaleFactor;
1099 pusher.push(pc.getRefPartR(), pc.getRefPartP(), dt);
1100 pc.getRefPartR() *= scaleFactor;
1101 }

1126void ParallelTracker::findStartPositions(const BorisPusher& pusher) {
1127 // Primary container (index 0) drives segment advances.
1128 auto primary = itsBunch_m->getParticleContainer(0);
1129
1130 if (zstart_m <= primary->get_sPos()) {
1131 return;
1132 }
1133
1134 StepSizeConfig stepSizesCopy(stepSizes_m);
1135
1136 double t = 0.0;
1137 itsBunch_m->setT(t);
1138
1139 dtCurrentTrack_m = stepSizesCopy.getdT();
1140 selectDT();
1141
1142 const auto& containers = itsBunch_m->getParticleContainers();
1143
1144 while (true) {
1145 autophaseCavities(pusher);
1146
1147 t += itsBunch_m->getdT();
1148 itsBunch_m->setT(t);
1149
1150 std::vector<Vector_t<double, 3>> oldRs(containers.size());
1151 for (size_t i = 0; i < containers.size(); ++i) {
1152 if (containers[i]) {
1153 oldRs[i] = containers[i]->getRefPartR();
1154 }
1155 }
1156
1157 updateReferenceParticles(pusher);
1158
1159 for (size_t i = 0; i < containers.size(); ++i) {
1160 if (!containers[i]) {
1161 continue;
1162 }
1163 Vector_t<double, 3> dR = containers[i]->getRefPartR() - oldRs[i];
1164 containers[i]->set_sPos(containers[i]->get_sPos() + euclidean_norm(dR));
1165 }
1166
1167 Vector_t<double, 3> tmp =
1168 primary->getRefPartP() * Physics::c / Util::getGamma(primary->getRefPartP());
1169 double speed = euclidean_norm(tmp);
1170
1171 if (primary->get_sPos() > stepSizesCopy.getZStop()) {
1172 ++stepSizesCopy;
1173
1174 if (stepSizesCopy.reachedEnd()) {
1175 --stepSizesCopy;
1176 const double zTarget = stepSizesCopy.getZStop();
1177 const double tauPrimary = (zTarget - primary->get_sPos()) / speed;
1178 itsBunch_m->setT(itsBunch_m->getT() + tauPrimary);
1179 for (const auto& pc : containers) {
1180 if (!pc) {
1181 continue;
1182 }
1183 Vector_t<double, 3> pv =
1184 pc->getRefPartP() * Physics::c / Util::getGamma(pc->getRefPartP());
1185 double speed_i = euclidean_norm(pv);
1186 double tau_i = (zTarget - pc->get_sPos()) / speed_i;
1187 pc->applyFractionalStep(pusher, tau_i, zstart_m);
1188 }
1189
1190 break;
1191 }
1192
1193 dtCurrentTrack_m = stepSizesCopy.getdT();
1194 selectDT();
1195 }
1196
1197 if (std::abs(primary->get_sPos() - zstart_m) <= 0.5 * itsBunch_m->getdT() * speed) {
1198 const double zTarget = zstart_m;
1199 const double tauPrimary = (zTarget - primary->get_sPos()) / speed;
1200 itsBunch_m->setT(itsBunch_m->getT() + tauPrimary);
1201 for (const auto& pc : containers) {
1202 if (!pc) {
1203 continue;
1204 }
1205 Vector_t<double, 3> pv =
1206 pc->getRefPartP() * Physics::c / Util::getGamma(pc->getRefPartP());
1207 double speed_i = euclidean_norm(pv);
1208 double tau_i = (zTarget - pc->get_sPos()) / speed_i;
1209 pc->applyFractionalStep(pusher, tau_i, zstart_m);
1210 }
1211
1212 break;
1213 }
1214 }
1215
1216 changeDT();
1217}
