185void Gaussian::emitParticles(double t, double dt) {
186 // One-shot delayed emission for GAUSS: emit once when [t, t+dt] crosses t0_m.
187 const double tStart = t;
188 const double tEnd = t + dt;
189
190 if (hasEmittedOnce_m) {
191 return;
192 }
193
194 // Only meaningful for t0 > 0.
195 if (t0_m <= 0.0) {
196 return;
197 }
198
199 // Fire when the time interval [tStart, tEnd] crosses t0_m.
200 if (!(tStart <= t0_m && t0_m < tEnd)) {
201 return;
202 }
203
204 if (!opalDist_m) {
205 return;
206 }
207 size_t Ndist = opalDist_m->getNumParticles();
208 if (Ndist == 0) {
209 return;
210 }
211
212 const size_t nlocalBefore = pc_m->getLocalNum();
213
214 // Mark as emitted so generateParticles will not early-return.
215 hasEmittedOnce_m = true;
216 Vector_t<double, 3> dummyNr(0.0);
217 generateParticles(Ndist, dummyNr);
218
219 // Set per-particle dt for newly created particles. Note that t0 can be in
220 // between particles, so we set the remaining fraction of the current time
221 // step for the new particles to ensure correct sub-stepping in the
222 // subsequent push.
223 //
224 // switchToUnitlessPositions(use_dt_per_particle=true) in pushParticles scales
225 // R by 1/(c * dtview(i)). New particles come with dtview = 0 (Kokkos
226 // zero-init), so that division produces inf, and the subsequent multiply by
227 // c*0 in switchOffUnitlessPositions gives inf*0 = NaN positions.
228 // Assign the remaining fraction of the current timestep so the push in
229 // timeIntegration2 moves them by the correct sub-step distance.
230 //
231 // Note that this is not necessary inside generateParticles, since for injection at start,
232 // setTime is called before any calculations inside ParallelTracker::execute.
233 const size_t nlocalAfter = pc_m->getLocalNum();
234 const size_t nNew = nlocalAfter - nlocalBefore;
235 if (nNew > 0) {
236 const double fracDt = tEnd - t0_m;
237 auto dtview = pc_m->dt.getView();
238 const size_t offset = nlocalBefore;
239 Kokkos::parallel_for(
240 "Gaussian_setDt", nNew,
241 KOKKOS_LAMBDA(const size_t j) { dtview(offset + j) = fracDt; });
242 Kokkos::fence();
243 }
