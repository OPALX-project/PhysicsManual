53 Iterator get_pos(Iterator err_pos, int& line) const {
54 line = 1;
55 Iterator i = first;
56 Iterator line_start = first;
57 while (i != err_pos) {
58 bool eol = false;
59 if (i != err_pos && *i == '\r') // CR
60 {
61 eol = true;
62 line_start = ++i;
63 }
64 if (i != err_pos && *i == '\n') // LF
65 {
66 eol = true;
67 line_start = ++i;
68 }
69 if (eol)
70 ++line;
71 else
72 ++i;
73 }
74 return line_start;
75 }
