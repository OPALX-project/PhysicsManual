20 /* Kinematics of mu -> e + nu_e + nu_mu (three-body decay).
21 *
22 * The two neutrinos are unobserved, so the electron energy in the muon
23 * rest frame is not fixed but follows the Michel spectrum:
24 * f(x) = 2 x^2 (3 - 2x), x = E_e / E_max, E_max = m_mu / 2.
25 *
26 * Procedure per daughter:
27 * 1. Sample x from the Michel spectrum via rejection sampling.
28 * 2. Compute daughter energy E = x * E_max and momentum |p| = sqrt(E^2 - m_e^2).
29 * 3. Pick an isotropic direction in the parent rest frame.
30 * 4. Lorentz-boost the rest-frame 4-momentum to the lab frame using the
31 * parent's beta*gamma vector.
32 * 5. Store the result as beta*gamma = p_lab / m_daughter. */
33 const double eMax = parentMassGeV_m / 2.0;
34 const double xMin = daughterMassGeV_m / eMax;
35 const double fMax = 2.0; // Michel spectrum upper bound at x = 1
36
37 Kokkos::parallel_for(

40 auto gen = pool.get_state();

42 /* Step 1: Sample reduced energy x from Michel spectrum (rejection). */
43 double x;
44 do {
45 x = xMin + (1.0 - xMin) * gen.drand(0.0, 1.0);
46 } while (gen.drand(0.0, fMax) >= 2.0 * x * x * (3.0 - 2.0 * x));
47
48 /* Step 2: Daughter energy and momentum magnitude in rest frame. */
49 const double eDaughter = x * eMax; // [GeV]
50 const double pMag = Kokkos::sqrt(
51 eDaughter * eDaughter - daughterMass * daughterMass); // [GeV/c]
52
53 /* Step 3: Isotropic direction in parent rest frame. */
54 const double cosTheta = 2.0 * gen.drand(0.0, 1.0) - 1.0;
55 const double sinTheta = Kokkos::sqrt(1.0 - cosTheta * cosTheta);
56 const double phi = 2.0 * Physics::pi * gen.drand(0.0, 1.0);
57 const double pxRF = pMag * sinTheta * Kokkos::cos(phi);
58 const double pyRF = pMag * sinTheta * Kokkos::sin(phi);
59 const double pzRF = pMag * cosTheta;

61 /* Step 4: Lorentz boost to lab frame.
62 * Parent momentum is stored as beta*gamma (dimensionless).
63 * beta = P / gamma, with gamma = sqrt(1 + |P|^2).
64 * The boost formula for the spatial components is:
65 * p_lab = p_RF + [(gamma-1)(beta . p_RF)/beta^2 + gamma*E] * beta */
66 const auto& parentMom = parentP(j);
67 const double bg2 = parentMom[0] * parentMom[0] + parentMom[1] * parentMom[1]
68 + parentMom[2] * parentMom[2];
69 const double gamma = Kokkos::sqrt(1.0 + bg2);
70 const double betaX = parentMom[0] / gamma;
71 const double betaY = parentMom[1] / gamma;
72 const double betaZ = parentMom[2] / gamma;
73 const double beta2 = bg2 / (gamma * gamma);
74
75 double pxLab, pyLab, pzLab;
76 if (beta2 > 0.0) {
77 const double betaDotP = betaX * pxRF + betaY * pyRF + betaZ * pzRF;
78 const double factor = (gamma - 1.0) * betaDotP / beta2 + gamma * eDaughter;
79 pxLab = pxRF + factor * betaX;
80 pyLab = pyRF + factor * betaY;
81 pzLab = pzRF + factor * betaZ;
82 } else {

84 pyLab = pyRF;
85 pzLab = pzRF;
86 }

90 const pc_size_type idx = static_cast<pc_size_type>(oldDaughterLocal) + j;
91 dR(idx) = parentR(j);
92 dP(idx)[0] = pxLab / daughterMass;
93 dP(idx)[1] = pyLab / daughterMass;
