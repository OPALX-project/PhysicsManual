48inline void gsl_blas_dgemm(CBLAS_TRANSPOSE TransA, CBLAS_TRANSPOSE TransB,
49 double alpha, const gsl_matrix* A, const gsl_matrix* B,
50 double beta, gsl_matrix* C) {
51 size_t m = (TransA == CblasNoTrans) ? A->size1 : A->size2;
52 size_t n = (TransB == CblasNoTrans) ? B->size2 : B->size1;
53 size_t k = (TransA == CblasNoTrans) ? A->size2 : A->size1;
54
55 if (C->size1 != m || C->size2 != n) {
56 throw std::runtime_error("gsl_blas_dgemm: size mismatch");
57 }
58
59 // Initialize C with beta*C
60 if (beta != 1.0) {
61 for (size_t i = 0; i < m; ++i) {
62 for (size_t j = 0; j < n; ++j) {
63 *gsl_matrix_ptr(C, i, j) *= beta;
64 }
65 }
66 }
67
68 // Perform multiplication
69 for (size_t i = 0; i < m; ++i) {
70 for (size_t j = 0; j < n; ++j) {
71 double sum = 0.0;
72 for (size_t l = 0; l < k; ++l) {
73 double a_val = (TransA == CblasNoTrans) ?
74 *gsl_matrix_ptr(A, i, l) : *gsl_matrix_ptr(A, l, i);
75 double b_val = (TransB == CblasNoTrans) ?
76 *gsl_matrix_ptr(B, l, j) : *gsl_matrix_ptr(B, j, l);
77 sum += a_val * b_val;
78 }
79 *gsl_matrix_ptr(C, i, j) += alpha * sum;
80 }
81 }
82}

94inline void gsl_blas_zgemm(CBLAS_TRANSPOSE TransA, CBLAS_TRANSPOSE TransB,
95 gsl_complex alpha, const gsl_matrix_complex* A, const gsl_matrix_complex* B,
96 gsl_complex beta, gsl_matrix_complex* C) {
97 size_t m = (TransA == CblasNoTrans) ? A->size1 : A->size2;
98 size_t n = (TransB == CblasNoTrans) ? B->size2 : B->size1;
99 size_t k = (TransA == CblasNoTrans) ? A->size2 : A->size1;
100
101 if (C->size1 != m || C->size2 != n) {
102 throw std::runtime_error("gsl_blas_zgemm: size mismatch");
103 }
104
105 // Initialize C with beta*C
106 if (beta.dat[0] != 1.0 || beta.dat[1] != 0.0) {
107 for (size_t i = 0; i < m; ++i) {
108 for (size_t j = 0; j < n; ++j) {
109 *gsl_matrix_complex_ptr(C, i, j) = gsl_complex_mul(*gsl_matrix_complex_ptr(C, i, j), beta);
110 }
111 }
112 }
113
114 // Perform multiplication
115 for (size_t i = 0; i < m; ++i) {
116 for (size_t j = 0; j < n; ++j) {
117 gsl_complex sum = gsl_complex(0.0, 0.0);
118 for (size_t l = 0; l < k; ++l) {
119 gsl_complex a_val = (TransA == CblasNoTrans) ?
120 *gsl_matrix_complex_ptr(A, i, l) : *gsl_matrix_complex_ptr(A, l, i);
121 if (TransA == CblasConjTrans) {
122 a_val = gsl_complex_conjugate(a_val);
123 }
124 gsl_complex b_val = (TransB == CblasNoTrans) ?
125 *gsl_matrix_complex_ptr(B, l, j) : *gsl_matrix_complex_ptr(B, j, l);
126 if (TransB == CblasConjTrans) {
127 b_val = gsl_complex_conjugate(b_val);
128 }
129 sum = gsl_complex_add(sum, gsl_complex_mul(a_val, b_val));
130 }
131 *gsl_matrix_complex_ptr(C, i, j) = gsl_complex_add(*gsl_matrix_complex_ptr(C, i, j),
132 gsl_complex_mul(alpha, sum));
133 }
134 }

146inline void gsl_blas_dgemv(CBLAS_TRANSPOSE TransA,
147 double alpha, const gsl_matrix* A, const gsl_vector* x,
148 double beta, gsl_vector* y) {
149 size_t m = A->size1;
150 size_t n = A->size2;
151
152 if (TransA == CblasNoTrans) {
153 if (x->size != n || y->size != m) {
154 throw std::runtime_error("gsl_blas_dgemv: size mismatch");
155 }
156 // y = beta*y
157 for (size_t i = 0; i < m; ++i) {
158 *gsl_vector_ptr(y, i) *= beta;
159 }
160 // y += alpha*A*x
161 for (size_t i = 0; i < m; ++i) {
162 double sum = 0.0;
163 for (size_t j = 0; j < n; ++j) {
164 sum += *gsl_matrix_ptr(A, i, j) * *gsl_vector_ptr(x, j);
165 }
166 *gsl_vector_ptr(y, i) += alpha * sum;
167 }
168 } else {
169 if (x->size != m || y->size != n) {
170 throw std::runtime_error("gsl_blas_dgemv: size mismatch");
171 }
172 // y = beta*y
173 for (size_t i = 0; i < n; ++i) {
174 *gsl_vector_ptr(y, i) *= beta;
175 }
176 // y += alpha*A^T*x
177 for (size_t i = 0; i < n; ++i) {
178 double sum = 0.0;
179 for (size_t j = 0; j < m; ++j) {
180 sum += *gsl_matrix_ptr(A, j, i) * *gsl_vector_ptr(x, j);
181 }
182 *gsl_vector_ptr(y, i) += alpha * sum;
183 }
184 }

196inline void gsl_blas_zgemv(CBLAS_TRANSPOSE TransA,
197 gsl_complex alpha, const gsl_matrix_complex* A, const gsl_vector_complex* x,
198 gsl_complex beta, gsl_vector_complex* y) {
199 size_t m = A->size1;
200 size_t n = A->size2;
201
202 if (TransA == CblasNoTrans) {
203 if (x->size != n || y->size != m) {
204 throw std::runtime_error("gsl_blas_zgemv: size mismatch");
205 }
206 // y = beta*y
207 for (size_t i = 0; i < m; ++i) {
208 *gsl_vector_complex_ptr(y, i) = gsl_complex_mul(*gsl_vector_complex_ptr(y, i), beta);
209 }
210 // y += alpha*A*x
211 for (size_t i = 0; i < m; ++i) {
212 gsl_complex sum = gsl_complex(0.0, 0.0);
213 for (size_t j = 0; j < n; ++j) {
214 sum = gsl_complex_add(sum, gsl_complex_mul(*gsl_matrix_complex_ptr(A, i, j),
215 *gsl_vector_complex_ptr(x, j)));
216 }
217 *gsl_vector_complex_ptr(y, i) = gsl_complex_add(*gsl_vector_complex_ptr(y, i),
218 gsl_complex_mul(alpha, sum));
219 }
220 } else {
221 if (x->size != m || y->size != n) {
222 throw std::runtime_error("gsl_blas_zgemv: size mismatch");
223 }
224 // y = beta*y
225 for (size_t i = 0; i < n; ++i) {
226 *gsl_vector_complex_ptr(y, i) = gsl_complex_mul(*gsl_vector_complex_ptr(y, i), beta);
227 }
228 // y += alpha*A^T*x or A^H*x
229 for (size_t i = 0; i < n; ++i) {
230 gsl_complex sum = gsl_complex(0.0, 0.0);
231 for (size_t j = 0; j < m; ++j) {
232 gsl_complex a_val = *gsl_matrix_complex_ptr(A, j, i);
233 if (TransA == CblasConjTrans) {
234 a_val = gsl_complex_conjugate(a_val);
235 }
236 sum = gsl_complex_add(sum, gsl_complex_mul(a_val, *gsl_vector_complex_ptr(x, j)));
237 }
238 *gsl_vector_complex_ptr(y, i) = gsl_complex_add(*gsl_vector_complex_ptr(y, i),
239 gsl_complex_mul(alpha, sum));
240 }
241 }
