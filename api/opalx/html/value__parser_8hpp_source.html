172 bool parseQuotedString(std::string& value) {
173 skipWhitespace();
174 if (pos_ >= input_.length() || input_[pos_] != '"') {
175 return false;
176 }
177 pos_++; // skip opening quote
178 value.clear();

180 while (pos_ < input_.length() && input_[pos_] != '"') {
181 if (input_[pos_] == '\\' && pos_ + 1 < input_.length()) {
182 pos_++;

184 } else {
185 value += input_[pos_];
186 }
187 pos_++;
188 }

192 return true;

197 bool parseString(std::string& value) {
198 skipWhitespace();
199 if (pos_ < input_.length() && input_[pos_] == '"') {
200 return parseQuotedString(value);
201 }
202
203 // Parse unquoted string (identifier)
204 size_t start = pos_;
205 while (pos_ < input_.length()) {
206 char c = input_[pos_];
207 if (std::isalnum(static_cast<unsigned char>(c)) ||
208 c == '@' || c == '#' || c == ':' || c == '+' ||
209 c == '-' || c == '%' || c == '.' || c == '_' ||
210 c == '$' || c == '&' || c == '/') {
211 pos_++;
212 } else {
213 break;
214 }
215 }
216
217 if (pos_ > start) {

219 return true;
220 }
