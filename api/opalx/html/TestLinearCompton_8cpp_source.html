40 double integrateSpectrumCompositeSimpson(double incomingPhotonEnergyERFGeV,
41 double minEnergy,
42 double maxEnergy,
43 int panels) {
44 const double h = (maxEnergy - minEnergy) / panels;
45 double sum = Physics::LinearCompton::differentialCrossSectionOmegaERF(incomingPhotonEnergyERFGeV,
46 minEnergy)
47 + Physics::LinearCompton::differentialCrossSectionOmegaERF(incomingPhotonEnergyERFGeV,
48 maxEnergy);
49
50 for (int i = 1; i < panels; ++i) {
51 const double energy = minEnergy + i * h;
52 const double weight = (i % 2 == 0) ? 2.0 : 4.0;
53 sum += weight * Physics::LinearCompton::differentialCrossSectionOmegaERF(
54 incomingPhotonEnergyERFGeV,
55 energy);
56 }
57
58 return sum * h / 3.0;
59 }
60}
