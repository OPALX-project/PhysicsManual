170inline std::set<SetStatistics> LossDataSink::computeStatistics(unsigned int numStatistics) {
171 std::set<SetStatistics> stats;
172
173 splitSets(numStatistics);
174
175 for (unsigned int i = 0; i < numStatistics; ++i) {
176 auto setStats = computeSetStatistics(i);
177 if (setStats.nTotal_m > 0) {
178 stats.insert(setStats);
179 }
180 }
181
182 return stats;
183}
