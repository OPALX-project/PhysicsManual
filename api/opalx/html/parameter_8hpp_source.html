66 parser::ValueParser parser(input, pos);
67 switch (*this->type_m) {
68 case ast::FLOAT: {
69 float f = 0.0f;

74 }
75 break;
76 }
77 case ast::DOUBLE: {

84 break;
85 }

121 }
122 }
