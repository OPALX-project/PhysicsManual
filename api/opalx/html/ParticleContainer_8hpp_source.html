398 void updateRefToLabCSTrafo(double bunchDT) {
399 Vector_t<double, 3> R = toLabTrafo_m.transformFrom(refPartR_m);
400 Vector_t<double, 3> P = toLabTrafo_m.rotateFrom(refPartP_m);
401
402 const double ds =
403 std::copysign(1.0, bunchDT) * std::sqrt(R[0] * R[0] + R[1] * R[1] + R[2] * R[2]);
404 sPos_m += ds;
405
406 CoordinateSystemTrafo update(R, getQuaternion(P, Vector_t<double, 3>(0, 0, 1)));
407 transformBunch(update);
408 toLabTrafo_m = toLabTrafo_m * update.inverted();
409 }

412 template <typename Pusher>
413 void applyFractionalStep(const Pusher& pusher, double tau, double pathLengthTarget) {
414 refPartR_m /= (Physics::c * 2 * tau);
415 pusher.push(refPartR_m, refPartP_m, tau);
416 refPartR_m *= (Physics::c * 2 * tau);
417
418 sPos_m = pathLengthTarget;
419 toLabTrafo_m.transformFrom(refPartR_m);
420 Vector_t<double, 3> R = refPartR_m;
421 toLabTrafo_m.rotateFrom(refPartP_m);
422 Vector_t<double, 3> P = refPartP_m;
423 CoordinateSystemTrafo update(R, getQuaternion(P, Vector_t<double, 3>(0, 0, 1)));
424 toLabTrafo_m = toLabTrafo_m * update.inverted();
425 }

551 size_type deleteParticlesOutside(double sigmasAway) {
552 size_type nLocal = this->getLocalNum();
553
554 if (nLocal == 0 && this->getTotalNum() == 0) return 0;
555 if (sigmasAway <= 0.0) return 0;
556
557 // Force fresh moments: cached values from bunchUpdate may not reflect the exact
558 // particle state at this point (emission/migration ordering can shift R between
559 // the last bunchUpdate and this call). Safety-critical deletion always recomputes.
560 markMomentsDirty();
561 updateMoments();
562
563 Vector_t<double, Dim> meanR = getMeanR();
564 Vector_t<double, Dim> rmsR = getRmsR();
565
566 double lb0 = meanR[0] - sigmasAway * rmsR[0];
567 double lb1 = meanR[1] - sigmasAway * rmsR[1];
568 double lb2 = meanR[2] - sigmasAway * rmsR[2];
569 double ub0 = meanR[0] + sigmasAway * rmsR[0];
570 double ub1 = meanR[1] + sigmasAway * rmsR[1];
571 double ub2 = meanR[2] + sigmasAway * rmsR[2];
572
573 Kokkos::View<bool*> invalid("deleteParticlesOutside::invalid", nLocal);
574 auto Rview = this->R.getView();
575
576 size_type localDestroyNum = 0;
577 Kokkos::parallel_reduce(
578 "ParticleContainer::deleteParticlesOutside::mark", nLocal,
579 KOKKOS_LAMBDA(const size_type i, size_type& count) {
580 bool outside = (Rview(i)[0] < lb0 || Rview(i)[0] > ub0)
581 || (Rview(i)[1] < lb1 || Rview(i)[1] > ub1)
582 || (Rview(i)[2] < lb2 || Rview(i)[2] > ub2);
583 invalid(i) = outside;
584 count += outside ? 1 : 0;
585 },
586 localDestroyNum);
587 Kokkos::fence();
588
589 size_type globalDestroyNum = 0;
590 ippl::Comm->allreduce(localDestroyNum, globalDestroyNum, 1, std::plus<size_type>());
591
592 if (globalDestroyNum == 0) return 0;
593
594 this->destroy(invalid, localDestroyNum);
595
596 // Only called if globalDestroyNum > 0, i.e. if any particles were destroyed --> statistics
597 // changed --> moments are dirty
598 markMomentsDirty();
599
600 return globalDestroyNum;
601 }
