426 void updateRefToLabCSTrafo(double bunchDT) {
427 Vector_t<double, 3> R = toLabTrafo_m.transformFrom(refPartR_m);
428 Vector_t<double, 3> P = toLabTrafo_m.rotateFrom(refPartP_m);
429
430 const double ds =
431 std::copysign(1.0, bunchDT) * std::sqrt(R[0] * R[0] + R[1] * R[1] + R[2] * R[2]);
432 sPos_m += ds;
433
434 CoordinateSystemTrafo update(R, getQuaternion(P, Vector_t<double, 3>(0, 0, 1)));
435 transformBunch(update);
436 toLabTrafo_m = toLabTrafo_m * update.inverted();
437 }

440 template <typename Pusher>
441 void applyFractionalStep(const Pusher& pusher, double tau, double pathLengthTarget) {
442 refPartR_m /= (Physics::c * 2 * tau);
443 pusher.push(refPartR_m, refPartP_m, tau);
444 refPartR_m *= (Physics::c * 2 * tau);
445
446 sPos_m = pathLengthTarget;
447 toLabTrafo_m.transformFrom(refPartR_m);
448 Vector_t<double, 3> R = refPartR_m;
449 toLabTrafo_m.rotateFrom(refPartP_m);
450 Vector_t<double, 3> P = refPartP_m;
451 CoordinateSystemTrafo update(R, getQuaternion(P, Vector_t<double, 3>(0, 0, 1)));
452 toLabTrafo_m = toLabTrafo_m * update.inverted();
453 }

579 size_type deleteParticlesOutside(double sigmasAway) {
580 size_type nLocal = this->getLocalNum();
581
582 if (nLocal == 0 && this->getTotalNum() == 0) return 0;
583 if (sigmasAway <= 0.0) return 0;
584
585 // Force fresh moments: cached values from bunchUpdate may not reflect the exact
586 // particle state at this point (emission/migration ordering can shift R between
587 // the last bunchUpdate and this call). Safety-critical deletion always recomputes.
588 markMomentsDirty();
589 updateMoments();
590
591 Vector_t<double, Dim> meanR = getMeanR();
592 Vector_t<double, Dim> rmsR = getRmsR();
593
594 double lb0 = meanR[0] - sigmasAway * rmsR[0];
595 double lb1 = meanR[1] - sigmasAway * rmsR[1];
596 double lb2 = meanR[2] - sigmasAway * rmsR[2];
597 double ub0 = meanR[0] + sigmasAway * rmsR[0];
598 double ub1 = meanR[1] + sigmasAway * rmsR[1];
599 double ub2 = meanR[2] + sigmasAway * rmsR[2];
600
601 Kokkos::View<bool*> invalid("deleteParticlesOutside::invalid", nLocal);
602 auto Rview = this->R.getView();
603
604 size_type localDestroyNum = 0;
605 Kokkos::parallel_reduce(
606 "ParticleContainer::deleteParticlesOutside::mark", nLocal,
607 KOKKOS_LAMBDA(const size_type i, size_type& count) {
608 bool outside = (Rview(i)[0] < lb0 || Rview(i)[0] > ub0)
609 || (Rview(i)[1] < lb1 || Rview(i)[1] > ub1)
610 || (Rview(i)[2] < lb2 || Rview(i)[2] > ub2);
611 invalid(i) = outside;
612 count += outside ? 1 : 0;
613 },
614 localDestroyNum);
615 Kokkos::fence();
616
617 size_type globalDestroyNum = 0;
618 ippl::Comm->allreduce(localDestroyNum, globalDestroyNum, 1, std::plus<size_type>());
619
620 if (globalDestroyNum == 0) return 0;
621
622 destroyParticles(invalid, localDestroyNum);
623
624 // Only called if globalDestroyNum > 0, i.e. if any particles were destroyed --> statistics
625 // changed --> moments are dirty
626 markMomentsDirty();
627
628 return globalDestroyNum;
629 }
