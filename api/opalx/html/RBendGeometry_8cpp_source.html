60Euclid3D RBendGeometry::getEntrancePatch() const {
61 double d = getElementLength() / 4.0 * tan(half_angle / 2.0);
62 return Euclid3D::YRotation(-half_angle) * Euclid3D::translation(d, 0.0, 0.0);
63}
