42void Help::execute() {
43 if (itsAttr[0]) {
44 std::string name = Attributes::getString(itsAttr[0]);
45
46 if (Object* object = OpalData::getInstance()->find(name)) {
47 object->printHelp(std::cerr);
48 } else {
49 *ippl::Error << "\n" << *this << "Unknown object \"" << name << "\".\n" << endl;
50 }
51 } else {
52 printHelp(std::cerr);
53 }
