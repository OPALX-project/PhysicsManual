136 + leverz * leverr * Ez(index2 + 1);

139 + leverz * (1.0 - leverr) * Bt(index1 + 1)
140 + (1.0 - leverz) * leverr * Bt(index2)

145 E(1) += EfieldR * R(1) / RR;
146 B(0) -= BfieldT * R(1) / RR;
147 B(1) += BfieldT * R(0) / RR;
148 }
149 E(2) += EfieldZ;
150 return;
