34PeakFinder::PeakFinder(std::string outfn, double min, double max, double binWidth, bool singlemode)
35 : outputName_m(outfn),
36 binWidth_m(binWidth),
37 min_m(min),

60void PeakFinder::evaluate(const int& turn) {
61 if (first_m) {
62 turn_m = turn;
63 first_m = false;
64 }
65
66 if (turn_m != turn) {
67 finished_m = true;
68 }
69
70 bool globFinished = false;
71
72 if (!singlemode_m)
73 ippl::Comm->allreduce(finished_m, globFinished, 1, std::logical_and<bool>());
74 else
75 globFinished = finished_m;
76
77 if (globFinished) {
78 this->computeCentroid_m();
79
80 turn_m = turn;
81
82 // reset
83 peakRadius_m = 0.0;
84 registered_m = 0;
85 finished_m = false;
86 }
87}

119void PeakFinder::computeCentroid_m() {
120 double globPeakRadius = 0.0;
121 int globRegister = 0;
122
123 if (!singlemode_m) {
124 ippl::Comm->reduce(peakRadius_m, globPeakRadius, 1, std::plus<double>());
125 ippl::Comm->reduce(registered_m, globRegister, 1, std::plus<int>());
126 } else {
127 globPeakRadius = peakRadius_m;
128 globRegister = registered_m;
129 }
130
131 if (ippl::Comm->rank() == 0) {
132 if (globRegister > 0) peaks_m.push_back(globPeakRadius / double(globRegister));
133 }
134}

136void PeakFinder::createHistogram_m() {
137 /*
138 * create local histograms
139 */
140
141 globHist_m.resize(nBins_m);
142 container_t locHist(nBins_m, 0.0);
143
144 double invBinWidth = 1.0 / binWidth_m;
145 for (container_t::iterator it = radius_m.begin(); it != radius_m.end(); ++it) {
146 int bin = static_cast<int>(std::abs(*it - min_m) * invBinWidth);
147 if (bin < 0 || (unsigned int)bin >= nBins_m)
148 continue; // Probe might save particles outside its boundary
149 ++locHist[bin];
150 }
151
152 /*
153 * create global histograms
154 */
155 if (!singlemode_m) {
156 reduce(locHist.data(), globHist_m.data(), locHist.size(), std::plus<double>());
157 } else {
158 globHist_m.swap(locHist);
159 }
160}
