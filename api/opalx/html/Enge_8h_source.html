161 if (n == 0) {
162 return (getEnge(x - _x0, n) + getEnge(-x - _x0, n)) - 1.;
163 } else {
164 if (n % 2 != 0)
165 return getEnge(x - _x0, n) - getEnge(-x - _x0, n);
166 else
167 return getEnge(x - _x0, n) + getEnge(-x - _x0, n);
168 }
