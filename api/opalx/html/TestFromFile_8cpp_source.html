96TEST_F(FromFileTest, GeneratesParticlesFromAsciiFile) {
97 writeSampleFile();
98
99 auto fc = std::shared_ptr<FieldContainer_t>();

101
102 size_t requested = 10; // larger than available in file to test clamping
103
104 // Preallocate capacity so SamplingBase::computeLocalEmitCount
105 // can distribute up to the requested number of particles.
106 const int nranks = std::max(1, ippl::Comm->size());
107 const size_t nranksU = static_cast<size_t>(nranks);
108 const size_t maxLocalNum =
109 requested / nranksU + 2 * nranksU + 1;
110 pc->create(maxLocalNum);
111 Kokkos::View<bool*> tmp_invalid("tmp_invalid", maxLocalNum);
112 pc->destroy(tmp_invalid, maxLocalNum);
113
114 sampler.generateParticles(requested, nr);
115
116 int rank;
117 MPI_Comm_rank(MPI_COMM_WORLD, &rank);
118
119 // Sum local particles over all ranks
120 size_t localN = pc->getLocalNum();
121 size_t globalN = 0;
122 MPI_Allreduce(&localN, &globalN, 1, MPI_UNSIGNED_LONG, MPI_SUM, MPI_COMM_WORLD);
123
124 EXPECT_EQ(globalN, 5u);
125
126 // Copy views to host space for GPU compatibility
127 auto Rview_d = pc->R.getView();
128 auto Pview_d = pc->P.getView();

132 Kokkos::deep_copy(Rview, Rview_d);
133 Kokkos::deep_copy(Pview, Pview_d);
134
