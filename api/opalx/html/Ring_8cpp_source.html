140bool Ring::apply(
141 const Vector_t<double, 3>& R, const Vector_t<double, 3>& /*P*/, const double& t,
142 Vector_t<double, 3>& E, Vector_t<double, 3>& B) {
143 B = Vector_t<double, 3>({0.0, 0.0, 0.0});
144 E = Vector_t<double, 3>({0.0, 0.0, 0.0});
145
146 std::vector<RingSection*> sections =
147 getSectionsAt(R); // I think this doesn't actually use R -DW
148 bool outOfBounds = true;
149 // assume field maps don't set B, E to 0...
150 if (willDoAperture_m) {
151 double rSquared = R[0] * R[0] + R[1] * R[1];
152 if (rSquared < minR2_m || rSquared > maxR2_m) {
153 return true;
154 }
155 }
156
157 for (size_t i = 0; i < sections.size(); ++i) {
158 Vector_t<double, 3> B_temp({0.0, 0.0, 0.0});
159 Vector_t<double, 3> E_temp({0.0, 0.0, 0.0});
160 // Super-TEMP! cyclotron tracker now uses m internally, have to change to mm here to match
161 // old field limits -DW
162 outOfBounds &= sections[i]->getFieldValue(
163 R * Vector_t<double, 3>(1000.0),
164 refPartBunch_m->getParticleContainer()->getCentroid() * Vector_t<double, 3>(1000.0),
165 t, E_temp, B_temp);
166 B += (scale_m * B_temp);
167 E += (scale_m * E_temp);
168 }
169 return outOfBounds;
170}

206Rotation3D Ring::getRotationStartToEnd(Euclid3D delta) const {
207 // rotation from start to end in local coordinates
208 // Euclid3D/Rotation3D doesnt have a "getAngle" method so we use fairly
209 // obscure technique to extract it.
210 Vector3D rotationTest(1., 0., 0.);
211 rotationTest = delta.getRotation() * rotationTest;
212 double deltaAngle = std::atan2(rotationTest(2), rotationTest(0));
213 Rotation3D elementRotation = Rotation3D::ZRotation(deltaAngle);
214 return elementRotation;
215}

217void Ring::checkMidplane(Euclid3D delta) const {
218 if (std::abs(delta.getVector()(2)) > lengthTolerance_m
219 || std::abs(delta.getRotation().getAxis()(0)) > angleTolerance_m
220 || std::abs(delta.getRotation().getAxis()(1)) > angleTolerance_m) {
221 throw GeneralOpalException(
222 "Ring::checkMidplane", std::string("Placement of elements out of the ")
223 + "midplane is not supported by Ring");
224 }

227void Ring::rotateToCyclCoordinates(Euclid3D& delta) const {
228 Vector3D v = delta.getVector();
229 Vector3D r = delta.getRotation().getAxis();
230 // Euclid3D euclid(v(0), -v(2), v(1), r(0), -r(2), r(1));
231 Euclid3D euclid(v(2), v(0), -v(1), r(2), r(0), -r(1));
232 delta = euclid;

253void Ring::appendElement(const Component& element) {
254 if (isLocked_m) {
255 throw GeneralOpalException(
256 "Ring::appendElement",
257 "Attempt to append element " + element.getName() + " when ring is locked");
258 }
259 // delta is transform from start of bend to end with x, z as horizontal
260 // I failed to get Rotation3D to work so use rotations written by hand.
261 // Probably an error in my call to Rotation3D.
262 Euclid3D delta = element.getGeometry().getTotalTransform();
263 rotateToCyclCoordinates(delta);
264 checkMidplane(delta);
265
266 RingSection* section = new RingSection();
267 Vector_t<double, 3> startPos = getNextPosition();
268 Vector_t<double, 3> startNorm = getNextNormal();
269
270 section->setComponent(dynamic_cast<Component*>(element.clone()));
271 section->setStartPosition(startPos);
272 section->setStartNormal(startNorm);
273
274 double startF = std::atan2(startNorm(1), startNorm(0));
275 Vector_t<double, 3> endPos = Vector_t<double, 3>(
276 {+delta.getVector()(0) * std::cos(startF)
277 - delta.getVector()(1) * std::sin(startF),
278 +delta.getVector()(0) * std::sin(startF)
279 + delta.getVector()(1) * std::cos(startF),
280 0})
281 + startPos;
282 section->setEndPosition(endPos);
283
284 double endF = delta.getRotation().getAxis()(2); //+
285 // atan2(delta.getVector()(1), delta.getVector()(0));
286 Vector_t<double, 3> endNorm = Vector_t<double, 3>(
287 {+startNorm(0) * std::cos(endF) + startNorm(1) * std::sin(endF),
288 -startNorm(0) * std::sin(endF) + startNorm(1) * std::cos(endF), 0});
289 section->setEndNormal(endNorm);
290
291 section->setComponentPosition(startPos);
292 section->setComponentOrientation(Vector_t<double, 3>({0, 0, startF}));
293
294 section_list_m.push_back(section);
295
296 double dphi = atan2(startNorm(0), startNorm(1));
297 Inform msg("OPAL");
298 msg << "* Added " << element.getName() << " to Ring" << endl;
299 msg << "* Start position (" << section->getStartPosition()(0) << ", "
300 << section->getStartPosition()(1) << ", " << section->getStartPosition()(2) << ") normal ("
301 << section->getStartNormal()(0) << ", " << section->getStartNormal()(1) << ", "
302 << section->getStartNormal()(2) << "), phi " << dphi << endl;
303 msg << "* End position (" << section->getEndPosition()(0) << ", "
304 << section->getEndPosition()(1) << ", " << section->getEndPosition()(2) << ") normal ("
305 << section->getEndNormal()(0) << ", " << section->getEndNormal()(1) << ", "
306 << section->getEndNormal()(2) << ")" << endl;
307}

309void Ring::lockRing() {
310 Inform msg("OPAL ");
311 if (isLocked_m) {
312 throw GeneralOpalException(
313 "Ring::lockRing", "Attempt to lock ring when it is already locked");
314 }
315 // check for any elements at all
316 size_t sectionListSize = section_list_m.size();
317 if (sectionListSize == 0) {
318 throw GeneralOpalException("Ring::lockRing", "Failed to find any elements in Ring");
319 }
320 // Apply symmetry properties; I think it is fastest to just duplicate
321 // elements rather than try to do some angle manipulations when we do field
322 // lookups because we do field lookups in Cartesian coordinates in general.
323 msg << "Applying symmetry to Ring" << endl;
324 for (int i = 1; i < symmetry_m; ++i) {
325 for (size_t j = 0; j < sectionListSize; ++j) {
326 appendElement(*section_list_m[j]->getComponent());
327 }
328 }
329 // resetAzimuths();
330 // Check that the ring is closed within tolerance and close exactly
331 if (isClosed_m) checkAndClose();
332 buildRingSections();
333 isLocked_m = true;
334 for (size_t i = 0; i < section_list_m.size(); i++) {
335 }
336}

338void Ring::resetAzimuths() {
339 for (size_t i = 0; i < section_list_m.size(); ++i) {
340 Vector_t<double, 3> startPos = section_list_m[i]->getEndPosition();
341 Vector_t<double, 3> startDir = section_list_m[i]->getEndNormal();
342 Vector_t<double, 3> endPos = section_list_m[i]->getEndPosition();
343 Vector_t<double, 3> endDir = section_list_m[i]->getEndNormal();
344 if (!section_list_m[i]->isOnOrPastStartPlane(endPos)) {
345 section_list_m[i]->setEndPosition(startPos);
346 section_list_m[i]->setEndNormal(startDir);
347 section_list_m[i]->setStartPosition(endPos);
348 section_list_m[i]->setStartNormal(endDir);
349 }
350 }

353void Ring::checkAndClose() {
354 Vector_t<double, 3> first_pos = section_list_m[0]->getStartPosition();
355 Vector_t<double, 3> first_norm = section_list_m[0]->getStartNormal();
356 Vector_t<double, 3> last_pos = section_list_m.back()->getEndPosition();
357 Vector_t<double, 3> last_norm = section_list_m.back()->getEndNormal();
358 for (int i = 0; i < 3; ++i) {
359 if (std::abs(first_pos(i) - last_pos(i)) > lengthTolerance_m
360 || std::abs(first_norm(i) - last_norm(i)) > angleTolerance_m)
361 throw GeneralOpalException("Ring::lockRing", "Ring is not closed");
362 }
363 section_list_m.back()->setEndPosition(first_pos);
364 section_list_m.back()->setEndNormal(first_norm);
365}

367void Ring::buildRingSections() {
368 size_t nSections = 2. * Physics::pi / phiStep_m + 1;
369 ringSections_m = std::vector<std::vector<RingSection*> >(nSections);
370 for (size_t i = 0; i < ringSections_m.size(); ++i) {
371 double phi0 = i * phiStep_m;
372 double phi1 = (i + 1) * phiStep_m;
373 // std::cerr << phi0 << " " << phi1 << std::endl;
374 for (size_t j = 0; j < section_list_m.size(); ++j) {
375 if (section_list_m[j]->doesOverlap(phi0, phi1))
376 ringSections_m[i].push_back(section_list_m[j]);
377 }
378 }
379}
