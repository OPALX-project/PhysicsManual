2// Class DataSink
3// This class acts as an observer during the calculation. It generates diagnostic
4// output of the accelerated beam such as statistical beam descriptors of particle
5// positions, momenta, beam phase space (emittance) etc. These are written to file
6// at periodic time steps during the calculation.
7//
8// This class also writes the full beam phase space to an H5 file at periodic time
9// steps in the calculation (this period is different from that of the statistical
10// numbers).
11
12// Class also writes processor load balancing data to file to track parallel
13// calculation efficiency.

84void DataSink::dumpH5(
85 PartBunch_t& beam,
86 const std::vector<std::array<Vector_t<double, 3>, 2>>& fdextPerContainer) const {
87 if (!Options::enableHDF5) {
88 return;
89 }
90
91 const size_t n = beam.getNumParticleContainers();
92 if (fdextPerContainer.size() < n) {
93 throw OpalException(
94 "DataSink::dumpH5",
95 "fdextPerContainer size (" + std::to_string(fdextPerContainer.size()) + ") < num containers ("
96 + std::to_string(n) + ").");
97 }
98
99 for (size_t i = 0; i < n; ++i) {
100 if (i >= h5Writers_m.size()) {
101 break;
102 }
103 auto pc = beam.getParticleContainer(i);
104 if (!pc || pc->getTotalNum() == 0) {
105 continue;
106 }
107 Vector_t<double, 3> fd[2] = {fdextPerContainer[i][0], fdextPerContainer[i][1]};
108 h5Writers_m[i]->writePhaseSpace(beam, fd, i);
109 }
110}

133void DataSink::dumpSDDS(
134 PartBunch_t& beam,
135 const std::vector<std::array<Vector_t<double, 3>, 2>>& fdextPerContainer,
136 const losses_t& losses,
137 const double& azimuth) const {
138
139 const size_t n = beam.getNumParticleContainers();
140 if (fdextPerContainer.size() < n) {
141 throw OpalException(
142 "DataSink::dumpSDDS",
143 "fdextPerContainer size (" + std::to_string(fdextPerContainer.size()) + ") < num containers ("
144 + std::to_string(n) + ").");
145 }
146
147 IpplTimings::startTimer(StatMarkerTimer_m);
148
149 for (size_t i = 0; i < n; ++i) {
150 if (i >= statWriters_m.size()) {
151 break;
152 }
153 auto pc = beam.getParticleContainer(i);
154 if (!pc || pc->getTotalNum() == 0) {
155 continue;
156 }
157
158 pc->updateMoments();
159
160 size_t npOutside = 0;
161 if (Options::beamHaloBoundary > 0) {
162 npOutside = beam.calcNumPartsOutside(
163 Options::beamHaloBoundary * pc->getRmsR());
164 }
165
166 Vector_t<double, 3> fd[2] = {fdextPerContainer[i][0], fdextPerContainer[i][1]};
167 statWriters_m[i]->write(beam, fd, losses, azimuth, npOutside, i);
168 }
169
170 beam.gatherLoadBalanceStatistics();
171 beam.calcBeamParameters();
172
173 IpplTimings::stopTimer(StatMarkerTimer_m);
174}

207void DataSink::writeImpactStatistics(
208 const PartBunch_t& beam, long long& step, size_t& impact, double& sey_num,
209 size_t numberOfFieldEmittedParticles, bool nEmissionMode, std::string fn) {
210 double charge = 0.0;
211 size_t Npart = 0;
212 double Npart_d = 0.0;
213 if (!nEmissionMode) {
214 const auto pc = beam.getParticleContainers()[0];
215 charge = -1.0 * pc->getTotalCharge();
216 // reduce(charge, charge, OpAddAssign());
217 Npart_d = -1.0 * charge / pc->getChargePerParticle();
218 } else {
219 Npart = beam.getParticleContainers()[0]->getTotalNum();
220 }
221 if (ippl::Comm->rank() == 0) {
222 std::string ffn = fn + std::string(".dat");
223
224 std::unique_ptr<Inform> ofp(new Inform(nullptr, ffn.c_str(), Inform::APPEND, 0));
225 Inform& fid = *ofp;
226 fid.precision(6);
227 fid << std::setiosflags(std::ios::scientific);
228 double t = beam.getT() * Units::s2ns;
229 if (!nEmissionMode) {
230 if (step == 0) {
231 fid << "#Time/ns" << std::setw(18) << "#Geometry impacts" << std::setw(18)
232 << "tot_sey" << std::setw(18) << "TotalCharge" << std::setw(18) << "PartNum"
233 << " numberOfFieldEmittedParticles " << endl;
234 }
235 fid << t << std::setw(18) << impact << std::setw(18) << sey_num << std::setw(18)
236 << charge << std::setw(18) << Npart_d << std::setw(18)
237 << numberOfFieldEmittedParticles << endl;
238 } else {
239 if (step == 0) {
240 fid << "#Time/ns" << std::setw(18) << "#Geometry impacts" << std::setw(18)
241 << "tot_sey" << std::setw(18) << "ParticleNumber"
242 << " numberOfFieldEmittedParticles " << endl;
243 }
244 fid << t << std::setw(18) << impact << std::setw(18) << sey_num << std::setw(18)
245 << double(Npart) << std::setw(18) << numberOfFieldEmittedParticles << endl;
246 }
247 }

250void DataSink::dumpBinConfig(long long step,
251 double time,
252 bool preMerge,
253 const std::vector<std::size_t>& binCounts,
254 const std::vector<double>& binWidths,
255 double xMin,
256 const std::string& fileName) {
257 if (ippl::Comm->rank() != 0) {
258 return;
259 }
260
261 Inform m("DataSink::dumpBinConfig");
262
263 if (binCounts.size() != binWidths.size()) {
264 m << level4 << "Invalid bin configuration: binCounts.size() = " << binCounts.size()
265 << ", binWidths.size() = " << binWidths.size() << endl;
266 throw OpalException("DataSink::dumpBinConfig",
267 "binCounts and binWidths must have the same length.");
268 }
269
270 if (!binConfigWriter_m) {
271 m << level4 << "Creating BinConfigWriter for JSON binning output file \""
272 << fileName << "\"." << endl;
273 binConfigWriter_m = std::make_unique<BinConfigWriter>(fileName);
274 }
275
276 m << level5 << "Dumping bin configuration snapshot: step=" << step
277 << ", time=" << time
278 << ", preMerge=" << (preMerge ? 1 : 0)
279 << ", nBins=" << binCounts.size()
280 << ", xMin=" << xMin << endl;
281
282 binConfigWriter_m->writeEntry(step, time, preMerge, binCounts, binWidths, xMin);
283}

285void DataSink::rewindLines() {
286 unsigned int linesToRewind = 0;
287
288 for (size_t i = 0; i < h5Writers_m.size(); ++i) {
289 double spos = h5Writers_m[i]->getLastPosition();
290 if (i < statWriters_m.size() && statWriters_m[i]->exists()) {
291 unsigned int li = statWriters_m[i]->rewindToSpos(spos);
292 statWriters_m[i]->replaceVersionString();
293 linesToRewind = std::max(linesToRewind, li);
294 }
295 h5Writers_m[i]->close();
296 }
297
298 if (linesToRewind > 0) {
299 for (size_t i = 0; i < sddsWriter_m.size(); ++i) {
300 sddsWriter_m[i]->rewindLines(linesToRewind);
301 sddsWriter_m[i]->replaceVersionString();
302 }
303 }

306void DataSink::init(bool restart, const std::vector<H5PartWrapper*>& h5wrappers, size_t numParticleContainers) {
307 std::string fn = OpalData::getInstance()->getInputBasename();
308
309 lossWrCounter_m = 0;
310 StatMarkerTimer_m = IpplTimings::getTimer("Write Stat");
311
312 statWriters_m.clear();
313 for (size_t i = 0; i < numParticleContainers; ++i) {
314 std::string stem = diagnosticStemForContainer(fn, numParticleContainers, i);
315 statWriters_m.push_back(statWriter_t(new StatWriter(stem + std::string(".stat"), restart)));
316 }
317
318 sddsWriter_m.clear();
319 sddsWriter_m.push_back(sddsWriter_t(new LBalWriter(fn + std::string(".lbal"), restart)));
320
321 h5Writers_m.clear();
322 if (Options::enableHDF5) {
323 if (h5wrappers.size() != numParticleContainers) {
324 throw OpalException(
325 "DataSink::init",
326 "HDF5 enabled: expected " + std::to_string(numParticleContainers) + " H5PartWrapper(s), got "
327 + std::to_string(h5wrappers.size()) + ".");
328 }
329 for (H5PartWrapper* w : h5wrappers) {
330 h5Writers_m.push_back(h5Writer_t(new H5Writer(w, restart)));
331 }
332 } else if (!h5wrappers.empty()) {
333 throw OpalException("DataSink::init", "H5 wrappers passed while HDF5 is disabled.");
334 }
