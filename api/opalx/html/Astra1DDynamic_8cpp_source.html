437bool Astra1DDynamic::getFieldDerivative(
438 const Vector_t<double, 3>& R,
439 Vector_t<double, 3> &E,
440 Vector_t<double, 3> &/*B*/,
441 const DiffDirection &/*dir*/) const
442{
443 const double kz = Physics::two_pi * (R(2) - zbegin_m) / length_m + Physics::pi;
444 double ezp = 0.0;
445
446 auto FourCoefs = FourCoefs_m.h_view;
447
448 int n = 1;
449 for (int l = 1; l < accuracy_m; ++l, n += 2) {
450 ezp += Physics::two_pi / length_m * l *
451 (-FourCoefs(n) * std::sin(kz * l)
452 - FourCoefs(n + 1) * std::cos(kz * l));
453 }
454
455 E(2) += ezp;
456
457 return false;
458}

502{
