366bool Astra1DDynamic::getFieldDerivative(
367 const Vector_t<double, 3>& R, Vector_t<double, 3>& E, Vector_t<double, 3>& /*B*/,
368 const DiffDirection& /*dir*/) const {
369 const double kz = Physics::two_pi * (R(2) - zbegin_m) / length_m + Physics::pi;
370 double ezp = 0.0;
371
372 auto FourCoefs = FourCoefs_m.h_view;
373
374 int n = 1;
375 for (int l = 1; l < accuracy_m; ++l, n += 2) {
376 ezp += Physics::two_pi / length_m * l
377 * (-FourCoefs(n) * std::sin(kz * l) - FourCoefs(n + 1) * std::cos(kz * l));
378 }
379
380 E(2) += ezp;
381
382 return false;
