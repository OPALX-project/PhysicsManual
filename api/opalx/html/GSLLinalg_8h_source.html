66inline int gsl_linalg_LU_decomp(gsl_matrix* A, gsl_permutation* p, int* signum) {
67 if (A->size1 != A->size2) {
68 return -1; // Error: not square
69 }
70
71 size_t n = A->size1;
72 *signum = 1;
73
74 // Initialize permutation
75 for (size_t i = 0; i < n; ++i) {
76 p->data[i] = i;
77 }
78
79 // Perform LU decomposition with partial pivoting
80 for (size_t k = 0; k < n - 1; ++k) {
81 // Find pivot
82 size_t max_row = k;
83 double max_val = std::abs(*gsl_matrix_ptr(A, k, k));
84 for (size_t i = k + 1; i < n; ++i) {
85 double val = std::abs(*gsl_matrix_ptr(A, i, k));
86 if (val > max_val) {
87 max_val = val;
88 max_row = i;
89 }
90 }
91
92 // Swap rows if needed
93 if (max_row != k) {
94 *signum = -*signum;
95 std::swap(p->data[k], p->data[max_row]);
96 for (size_t j = 0; j < n; ++j) {
97 std::swap(*gsl_matrix_ptr(A, k, j), *gsl_matrix_ptr(A, max_row, j));
98 }
99 }
100
101 // Check for singularity
102 if (std::abs(*gsl_matrix_ptr(A, k, k)) < 1e-15) {
103 return -1; // Singular matrix
104 }
105
106 // Eliminate
107 for (size_t i = k + 1; i < n; ++i) {
108 double factor = *gsl_matrix_ptr(A, i, k) / *gsl_matrix_ptr(A, k, k);
109 *gsl_matrix_ptr(A, i, k) = factor;
110 for (size_t j = k + 1; j < n; ++j) {
111 *gsl_matrix_ptr(A, i, j) -= factor * *gsl_matrix_ptr(A, k, j);
112 }
113 }
114 }
115
116 return 0; // Success
117}

124inline double gsl_linalg_LU_det(const gsl_matrix* LU, int signum) {

127 for (size_t i = 0; i < n; ++i) {
128 det *= *gsl_matrix_ptr(LU, i, i);

139inline int gsl_linalg_LU_invert(const gsl_matrix* LU, const gsl_permutation* p, gsl_matrix* inverse) {
140 size_t n = LU->size1;
141 if (inverse->size1 != n || inverse->size2 != n) {
142 return -1;
143 }
144
145 // Initialize inverse as permuted identity: P*I
146 for (size_t j = 0; j < n; ++j) {
147 for (size_t i = 0; i < n; ++i) {
148 *gsl_matrix_ptr(inverse, i, j) = (p->data[i] == j) ? 1.0 : 0.0;
149 }
150 }
151
152 // Forward substitution: L*y = P*I (solve for y)
153 // L is unit lower triangular stored in LU below diagonal
154 for (size_t j = 0; j < n; ++j) {
155 for (size_t i = 0; i < n; ++i) {
156 double sum = 0.0;
157 for (size_t k = 0; k < i; ++k) {
158 sum += *gsl_matrix_ptr(LU, i, k) * *gsl_matrix_ptr(inverse, k, j);
159 }
160 *gsl_matrix_ptr(inverse, i, j) -= sum;
161 }
162 }
163
164 // Back substitution: U*x = y (solve for x, which is the inverse)
165 // U is upper triangular stored in LU on and above diagonal
166 for (int i = static_cast<int>(n) - 1; i >= 0; --i) {
167 for (size_t j = 0; j < n; ++j) {
168 double sum = 0.0;
169 for (size_t k = i + 1; k < n; ++k) {
170 sum += *gsl_matrix_ptr(LU, i, k) * *gsl_matrix_ptr(inverse, k, j);
171 }
172 *gsl_matrix_ptr(inverse, i, j) = (*gsl_matrix_ptr(inverse, i, j) - sum) / *gsl_matrix_ptr(LU, i, i);
173 }
174 }
175
176 return 0;
177}

185inline int gsl_linalg_complex_LU_decomp(gsl_matrix_complex* A, gsl_permutation* p, int* signum) {
186 if (A->size1 != A->size2) {
187 return -1;
188 }
189
190 size_t n = A->size1;
191 *signum = 1;
192
193 // Initialize permutation
194 for (size_t i = 0; i < n; ++i) {
195 p->data[i] = i;
196 }
197
198 // Perform LU decomposition with partial pivoting
199 for (size_t k = 0; k < n - 1; ++k) {
200 // Find pivot
201 size_t max_row = k;
202 double max_val = gsl_complex_abs(*gsl_matrix_complex_ptr(A, k, k));
203 for (size_t i = k + 1; i < n; ++i) {
204 double val = gsl_complex_abs(*gsl_matrix_complex_ptr(A, i, k));
205 if (val > max_val) {
206 max_val = val;
207 max_row = i;
208 }
209 }
210
211 // Swap rows if needed
212 if (max_row != k) {
213 *signum = -*signum;
214 std::swap(p->data[k], p->data[max_row]);
215 for (size_t j = 0; j < n; ++j) {
216 std::swap(*gsl_matrix_complex_ptr(A, k, j), *gsl_matrix_complex_ptr(A, max_row, j));
217 }
218 }
219
220 // Check for singularity
221 if (gsl_complex_abs(*gsl_matrix_complex_ptr(A, k, k)) < 1e-15) {
222 return -1;
223 }
224
225 // Eliminate
226 for (size_t i = k + 1; i < n; ++i) {
227 gsl_complex factor = gsl_complex_div(*gsl_matrix_complex_ptr(A, i, k),
228 *gsl_matrix_complex_ptr(A, k, k));
229 *gsl_matrix_complex_ptr(A, i, k) = factor;
230 for (size_t j = k + 1; j < n; ++j) {
231 *gsl_matrix_complex_ptr(A, i, j) = gsl_complex_sub(*gsl_matrix_complex_ptr(A, i, j),
232 gsl_complex_mul(factor, *gsl_matrix_complex_ptr(A, k, j)));
233 }
234 }
235 }
236
237 return 0;
238}

260inline int gsl_linalg_complex_LU_invert(const gsl_matrix_complex* LU, const gsl_permutation* p, gsl_matrix_complex* inverse) {
261 size_t n = LU->size1;
262 if (inverse->size1 != n || inverse->size2 != n) {
263 return -1;
264 }
265
266 // Initialize inverse as permuted identity: P*I
267 for (size_t j = 0; j < n; ++j) {
268 for (size_t i = 0; i < n; ++i) {
269 *gsl_matrix_complex_ptr(inverse, i, j) = (p->data[i] == j) ? gsl_complex(1.0, 0.0) : gsl_complex(0.0, 0.0);
270 }
271 }
272
273 // Forward substitution: L*y = P*I (solve for y)
274 // L is unit lower triangular stored in LU below diagonal
275 for (size_t j = 0; j < n; ++j) {
276 for (size_t i = 0; i < n; ++i) {
277 gsl_complex sum = gsl_complex(0.0, 0.0);
278 for (size_t k = 0; k < i; ++k) {
279 sum = gsl_complex_add(sum, gsl_complex_mul(*gsl_matrix_complex_ptr(LU, i, k),
280 *gsl_matrix_complex_ptr(inverse, k, j)));
281 }
282 *gsl_matrix_complex_ptr(inverse, i, j) = gsl_complex_sub(*gsl_matrix_complex_ptr(inverse, i, j), sum);
283 }
284 }
285
286 // Back substitution: U*x = y (solve for x, which is the inverse)
287 // U is upper triangular stored in LU on and above diagonal
288 for (int i = static_cast<int>(n) - 1; i >= 0; --i) {
289 for (size_t j = 0; j < n; ++j) {
290 gsl_complex sum = gsl_complex(0.0, 0.0);
291 for (size_t k = i + 1; k < n; ++k) {
292 sum = gsl_complex_add(sum, gsl_complex_mul(*gsl_matrix_complex_ptr(LU, i, k),
293 *gsl_matrix_complex_ptr(inverse, k, j)));
294 }
295 *gsl_matrix_complex_ptr(inverse, i, j) = gsl_complex_div(gsl_complex_sub(*gsl_matrix_complex_ptr(inverse, i, j), sum),
296 *gsl_matrix_complex_ptr(LU, i, i));
297 }
298 }
299
300 return 0;
301}
