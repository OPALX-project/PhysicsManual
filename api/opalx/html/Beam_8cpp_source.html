125void Beam::execute() {
126 const bool photon = itsAttr[PARTICLE] && getParticleName() == photonParticleName;
127
128 if (photon) {
129 if (!itsAttr[ENERGY]) {
130 throw OpalException("Beam::execute()", "\"ENERGY\" must be set for PARTICLE=PHOTON.");
131 }
132 if (itsAttr[MASS]) {
133 throw OpalException(
134 "Beam::execute()",
135 "\"MASS\" is not allowed for PARTICLE=PHOTON. Use \"ENERGY\".");
136 }
137 if (itsAttr[CHARGE]) {
138 throw OpalException(
139 "Beam::execute()", "\"CHARGE\" is not allowed for PARTICLE=PHOTON.");
140 }
141 if (itsAttr[PC]) {
142 throw OpalException(
143 "Beam::execute()",
144 "\"PC\" is not allowed for PARTICLE=PHOTON. Use \"ENERGY\".");
145 }
146 if (itsAttr[GAMMA]) {
147 throw OpalException(
148 "Beam::execute()",
149 "\"GAMMA\" is not allowed for PARTICLE=PHOTON. Use \"ENERGY\".");
150 }
151 if (itsAttr[SOURCES]) {
152 throw OpalException(
153 "Beam::execute()", "\"SOURCES\" is not allowed for PARTICLE=PHOTON.");
154 }
155 }
156
157 if (itsAttr[BCURRENT] || itsAttr[BFREQ]) {
158 throw OpalException(
159 "Beam::execute()",
160 "\"BCURRENT\" and \"BFREQ\" are no longer used in OPALX. "
161 "Use \"BCHARGE\" [C] to specify the bunch charge directly.");
162 }
163
164 update();
165
166 if (!(itsAttr[PARTICLE]) && (!itsAttr[MASS] || !(itsAttr[CHARGE]))) {
167 throw OpalException(
168 "Beam::execute()",
169 "The beam particle hasn't been set. "
170 "Set either \"PARTICLE\" or \"MASS\" and \"CHARGE\".");
171 }
172
173 if (!(itsAttr[NALLOC])) {
174 throw OpalException("Beam::execute()", "\"NALLOC\" must be set.");
175 }
176
177 if (photon) {
178 const double energy = Attributes::getReal(itsAttr[ENERGY]);
179 if (energy <= 0.0) {
180 throw OpalException(
181 "Beam::execute()", "\"ENERGY\" should be greater than 0 for PARTICLE=PHOTON.");
182 }
183 return;
184 }
185
186 // Beam-only validation: each non-photon beam must specify its EMISSIONSOURCELIST.
187 (void)getEmissionSourceListName();
188
189 // Currently supported global process names (extend as new processes are implemented).
190 for (const std::string& name : getGlobalProcessNames()) {
191 if (name != "DECAY") {
192 throw OpalException(
193 "Beam::execute()", "Unsupported entry in \"GLOBALPROCESSES\": \"" + name
194 + "\". Supported values: DECAY.");
195 }
196 }

275void Beam::update() {
276 if (itsAttr[PARTICLE]) {
277 std::string pName = getParticleName();
278 ParticleType pType = ParticleProperties::getParticleType(pName);
279 if (!itsAttr[MASS]) {
280 Attributes::setReal(itsAttr[MASS], ParticleProperties::getParticleMass(pType));
281 }
282 if (!itsAttr[CHARGE]) {
283 Attributes::setReal(itsAttr[CHARGE], ParticleProperties::getParticleCharge(pType));
284 }
285 }
286
287 if (isPhoton()) {
288 reference = PartData();
289 reference.setQ(0.0);
290 reference.setM(0.0);
291 return;
292 }
293
294 // Set up particle reference; convert all to eV for OPALX.
295 double mass = (itsAttr[MASS] ? getMass() : Physics::m_p) * Units::GeV2eV;
296 double charge = itsAttr[CHARGE] ? getCharge() : 1.0;
297
298 reference = PartData(charge, mass, 1.0);
299
300 if (itsAttr[GAMMA]) {
301 double gamma = Attributes::getReal(itsAttr[GAMMA]);
302 if (gamma > 1.0) {
303 reference.setGamma(gamma);
304 } else {
305 throw OpalException("Beam::update()", "\"GAMMA\" should be greater than 1.");
306 }
307 } else if (itsAttr[ENERGY]) {
308 double energy = Attributes::getReal(itsAttr[ENERGY]) * Units::GeV2eV;
309 if (energy > reference.getM()) {
310 reference.setE(energy);
311 } else {
312 throw OpalException("Beam::update()", "\"ENERGY\" should be greater than \"MASS\".");
313 }
314 } else if (itsAttr[PC]) {
315 double pc = Attributes::getReal(itsAttr[PC]) * Units::GeV2eV;
316 if (pc > 0.0) {
317 reference.setP(pc);
318 } else {
319 throw OpalException("Beam::update()", "\"PC\" should be greater than 0.");
320 }
321 }
322
323 // Set default name.
324 if (getOpalName().empty()) setOpalName("UNNAMED_BEAM");
325}

327void Beam::print(std::ostream& os) const {
328 double charge = Attributes::getReal(itsAttr[CHARGE]);
329 os << "* ************* B E A M ************************************************************ "
330 << std::endl;
331 os << "* BEAM " << getOpalName() << '\n'
332 << "* PARTICLE " << Attributes::getString(itsAttr[PARTICLE]) << '\n'
333 << "* REST MASS " << Attributes::getReal(itsAttr[MASS]) << " [GeV]\n"
334 << "* CHARGE " << (charge > 0 ? '+' : '-') << "e * " << std::abs(charge) << " \n"
335 << "* MOMENTUM " << reference.getP() << " [eV/c]\n"
336 << "* MOMENTUM " << Attributes::getReal(itsAttr[PC]) << " [GeV/c]\n"
337 << "* BCHARGE " << Attributes::getReal(itsAttr[BCHARGE]) << " [C]\n"
338 << "* NALLOC " << Attributes::getReal(itsAttr[NALLOC]) << '\n';
339 os << "* ********************************************************************************** "
340 << std::endl;
