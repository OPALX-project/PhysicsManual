95void OpalRingDefinition::update() {
96 OpalElement::update();
97
98 Ring* ring = dynamic_cast<Ring*>(getElement());
99
100 ring->setBeamPhiInit(Attributes::getReal(itsAttr[BEAM_PHIINIT]));
101 ring->setBeamPRInit(Attributes::getReal(itsAttr[BEAM_PRINIT]));
102 ring->setBeamRInit(Attributes::getReal(itsAttr[BEAM_RINIT]) * Units::m2mm);
103 ring->setLatticeRInit(Attributes::getReal(itsAttr[LAT_RINIT]) * Units::m2mm);
104
105 ring->setLatticePhiInit(Attributes::getReal(itsAttr[LAT_PHIINIT]) * Units::deg2rad);
106 ring->setLatticeThetaInit(Attributes::getReal(itsAttr[LAT_THETAINIT]) * Units::deg2rad);
107 ring->setSymmetry(Attributes::getReal(itsAttr[SYMMETRY]));
108 ring->setScale(Attributes::getReal(itsAttr[SCALE]));
109
110 ring->setHarmonicNumber(Attributes::getReal(itsAttr[HARMONIC_NUMBER]));
111 ring->setRFFreq(Attributes::getReal(itsAttr[RFFREQ]));
112 ring->setIsClosed(Attributes::getBool(itsAttr[IS_CLOSED]));
113
114 double minR = -1;
115 double maxR = -1;
116
117 if (itsAttr[MIN_R]) {
118 minR = Attributes::getReal(itsAttr[MIN_R]);
119 if (!itsAttr[MAX_R]) {
120 throw(""); // EXCEPTION
121 }
122 }
123 if (itsAttr[MAX_R]) {
124 maxR = Attributes::getReal(itsAttr[MAX_R]);
125 if (!itsAttr[MIN_R]) {
126 throw(""); // EXCEPTION
127 }
128 ring->setRingAperture(minR, maxR);
129 }
130
132 OpalElement::updateUnknown(ring);
133}
