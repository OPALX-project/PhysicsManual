117bool RFCavity::apply(const std::shared_ptr<ParticleContainer_t>& pc) {
118 // RF parameters (copied to device)
119 double freq = frequency_m;
120 double scale = scale_m + scaleError_m;
121 double phase = phase_m + phaseError_m;
122
123 const double startField = startField_m;
124 const double endField = endField_m;
125
126 // Reference particle time
127 const double t =
128 RefPartBunch_m->getT() + 0.5 * RefPartBunch_m->getdT(); // To be consistent with OPAL
129
130 // RF phase for all particles at this step
131 const double phi = freq * t + phase;
132 const double cosphi = Kokkos::cos(phi);
133 const double sinphi = Kokkos::sin(phi);
134
135 auto* dynamicFieldmap = dynamic_cast<FM2DDynamic*>(fieldmap_m);
136 if (dynamicFieldmap == nullptr) {
137 throw GeneralOpalException(
138 "RFCavity::apply",
139 "RFCavity particle application currently requires an FM2DDynamic field map.");
140 }
141
142 dynamicFieldmap->applyRFField(pc, scale * cosphi, -scale * sinphi, startField, endField);
143
144 return false;
145}

159bool RFCavity::apply(
160 const Vector_t<double, 3>& R, const Vector_t<double, 3>& /*P*/, const double& t,
161 Vector_t<double, 3>& E, Vector_t<double, 3>& B) {
162 if (R(2) >= startField_m && R(2) < endField_m) {
163 Vector_t<double, 3> tmpE({0.0, 0.0, 0.0}), tmpB({0.0, 0.0, 0.0});
164
165 bool outOfBounds = fieldmap_m->getFieldstrength(R, tmpE, tmpB);
166 if (outOfBounds) return getFlagDeleteOnTransverseExit();
167
168 E += (scale_m + scaleError_m) * std::cos(frequency_m * t + phase_m + phaseError_m) * tmpE;
169 B -= (scale_m + scaleError_m) * std::sin(frequency_m * t + phase_m + phaseError_m) * tmpB;
170 }
171 return false;
172}

351void RFCavity::getMomentaKick(
352 const double normalRadius, double momentum[], const double t, const double dtCorrt,
353 const int PID, const double restMass, const int chargenumber) {
354 double derivate;
355
356 double momentum2 =
357 momentum[0] * momentum[0] + momentum[1] * momentum[1] + momentum[2] * momentum[2];
358 double betgam = std::sqrt(momentum2);
359
360 double gamma = std::sqrt(1.0 + momentum2);
361 double beta = betgam / gamma;
362
363 double Voltage = spline(normalRadius, &derivate) * scale_m * Units::MVpm2Vpm;
364
365 double Ufactor = 1.0;
366
367 double frequency = frequency_m * frequencyTD_m->getValue(t);
368
369 if (gapwidth_m > 0.0) {
370 double transit_factor = 0.5 * frequency * gapwidth_m * Units::mm2m / (Physics::c * beta);
371 Ufactor = std::sin(transit_factor) / transit_factor;
372 }
373
374 Voltage *= Ufactor;
375 // rad/s, ns --> rad
376 double nphase = (frequency * (t + dtCorrt) * Units::ns2s) - phi0_m * Units::deg2rad;
377 double dgam = Voltage * std::cos(nphase) / (restMass);
378
379 double tempdegree = std::fmod(nphase * Units::rad2deg, 360.0);
380 if (tempdegree > 270.0) tempdegree -= 360.0;
381
382 gamma += dgam;
383
384 double newmomentum2 = std::pow(gamma, 2) - 1.0;
385
386 double pr = momentum[0] * cosAngle_m + momentum[1] * sinAngle_m;
387 double ptheta = std::sqrt(newmomentum2 - std::pow(pr, 2));
388 double px = pr * cosAngle_m - ptheta * sinAngle_m; // x
389 double py = pr * sinAngle_m + ptheta * cosAngle_m; // y
390
391 double rotate = -derivate * (scale_m * Units::MVpm2Vpm) / ((rmax_m - rmin_m) * Units::mm2m)
392 * std::sin(nphase) / (frequency * Physics::two_pi)
393 / (betgam * restMass / Physics::c / chargenumber); // radian
394
396 momentum[0] = std::cos(rotate) * px + std::sin(rotate) * py;
397 momentum[1] = -std::sin(rotate) * px + std::cos(rotate) * py;
398
399 if (PID == 0) {
400 Inform m("OPAL", *gmsg, ippl::Comm->rank());
401
402 m << "* Cavity " << getName() << " Phase= " << tempdegree
403 << " [deg] transit time factor= " << Ufactor
404 << " dE= " << dgam * restMass * Units::eV2MeV << " [MeV]"
405 << " E_kin= " << (gamma - 1.0) * restMass * Units::eV2MeV
406 << " [MeV] Time dep freq = " << frequencyTD_m->getValue(t) << endl;
407 }
408}

410/* cubic spline subrutine */
411double RFCavity::spline(double z, double* za) {
412 double splint;
413
414 // domain-test and handling of case "1-support-point"
415 if (num_points_m < 1) {
416 throw GeneralOpalException("RFCavity::spline", "no support points!");
417 }
418 if (num_points_m == 1) {
419 splint = RNormal_m[0];
420 *za = 0.0;
421 return splint;
422 }
423
424 // search the two support-points
425 int il, ih;
426 il = 0;
427 ih = num_points_m - 1;
428 while ((ih - il) > 1) {
429 int i = (int)((il + ih) / 2.0);
430 if (z < RNormal_m[i]) {
431 ih = i;
432 } else if (z > RNormal_m[i]) {
433 il = i;
434 } else if (z == RNormal_m[i]) {
435 il = i;
436 ih = i + 1;
437 break;
438 }
439 }
440
441 double x1 = RNormal_m[il];
442 double x2 = RNormal_m[ih];
443 double y1 = VrNormal_m[il];
444 double y2 = VrNormal_m[ih];
445 double y1a = DvDr_m[il];
446 double y2a = DvDr_m[ih];
447 //
448 // determination of the requested function-values and its derivatives
449 //
450 double dx = x2 - x1;
451 double dy = y2 - y1;
452 double u = (z - x1) / dx;
453 double u2 = u * u;
454 double u3 = u2 * u;
455 double dy2 = -2.0 * dy;
456 double ya2 = y2a + 2.0 * y1a;
457 double dy3 = 3.0 * dy;
458 double ya3 = y2a + y1a;
459 double yb2 = dy2 + dx * ya3;
460 double yb4 = dy3 - dx * ya2;
461 splint = y1 + u * dx * y1a + u2 * yb4 + u3 * yb2;
462 *za = y1a + 2.0 * u / dx * yb4 + 3.0 * u2 / dx * yb2;
463 // if(m>=1) za=y1a+2.0*u/dx*yb4+3.0*u2/dx*yb2;
464 // if(m>=2) za[1]=2.0/dx2*yb4+6.0*u/dx2*yb2;
465 // if(m>=3) za[2]=6.0/dx3*yb2;
466
467 return splint;
468}

477double RFCavity::getAutoPhaseEstimateFallback(double E0, double t0, double q, double mass) {
478 const double dt = 1e-13;
479 const double p0 = Util::getBetaGamma(E0, mass);
480 const double origPhase = getPhasem();
481 double dphi = Physics::pi / 18;
482
483 double phi = 0.0;
484 setPhasem(phi);
485 std::pair<double, double> ret = trackOnAxisParticle(p0, t0, dt, q, mass);
486 double phimax = 0.0;
487 double Emax = Util::getKineticEnergy(Vector_t<double, 3>({0.0, 0.0, ret.first}), mass);
488 phi += dphi;
489
490 for (unsigned int j = 0; j < 2; ++j) {
491 for (unsigned int i = 0; i < 36; ++i, phi += dphi) {
492 setPhasem(phi);
493 ret = trackOnAxisParticle(p0, t0, dt, q, mass);
494 double Ekin = Util::getKineticEnergy(Vector_t<double, 3>({0.0, 0.0, ret.first}), mass);
495 if (Ekin > Emax) {
496 Emax = Ekin;
497 phimax = phi;
498 }
499 }
500
501 phi = phimax - dphi;
502 dphi = dphi / 17.5;
503 }
504
505 phimax = phimax - std::round(phimax / Physics::two_pi) * Physics::two_pi;
506 phimax = std::fmod(phimax, Physics::two_pi);
507
508 const int prevPrecision = ippl::Info->precision(8);
509 Inform m("RFCavity::getAutoPhaseEstimateFallback");
510 m << level2 << "estimated phase= " << phimax << " rad = " << phimax * Units::rad2deg << " deg\n"
511 << "Ekin= " << Emax << " MeV" << std::setprecision(prevPrecision) << "\n"
512 << endl;
513
514 setPhasem(origPhase);
515 return phimax;
516}

518double RFCavity::getAutoPhaseEstimate(
519 const double& E0, const double& t0, const double& q, const double& mass) {
520 std::vector<double> t, E, t2, E2;
521 std::vector<double> F;
522 std::vector<std::pair<double, double> > G;
523 gsl_spline* onAxisInterpolants;
524 gsl_interp_accel* onAxisAccel;
525
526 double phi = 0.0, tmp_phi, dphi = 0.5 * Units::deg2rad;
527 double dz = 1.0, length = 0.0;
528 fieldmap_m->getOnaxisEz(G);
529 if (G.size() == 0) return 0.0;
530 double begin = (G.front()).first;
531 double end = (G.back()).first;
532 std::unique_ptr<double[]> zvals(new double[G.size()]);
533 std::unique_ptr<double[]> onAxisField(new double[G.size()]);
534
535 for (size_t j = 0; j < G.size(); ++j) {
536 zvals[j] = G[j].first;
537 onAxisField[j] = G[j].second;
538 }
539 onAxisInterpolants = gsl_spline_alloc(gsl_interp_cspline, G.size());
540 onAxisAccel = gsl_interp_accel_alloc();
541 gsl_spline_init(onAxisInterpolants, zvals.get(), onAxisField.get(), G.size());
542
543 length = end - begin;
544 dz = length / G.size();
545
546 G.clear();
547
548 unsigned int N = (int)std::floor(length / dz + 1);
549 dz = length / N;
550
551 F.resize(N);
552 double z = begin;
553 for (size_t j = 0; j < N; ++j, z += dz) {
554 F[j] = gsl_spline_eval(onAxisInterpolants, z, onAxisAccel);
555 }
556 gsl_spline_free(onAxisInterpolants);
557 gsl_interp_accel_free(onAxisAccel);
558
559 t.resize(N, t0);
560 t2.resize(N, t0);
561 E.resize(N, E0);
562 E2.resize(N, E0);
563
564 z = begin + dz;
565 for (unsigned int i = 1; i < N; ++i, z += dz) {
566 E[i] = E[i - 1] + dz * scale_m / mass;
567 E2[i] = E[i];
568 }
569
570 for (int iter = 0; iter < 10; ++iter) {
571 double A = 0.0;
572 double B = 0.0;
573 for (unsigned int i = 1; i < N; ++i) {
574 t[i] = t[i - 1] + getdT(i, E, dz, mass);
575 t2[i] = t2[i - 1] + getdT(i, E2, dz, mass);
576 A += scale_m * (1. + frequency_m * (t2[i] - t[i]) / dphi)
577 * getdA(i, t, dz, frequency_m, F);
578 B += scale_m * (1. + frequency_m * (t2[i] - t[i]) / dphi)
579 * getdB(i, t, dz, frequency_m, F);
580 }
581
582 if (std::abs(B) > 0.0000001) {
583 tmp_phi = std::atan(A / B);
584 } else {
585 tmp_phi = Physics::pi / 2;
586 }
587 if (q * (A * std::sin(tmp_phi) + B * std::cos(tmp_phi)) < 0) {
588 tmp_phi += Physics::pi;
589 }
590
591 if (std::abs(phi - tmp_phi) < frequency_m * (t[N - 1] - t[0]) / (10 * N)) {
592 for (unsigned int i = 1; i < N; ++i) {
593 E[i] = E[i - 1];
594 E[i] += q * scale_m * getdE(i, t, dz, phi, frequency_m, F);
595 }
596 const int prevPrecision = ippl::Info->precision(8);
597 Inform m("RFCavity::getAutoPhaseEstimate");
598 m << level2 << "estimated phase= " << tmp_phi << " rad = " << tmp_phi * Units::rad2deg
599 << " deg\n"
600 << "Ekin= " << E[N - 1] << " MeV" << std::setprecision(prevPrecision) << "\n"
601 << endl;
602
603 return tmp_phi;
604 }
605 phi = tmp_phi - std::round(tmp_phi / Physics::two_pi) * Physics::two_pi;
606
607 for (unsigned int i = 1; i < N; ++i) {
608 E[i] = E[i - 1];
609 E2[i] = E2[i - 1];
610 E[i] += q * scale_m * getdE(i, t, dz, phi, frequency_m, F);
611 E2[i] += q * scale_m * getdE(i, t2, dz, phi + dphi, frequency_m, F);
612 double a = E[i], b = E2[i];
613 if (std::isnan(a) || std::isnan(b)) {
614 return getAutoPhaseEstimateFallback(E0, t0, q, mass);
615 }
616 t[i] = t[i - 1] + getdT(i, E, dz, mass);
617 t2[i] = t2[i - 1] + getdT(i, E2, dz, mass);
618
619 E[i] = E[i - 1];
620 E2[i] = E2[i - 1];
621 E[i] += q * scale_m * getdE(i, t, dz, phi, frequency_m, F);
622 E2[i] += q * scale_m * getdE(i, t2, dz, phi + dphi, frequency_m, F);
623 }
624
625 double cosine_part = 0.0, sine_part = 0.0;
626 double p0 = Util::getBetaGamma(E0, mass);
627 cosine_part += scale_m * std::cos(frequency_m * t0) * F[0];
628 sine_part += scale_m * std::sin(frequency_m * t0) * F[0];
629
630 double totalEz0 = std::cos(phi) * cosine_part - std::sin(phi) * sine_part;
631
632 if (p0 + q * totalEz0 * (t[1] - t[0]) * Physics::c / mass < 0) {
633 // make totalEz0 = 0
634 tmp_phi = std::atan(cosine_part / sine_part);
635 if (std::abs(tmp_phi - phi) > Physics::pi) {
636 phi = tmp_phi + Physics::pi;
637 } else {
638 phi = tmp_phi;
639 }
640 }
641 }
642
643 const int prevPrecision = ippl::Info->precision(8);
644 Inform m("RFCavity::getAutoPhaseEstimate");
645 m << level2 << "estimated phase= " << tmp_phi << " rad = " << tmp_phi * Units::rad2deg
646 << " deg\n"
647 << "Ekin= " << E[N - 1] << " MeV" << std::setprecision(prevPrecision) << "\n"
648 << endl;
649
650 return phi;
651}

653std::pair<double, double> RFCavity::trackOnAxisParticle(
654 const double& p0, const double& t0, const double& dt, const double& /*q*/,
655 const double& mass, std::ofstream* out) {
656 Vector_t<double, 3> p({0, 0, p0});
657 double t = t0;
658
659 BorisPusher integrator;
660 const PartData& ref = *RefPartBunch_m->getParticleContainer()->getReference();
661 const double cdt = Physics::c * dt;
662 const double zbegin = startField_m;
663 const double zend = endField_m;
664
665 Vector_t<double, 3> z({0.0, 0.0, zbegin});
666 double dz = 0.5 * p(2) / Util::getGamma(p) * cdt;
667 Vector_t<double, 3> Ef(0.0), Bf(0.0);
668
669 if (out)
670 *out << std::setw(18) << z[2] << std::setw(18) << Util::getKineticEnergy(p, mass)
671 << std::endl;
672 while (z(2) + dz < zend && z(2) + dz > zbegin) {
673 z /= cdt;
674 integrator.push(z, p, dt);
675 z *= cdt;
676
677 Ef = 0.0;
678 Bf = 0.0;
679 if (z(2) >= zbegin && z(2) <= zend) {
680 applyToReferenceParticle(z, p, t + 0.5 * dt, Ef, Bf);
681 }
682
683 integrator.kick(z, p, Ef, Bf, dt, ref.getM(), ref.getQ());
684
685 dz = 0.5 * p(2) / std::sqrt(1.0 + dot(p, p)) * cdt;
686 z /= cdt;
687
688 integrator.push(z, p, dt);
689 z *= cdt;
690 t += dt;
691
692 if (out)
693 *out << std::setw(18) << z[2] << std::setw(18) << Util::getKineticEnergy(p, mass)
694 << std::endl;
695 }
696
697 const double beta = std::sqrt(1. - 1 / (dot(p, p) + 1.));
698 const double tErr = (z(2) - zend) / (Physics::c * beta);
699 return std::pair<double, double>(p(2), t - tErr);
700}
