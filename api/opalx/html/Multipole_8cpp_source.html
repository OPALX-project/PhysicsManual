37 double factorial(unsigned int n) {
38 if (n == 0) return 1.0;
39 if (n == 1) return 1.0;
40 if (n == 2) return 2.0;
41 if (n == 3) return 6.0;
42 if (n == 4) return 24.0;
43 if (n == 5) return 120.0;
44 if (n > 5) Kokkos::abort("factorial out of bounds");
45 return 0;
46 }

170void Multipole::setNormalComponent(int n, double v, double vError)
171{
172 if (n < 0) {
173 throw GeneralOpalException("Multipole::setNormalComponent",
174 "component index " + std::to_string(n) + " out of bounds");
175 }
176
177 if (n >= max_NormalComponent_m) {
178 max_NormalComponent_m = n+1;
179 Kokkos::resize(NormalComponents, max_NormalComponent_m);
180 Kokkos::resize(NormalComponentErrors, max_NormalComponent_m);
181 }
182
183 double valComp, valErr;
184
185 // TODO: Check the physics here (copied from OPAL)
186 if (n == DIPOLE) {
187 valComp = (v + vError) / 2.0;
188 valErr = valComp;
189 } else {
190 double fact = factorial(n);
191 valComp = (v + vError) / fact;
192 valErr = vError / fact;
193 }
194
195 // Copy to Device
196 auto sub_comp = Kokkos::subview(NormalComponents, n);
197 auto sub_err = Kokkos::subview(NormalComponentErrors, n);
198
199 Kokkos::deep_copy(sub_comp, valComp);
200 Kokkos::deep_copy(sub_err, valErr);

440KOKKOS_INLINE_FUNCTION
441void Multipole::computeField(
442 Vector_t<double, 3> R,
443 Vector_t<double, 3>& /*E*/,
444 Vector_t<double, 3>& B,
445 const Kokkos::View<double*> NormalComponents,
446 const Kokkos::View<double*> SkewComponents,
447 int max_NormalComponent,
448 int max_SkewComponent)
449{
450 // Use primitive double arrays instead of Vector_t to avoid constructor/destructor calls on GPU stack
451 double Rn_x[MAX_MP_ORDER + 1];
452 double Rn_y[MAX_MP_ORDER + 1];
453 double fact[MAX_MP_ORDER + 1];
454
455 // --- NORMAL COMPONENTS ---
456 {
457 Rn_x[0] = 1.0;
458 Rn_y[0] = 1.0; // Equivalent to Vector_t(1.0)
459 fact[0] = 1.0;
460
461 // Use local variable for loop bound to help optimizer
462 int count = (max_NormalComponent > MAX_MP_ORDER) ?
463 MAX_MP_ORDER : max_NormalComponent;
464
465 for (int i = 0; i < count; ++i) {
466 // Access Kokkos::View directly (device memory)
467 double norm = NormalComponents(i);
468
469 switch (i) {
470 case DIPOLE:
471 B(1) += norm;
472 break;
473
474 case QUADRUPOLE:
475 B(0) += norm * R(1);
476 B(1) += norm * R(0);
477 break;
478 }
479
480 Rn_x[i + 1] = Rn_x[i] * R(0);
481 Rn_y[i + 1] = Rn_y[i] * R(1);
482 fact[i + 1] = fact[i] / (double)(i + 1);
483 }
484 }
485
486 // --- SKEW COMPONENTS ---
487 {
488 Rn_x[0] = 1.0;
489 Rn_y[0] = 1.0;
490 fact[0] = 1.0;
491
492 int count = (max_SkewComponent > MAX_MP_ORDER) ?
493 MAX_MP_ORDER : max_SkewComponent;
494
495 for (int i = 0; i < count; ++i) {
496 double skew = SkewComponents(i);
497
498 switch (i) {
499 case DIPOLE:
500 B(0) -= skew;
501 break;
502
503 case QUADRUPOLE:
504 B(0) -= skew * R(0);
505 B(1) += skew * R(1);
506 break;
507 }
508
509 Rn_x[i + 1] = Rn_x[i] * R(0);
510 Rn_y[i + 1] = Rn_y[i] * R(1);
511 fact[i + 1] = fact[i] / (double)(i + 1);
512 }
513 }

525void Multipole::computeFieldHost(
526 const Vector_t<double, 3> R,
527 Vector_t<double, 3>& /*E*/,
528 Vector_t<double, 3>& B) const
529{
530 Vector_t<double, 3> Rn[MAX_MP_ORDER + 1];
531 double fact[MAX_MP_ORDER + 1];
532
533 auto NormalComponents_host = Kokkos::create_mirror_view(NormalComponents);
534 Kokkos::deep_copy(NormalComponents_host, NormalComponents);
535
536 auto SkewComponents_host = Kokkos::create_mirror_view(SkewComponents);
537 Kokkos::deep_copy(SkewComponents_host, SkewComponents);
538
539 // --- NORMAL COMPONENTS ---
540 {
541 Rn[0] = Vector_t<double, 3>(1.0);
542 fact[0] = 1.0;
543
544 // Use local variable for loop bound to help optimizer
545 int count = (max_NormalComponent_m > MAX_MP_ORDER) ?
546 MAX_MP_ORDER : max_NormalComponent_m;
547
548 for (int i = 0; i < count; ++i) {
549 // Access Kokkos::View directly (device memory)
550 double norm = NormalComponents_host(i);
551
552 switch (i) {
553 case DIPOLE:
554 B(1) += norm;
555 break;
556
557 case QUADRUPOLE:
558 B(0) += norm * R(1);
559 B(1) += norm * R(0);
560 break;
561
562 case SEXTUPOLE:
563 B(0) += 2 * norm * R(0) * R(1);
564 B(1) += norm * (Rn[2](0) - Rn[2](1));
565 break;
566
567 case OCTUPOLE:
568 B(0) += norm * (3 * Rn[2](0) * Rn[1](1) - Rn[3](1));
569 B(1) += norm * (Rn[3](0) - 3 * Rn[1](0) * Rn[2](1));
570 break;
571
572 case DECAPOLE:
573 B(0) += 4 * norm * (Rn[3](0) * Rn[1](1) - Rn[1](0) * Rn[3](1));
574 B(1) += norm * (Rn[4](0) - 6 * Rn[2](0) * Rn[2](1) + Rn[4](1));
575 break;
576
577 default: {
578 double powMinusOne = 1.0;
579 double Bx = 0.0, By = 0.0;
580 for (int j = 1; j <= (i + 1) / 2; ++j) {
581 Bx += powMinusOne * norm
582 * (Rn[i - 2 * j + 1](0) * fact[i - 2 * j + 1] * Rn[2 * j - 1](1)
583 * fact[2 * j - 1]);
584 By += powMinusOne * norm
585 * (Rn[i - 2 * j + 2](0) * fact[i - 2 * j + 2] * Rn[2 * j - 2](1)
586 * fact[2 * j - 2]);
587 powMinusOne *= -1.0;
588 }
589
590 if ((i + 1) / 2 == i / 2) {
591 int j = (i + 2) / 2;
592 By += powMinusOne * norm
593 * (Rn[i - 2 * j + 2](0) * fact[i - 2 * j + 2] * Rn[2 * j - 2](1)
594 * fact[2 * j - 2]);
595 }
596 B(0) += Bx;
597 B(1) += By;
598 }
599 }
600
601 Rn[i + 1](0) = Rn[i](0) * R(0);
602 Rn[i + 1](1) = Rn[i](1) * R(1);
603 fact[i + 1] = fact[i] / (double)(i + 1);
604 }
605 }
606
607 // --- SKEW COMPONENTS ---
608 {
609 Rn[0] = Vector_t<double, 3>(1.0);
610 fact[0] = 1.0;
611
612 int count = (max_SkewComponent_m > MAX_MP_ORDER) ?
613 MAX_MP_ORDER : max_SkewComponent_m;
614
615 for (int i = 0; i < count; ++i) {
616 double skew = SkewComponents_host(i);
617
618 switch (i) {
619 case DIPOLE:
620 B(0) -= skew;
621 break;
622
623 case QUADRUPOLE:
624 B(0) -= skew * R(0);
625 B(1) += skew * R(1);
626 break;
627
628 case SEXTUPOLE:
629 B(0) -= skew * (Rn[2](0) - Rn[2](1));
630 B(1) += 2 * skew * R(0) * R(1);
631 break;
632
633 case OCTUPOLE:
634 B(0) -= skew * (Rn[3](0) - 3 * Rn[1](0) * Rn[2](1));
635 B(1) += skew * (3 * Rn[2](0) * Rn[1](1) - Rn[3](1));
636 break;
637
638 case DECAPOLE:
639 B(0) -= skew * (Rn[4](0) - 6 * Rn[2](0) * Rn[2](1) + Rn[4](1));
640 B(1) += 4 * skew * (Rn[3](0) * Rn[1](1) - Rn[1](0) * Rn[3](1));
641 break;
642
643 default: {
644 double powMinusOne = 1.0;
645 double Bx = 0.0, By = 0.0;
646 for (int j = 1; j <= (i + 1) / 2; ++j) {
647 Bx -= powMinusOne * skew
648 * (Rn[i - 2 * j + 2](0) * fact[i - 2 * j + 2] * Rn[2 * j - 2](1)
649 * fact[2 * j - 2]);
650 By += powMinusOne * skew
651 * (Rn[i - 2 * j + 1](0) * fact[i - 2 * j + 1] * Rn[2 * j - 1](1)
652 * fact[2 * j - 1]);
653 powMinusOne *= -1.0;
654 }
655
656 if ((i + 1) / 2 == i / 2) {
657 int j = (i + 2) / 2;
658 Bx -= powMinusOne * skew
659 * (Rn[i - 2 * j + 2](0) * fact[i - 2 * j + 2] * Rn[2 * j - 2](1)
660 * fact[2 * j - 2]);
661 }
662
663 B(0) += Bx;
664 B(1) += By;
665 }
666 }
667
668 Rn[i + 1](0) = Rn[i](0) * R(0);
669 Rn[i + 1](1) = Rn[i](1) * R(1);
670 fact[i + 1] = fact[i] / (double)(i + 1);
671 }
672 }
