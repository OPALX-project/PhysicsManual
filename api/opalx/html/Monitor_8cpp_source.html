221void Monitor::writeStatistics() {
222 if (statFileEntries_sm.size() == 0)
223 return;
224
225 std::string fileName =
226 OpalData::getInstance()->getInputBasename() + std::string("_Monitors.stat");
227 auto instance = OpalData::getInstance();
228 bool hasPriorTrack = instance->hasPriorTrack();
229 bool inRestartRun = instance->inRestartRun();
230
231 auto it = statFileEntries_sm.begin();
232 double spos = it->first;
233 Util::rewindLinesSDDS(fileName, spos, false);
234
235 MonitorStatisticsWriter writer(fileName, hasPriorTrack || inRestartRun);
236
237 for (const auto& entry : statFileEntries_sm) {
238 writer.addRow(entry.second);
239 }
240
241 statFileEntries_sm.clear();
