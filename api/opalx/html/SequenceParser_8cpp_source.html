46void SequenceParser::parse(Statement &stat) const {
47 if (stat.keyword("ENDSEQUENCE")) {
48 // Finish setting up of sequence.
49 if (okFlag) {
50 try {
51 // Finish parsing the ENDSEQUENCE statement.
52 itsSequence->Object::parse(stat);
53 parseEnd(stat);
54
55 // Finish setting up of sequence.
56 itsSequence->addEndMarkers(itsLine);
57 fillPositions();
58 itsSequence->insertDrifts(itsLine);
59 itsSequence->storeLine(itsLine);
60 OpalData::getInstance()->makeDirty(itsSequence);
61 } catch(...) {
62 // In case of error, erase sequence and return to previous parser.
63 itsLine.clear();
64 stop();
65 throw;
66 }
67 }
68
69 // Return to previous parser.
70 itsLine.clear();
71 stop();
72 } else {
73 // Backtrack to start and parse new member.
74 try {
75 stat.start();
76 parseMember(stat);
77 } catch(...) {
78 okFlag = false;
79 throw;
80 }
81 }
82}

85void SequenceParser::fillPositions() const {
86 findFromPositions();
87 findNeighbourPositions();
88
89 while(! references.empty()) {
90 int count = 0;
91
92 for (RefList::iterator i = references.begin();
93 i != references.end();) {
94 if (i->fromPosition != 0 &&
95 i->fromPosition->itsFlag == SequenceMember::ABSOLUTE) {
96 for (std::list<SequenceMember *>::iterator j = i->itsList.begin();
97 j != i->itsList.end(); ++j) {
98 (*j)->itsPosition += i->fromPosition->itsPosition;
99 (*j)->itsFlag = SequenceMember::ABSOLUTE;
100 }
101
102 ++count;
103 i = references.erase(i);
104 } else {
105 ++i;
106 }
107 }
108
109 if (count == 0) {
110 throw OpalException("SequenceParser::fillPositions()",
111 "Circular \"FROM\" clauses.");
112 }
113 }

131void SequenceParser::findNeighbourPositions() const {
132 Sequence::ReferenceType type = Sequence::IS_CENTRE;
133 Attribute *attr = itsSequence->findAttribute("REFER");
134 std::string ref = Attributes::getString(*attr);
135
136 if (ref == "ENTRY") {
137 type = Sequence::IS_ENTRY;
138 } else if (ref == "CENTRE" || ref.empty()) {
139 type = Sequence::IS_CENTRE;
140 } else if (ref == "EXIT") {
141 type = Sequence::IS_EXIT;
142 } else {
143 throw OpalException("SequenceParser::findNeighbourPositions()",
144 "Unknown reference flag \"" + ref +
145 "\" in <sequence> \"" +
146 itsSequence->getOpalName() + "\".");
147 }
148
149 for (TLine::iterator i = itsLine.begin(); i != itsLine.end(); ++i) {
150 TLine::iterator j = i;
151 switch(i->itsFlag) {
152
153 case SequenceMember::DRIFT:
154 case SequenceMember::IMMEDIATE:
155 // Add scaled length of current.
156 if (type != Sequence::IS_ENTRY) {
157 std::string name = i->getElement()->getName();
158 Element *elem = Element::find(name);
159 double length = elem->getLength();
160
161 if (type == Sequence::IS_CENTRE) {
162 i->itsPosition += length / 2.0;
163 } else if (type == Sequence::IS_EXIT) {
164 i->itsPosition += length;
165 }
166 }
167
168 // Add scaled length of previous.
169 if (j != itsLine.begin()) {
170 --j;
171 if (type != Sequence::IS_EXIT) {
172 std::string name = j->getElement()->getName();
173 Element *elem = Element::find(name);
174 double length = elem->getLength();
175
176 if (type == Sequence::IS_ENTRY) {
177 i->itsPosition += length;
178 } else if (type == Sequence::IS_CENTRE) {
179 i->itsPosition += length / 2.0;
180 }
181 }
182
183 // Current position is relative to previous.
184 Reference ref("DRIFT", &(*i));
185 ref.fromPosition = &(*j);
186 references.push_back(ref);
187 } else {
188 // Current position is relative to sequence begin.
189 i->itsFlag = SequenceMember::ABSOLUTE;
190 }
191 break;
192
193 case SequenceMember::PREVIOUS:
194 if (j != itsLine.begin()) {
195 --j;
196 Reference ref("PREVIOUS", &(*i));
197 ref.fromPosition = &(*j);
198 references.push_back(ref);
199 } else {
200 i->itsFlag = SequenceMember::ABSOLUTE;
201 }
202 break;
203
204 case SequenceMember::NEXT:
205 ++j;
206 if (j != itsLine.end()) {
207 // Current position is relative to *j;
208 Reference ref("NEXT", &(*i));
209 ref.fromPosition = &(*j);
210 references.push_back(ref);
211 } else {
212 // Current position is relative to end of sequence.
213 Attribute *attr = itsSequence->findAttribute("L");
214 i->itsPosition += Attributes::getReal(*attr);
215 i->itsFlag = SequenceMember::ABSOLUTE;
216 }
217 break;
218
219 case SequenceMember::END: {
220 // Current position is relative to end of sequence.
221 Attribute *attr = itsSequence->findAttribute("L");
222 i->itsPosition += Attributes::getReal(*attr);
223 i->itsFlag = SequenceMember::ABSOLUTE;
224 }
225 break;
226
227 default:
228 ;
229 }
230 }

234void SequenceParser::parseMember(Statement &stat) const {
235 std::string objName;
236 std::string clsName;
237 SequenceMember member;
238
239 // Read object identifier.
240 if (stat.delimiter('-')) {
241 member.setReflectionFlag(true);
242 clsName = Expressions::parseString(stat, "Class name expected.");
243 member.itsType = SequenceMember::GLOBAL;
244 } else {
245 clsName = Expressions::parseString(stat, "Object or class name expected.");
246
247 if (stat.delimiter(':')) {
248 member.itsType = SequenceMember::LOCAL;
249 member.setReflectionFlag(stat.delimiter('-'));
250 objName = clsName;
251 clsName = Expressions::parseString(stat, "Class name expected.");
252 } else {
253 member.itsType = SequenceMember::GLOBAL;
254 }
255 }
256
257 // Find exemplar object.
258 if (Object *obj = OpalData::getInstance()->find(clsName)) {
259 std::shared_ptr<Object> copy;
260 if (obj) {
261 if (member.itsType == SequenceMember::LOCAL) {
262 // We must make a copy or a macro instance.
263 if (objName == itsSequence->getOpalName()) {
264 throw ParseError("SequenceParser::parseMember()",
265 "You cannot redefine \"" + objName +
266 "\" within the sequence of the same name.");
267 } else if (clsName == itsSequence->getOpalName()) {
268 throw ParseError("SequenceParser::parseMember()",
269 "You cannot refer to \"" + objName +
270 "\" within the sequence of the same name.");
271 } else if (OpalData::getInstance()->find(objName)) {
272 throw ParseError("SequenceParser::parseMember()",
273 "You cannot redefine \"" + objName +
274 "\" within a sequence.");
275 }
276
277 if (stat.delimiter('(')) {
278 copy = std::shared_ptr<Object>(obj->makeInstance(objName, stat, this));
279 } else if (BeamSequence *line = dynamic_cast<BeamSequence *>(obj)) {
280 copy = std::shared_ptr<Object>(line->copy(objName));
281 } else {
282 copy = std::shared_ptr<Object>(obj->clone(objName));
283 }
284 } else {
285 // We can use the global object.
286 copy = std::shared_ptr<Object>(obj);
287 }
288 }
289
290 // Test for correct object type.
291 if (Element *elem = dynamic_cast<Element *>(&*copy)) {
292 ElementBase *base = elem->getElement();
293 member.setElement(base->copyStructure());
294
295 // ada 4.5 2000 to speed up matching, add a pointer to
296 // opal elements in order to avoid serching the opal elements
297 member.OpalElement = std::shared_ptr<Element>(elem);
298
299 itsLine.push_back(member);
300
301 // Parse position and attributes.
302 parsePosition(stat, *copy, member.itsType == SequenceMember::LOCAL);
303 parseEnd(stat);
304
305 // For local instantiation define element.
306 if (member.itsType == SequenceMember::LOCAL) OpalData::getInstance()->define(&*copy);
307 } else {
308 throw ParseError("SequenceParser::parseMember()", "Object \"" +
309 objName + "\" is not an element.");
310 }
311 } else {
312 throw ParseError("SequenceParser::parseMember()",
313 "Element class name \"" + clsName + "\" is unknown.");
314 }

318void SequenceParser::parsePosition
319(Statement &stat, Object &elem, bool defined) const {
320 SequenceMember &member = itsLine.back();
321 double position = 0.0;
322 std::string fromName;
323 int flag = 0;
324
325 while(stat.delimiter(',')) {
326 std::string name =
327 Expressions::parseString(stat, "Attribute name expected.");
328
329 if (name == "AT") {
330 Expressions::parseDelimiter(stat, '=');
331 position = Expressions::parseRealConst(stat);
332 flag |= (flag & 0x0001) ? 0x0008 : 0x0001;
333 } else if (name == "DRIFT") {
334 Expressions::parseDelimiter(stat, '=');
335 position = Expressions::parseRealConst(stat);
336 flag |= (flag & 0x0002) ? 0x0008 : 0x0002;
337 } else if (name == "FROM") {
338 Expressions::parseDelimiter(stat, '=');
339 if (stat.delimiter('#')) {
340 if (stat.keyword("S")) {
341 fromName = "#S";
342 } else if (stat.keyword("E")) {
343 fromName = "#E";
344 } else {
345 throw ParseError("SequenceParser::parsePosition()",
346 "Expected 'S' or 'E'.");
347 }
348 } else {
349 fromName = Expressions::parseString
350 (stat, "Expected <name>, \"#S\", or \"#E\".");
351 }
352 flag |= (flag & 0x0004) ? 0x0008 : 0x0004;
353 } else if (defined) {
354 // May be element attribute.
355 if (Attribute *attr = elem.findAttribute(name)) {
356 if (stat.delimiter('=')) {
357 attr->parse(stat, true);
358 } else if (stat.delimiter(":=")) {
359 attr->parse(stat, false);
360 } else {
361 throw ParseError("SequenceParser::parsePosition()",
362 "Delimiter \"=\" or \":=\" expected.");
363 }
364 } else {
365 throw ParseError("SequenceParser::parsePosition()",
366 "Element \"" + elem.getOpalName() +
367 "\" has no attribute \"" + name + "\".");
368 }
369 } else {
370 throw ParseError("SequenceParser::parsePosition()",
371 "Overriding attributes not permitted here.");
372 }
373 }
374
375 // Check for valid data and store.
376 member.itsPosition = 0.0;
377
378 switch(flag) {
379
380 case 0x0001:
381 // "at=...".
382 member.itsPosition = position;
383 break;
384
385 case 0x0002:
386 // "drift=...". Set distance to drift length.
387 member.itsPosition = position;
388 member.itsFlag = SequenceMember::DRIFT;
389 break;
390
391 case 0x0000:
392 // No position data.
393 member.itsPosition = 0.0;
394 member.itsFlag = SequenceMember::IMMEDIATE;
395 break;
396
397 case 0x0005:
398 // "at=...,from=...".
399 member.itsPosition = position;
400
401 if (fromName == "PREVIOUS") {
402 // Position refers to previous element.
403 member.itsFlag = SequenceMember::PREVIOUS;
404 } else if (fromName == "NEXT") {
405 // Position refers to next element.
406 member.itsFlag = SequenceMember::NEXT;
407 } else if (fromName == "#S") {
408 // Position refers to begin of sequence.
409 member.itsFlag = SequenceMember::ABSOLUTE;
410 } else if (fromName == "#E") {
411 // Position refers to end of sequence.
412 member.itsFlag = SequenceMember::END;
413 } else {
414 // Position refers to named element.
415 member.itsFlag = SequenceMember::FROM;
416
417 // Find reference to fromName.
418 for (RefList::iterator i = references.begin();
419 i != references.end(); ++i) {
420 if (i->fromName == fromName) {
421 // Build reference to a new "FROM" name.
422 i->itsList.push_back(&member);
423 return;
424 }
425 }
426
427 // Add reference to an already known "FROM" name.
428 Reference ref;
429 ref.fromName = fromName;
430 ref.itsList.push_back(&member);
431 references.push_back(ref);
432 }
433
434 break;
435
436 default:
437 throw ParseError("SequenceParser:parsePosition()",
438 "Conflicting keywords.");
439 }
