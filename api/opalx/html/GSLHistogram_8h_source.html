29// 1D Histogram
31class gsl_histogram {
32public:
33 gsl_histogram() : n_(0), range_(), bin_(0) {}
34
38 void set_ranges_uniform(double xmin, double xmax) {
39 if (xmin >= xmax) {
40 throw std::invalid_argument("gsl_histogram: xmin must be < xmax");
41 }
42 range_.resize(n_ + 1);
43 double bin_width = (xmax - xmin) / n_;
44 for (size_t i = 0; i <= n_; ++i) {
45 range_[i] = xmin + i * bin_width;
46 }
47 }
48
52 void set_ranges(const double* range, size_t n) {
53 if (n != n_ + 1) {
54 throw std::invalid_argument("gsl_histogram: range size mismatch");
55 }
56 range_.assign(range, range + n);
57 }
58
61 void increment(double x) {
62 size_t bin = find_bin(x);
63 if (bin < bin_.size()) {
64 bin_[bin]++;
65 }
66 }
67
71 double get(size_t i) const {
72 if (i >= bin_.size()) {
73 throw std::out_of_range("gsl_histogram: bin index out of range");
74 }
75 return bin_[i];
76 }
77
80 size_t n() const { return n_; }
83 const std::vector<double>& range() const { return range_; }
86 const std::vector<double>& bin() const { return bin_; }
87
88 // Make members accessible for GSL compatibility functions
89 size_t n_;
90 std::vector<double> range_;
91 std::vector<double> bin_;
92
93private:
94 size_t find_bin(double x) const {
95 if (range_.empty()) return bin_.size();
96
97 // Handle out-of-range values
98 if (x < range_[0]) return bin_.size();
99 if (x >= range_.back()) return bin_.size();
100
101 // Binary search for the bin
102 auto it = std::upper_bound(range_.begin(), range_.end(), x);
103 return std::distance(range_.begin(), it) - 1;
104 }
105};

182// 2D Histogram
184class gsl_histogram2d {
185public:
186 gsl_histogram2d() : nx_(0), ny_(0), xrange_(), yrange_(), bin_() {}
187
193 void set_ranges_uniform(double xmin, double xmax, double ymin, double ymax) {
194 if (xmin >= xmax || ymin >= ymax) {
195 throw std::invalid_argument("gsl_histogram2d: invalid range");
196 }
197 xrange_.resize(nx_ + 1);
198 yrange_.resize(ny_ + 1);
199 double xbin_width = (xmax - xmin) / nx_;
200 double ybin_width = (ymax - ymin) / ny_;
201 for (size_t i = 0; i <= nx_; ++i) {
202 xrange_[i] = xmin + i * xbin_width;
203 }
204 for (size_t i = 0; i <= ny_; ++i) {
205 yrange_[i] = ymin + i * ybin_width;
206 }
207 }
208
213 void accumulate(double x, double y, double weight) {
214 size_t i = find_x_bin(x);
215 size_t j = find_y_bin(y);
216 if (i < nx_ && j < ny_) {
217 bin_[i * ny_ + j] += weight;
218 }
219 }
220
224 void increment(double x, double y) {
225 accumulate(x, y, 1.0);
226 }
227
233 void set_ranges(const double* xrange, size_t nx, const double* yrange, size_t ny) {
234 if (nx != nx_ + 1 || ny != ny_ + 1) {
235 throw std::invalid_argument("gsl_histogram2d: range size mismatch");
236 }
237 xrange_.assign(xrange, xrange + nx);
238 yrange_.assign(yrange, yrange + ny);
239 }
240
245 double get(size_t i, size_t j) const {
246 if (i >= nx_ || j >= ny_) {
247 throw std::out_of_range("gsl_histogram2d: bin index out of range");
248 }
249 return bin_[i * ny_ + j];
250 }
251
254 size_t nx() const { return nx_; }
257 size_t ny() const { return ny_; }
260 const std::vector<double>& xrange() const { return xrange_; }
263 const std::vector<double>& yrange() const { return yrange_; }
266 const std::vector<double>& bin() const { return bin_; }
267
268 // Make members accessible for GSL compatibility functions
269 size_t nx_, ny_;
270 std::vector<double> xrange_;
271 std::vector<double> yrange_;
272 std::vector<double> bin_;
273
274private:
275 size_t find_x_bin(double x) const {
276 if (xrange_.empty() || x < xrange_[0] || x >= xrange_.back()) {
277 return nx_;
278 }
279 auto it = std::upper_bound(xrange_.begin(), xrange_.end(), x);
280 return std::distance(xrange_.begin(), it) - 1;
281 }
282
283 size_t find_y_bin(double y) const {
284 if (yrange_.empty() || y < yrange_[0] || y >= yrange_.back()) {
285 return ny_;
286 }
287 auto it = std::upper_bound(yrange_.begin(), yrange_.end(), y);
288 return std::distance(yrange_.begin(), it) - 1;
289 }
290};

292// 2D Histogram PDF for sampling
294class gsl_histogram2d_pdf {
295public:
296 gsl_histogram2d_pdf() : nx_(0), ny_(0), xrange_(), yrange_(), pdf_(), cumsum_() {}
297
300 void init(const gsl_histogram2d* h) {
301 nx_ = h->nx();
302 ny_ = h->ny();
303 xrange_ = h->xrange();
304 yrange_ = h->yrange();
305
306 const auto& bin = h->bin();
307 double total = 0.0;
308 for (double val : bin) {
309 total += val;
310 }
311
312 pdf_.resize(bin.size());
313 if (total > 0) {
314 for (size_t i = 0; i < bin.size(); ++i) {
315 pdf_[i] = bin[i] / total;
316 }
317 }
318
319 // Compute cumulative distribution for sampling
320 cumsum_.resize(pdf_.size());
321 double cum = 0.0;
322 for (size_t i = 0; i < pdf_.size(); ++i) {
323 cum += pdf_[i];
324 cumsum_[i] = cum;
325 }
326 }
327
333 void sample(double u, double v, double* x, double* y) {
334 // Find bin using u (for x) and v (for y within that x slice)
335 size_t bin_idx = 0;
336 double target = u * cumsum_.back();
337 for (size_t i = 0; i < cumsum_.size(); ++i) {
338 if (cumsum_[i] >= target) {
339 bin_idx = i;
340 break;
341 }
342 }
343
344 size_t i = bin_idx / ny_;
345 size_t j = bin_idx % ny_;
346
347 if (i < nx_ && j < ny_) {
348 double xmin = xrange_[i];
349 double xmax = xrange_[i + 1];
350 double ymin = yrange_[j];
351 double ymax = yrange_[j + 1];
352 *x = xmin + (xmax - xmin) * v;
353 *y = ymin + (ymax - ymin) * u;
354 } else {
355 *x = 0.0;
356 *y = 0.0;
357 }
358 }
359
360private:
361 size_t nx_, ny_;
362 std::vector<double> xrange_;
363 std::vector<double> yrange_;
364 std::vector<double> pdf_;
365 std::vector<double> cumsum_;
366};
