45void IndexMap::print(std::ostream& out) const {
46 if (mapRange2Element_m.empty()) return;
47
48 out << "* Size of map " << mapRange2Element_m.size() << " sections " << std::endl;
49 out << std::fixed << std::setprecision(6);
50 auto mapIti = mapRange2Element_m.begin();
51 auto mapItf = mapRange2Element_m.end();
52
53 double totalLength = (*mapRange2Element_m.rbegin()).first.end;
54 unsigned int numDigits = std::floor(std::max(0.0, log(totalLength) / log(10.0))) + 1;
55
56 for (; mapIti != mapItf; mapIti++) {
57 const key_t key = (*mapIti).first;
58 const value_t val = (*mapIti).second;
59 out << "* Key: (" << std::setw(numDigits + 7) << std::right << key.begin << " - "
60 << std::setw(numDigits + 7) << std::right << key.end
61 << ") number of overlapping elements " << val.size() << "\n";
62
63 for (auto element : val) {
64 out << "* " << std::setw(25 + 2 * numDigits) << " " << element->getName() << "\n";
65 }
66 }
67}

69IndexMap::value_t IndexMap::query(key_t::first_type s, key_t::second_type ds) {
70 const double lowerLimit = s - ds; //(ds < s? s - ds: 0);
71 const double upperLimit = std::min(totalPathLength_m, s + ds);
72 value_t elementSet;
73
74 map_t::reverse_iterator rit = mapRange2Element_m.rbegin();
75 if (rit != mapRange2Element_m.rend() && lowerLimit > (*rit).first.end) {
76 throw OutOfBounds("IndexMap::query", "out of bounds");
77 }
78
79 map_t::iterator it = mapRange2Element_m.begin();
80 const map_t::iterator end = mapRange2Element_m.end();
81
82 for (; it != end; ++it) {
83 const double low = (*it).first.begin;
84 const double high = (*it).first.end;
85
86 if (lowerLimit < high && upperLimit >= low) break;
87 }
88
89 if (it == end) return elementSet;
90
91 map_t::iterator last = std::next(it);
92 for (; last != end; ++last) {
93 const double low = (*last).first.begin;
94
95 if (upperLimit < low) break;
96 }
97
98 for (; it != last; ++it) {
99 const value_t& a = (*it).second;
100 elementSet.insert(a.cbegin(), a.cend());
101 }
102
103 return elementSet;
104}

106void IndexMap::add(key_t::first_type initialS, key_t::second_type finalS, const value_t& val) {
107 if (initialS > finalS) {
108 std::swap(initialS, finalS);
109 }
110 key_t key{initialS, finalS * oneMinusEpsilon_m};
111
112 mapRange2Element_m.insert(std::pair<key_t, value_t>(key, val));
113 totalPathLength_m = (*mapRange2Element_m.rbegin()).first.end;
114
115 value_t::iterator setIt = val.begin();
116 const value_t::iterator setEnd = val.end();
117
118 for (; setIt != setEnd; ++setIt) {
119 if (mapElement2Range_m.find(*setIt) == mapElement2Range_m.end()) {
120 mapElement2Range_m.insert(std::make_pair(*setIt, key));
121 } else {
122 auto itpair = mapElement2Range_m.equal_range(*setIt);
123
124 bool extendedExisting = false;
125 for (auto it = itpair.first; it != itpair.second; ++it) {
126 key_t& currentRange = it->second;
127
128 if (almostEqual(key.begin, currentRange.end / oneMinusEpsilon_m)) {
129 currentRange.end = key.end;
130 extendedExisting = true;
131 break;
132 }
133 }
134 if (!extendedExisting) {
135 mapElement2Range_m.insert(std::make_pair(*setIt, key));
136 }
137 }
138 }

141void IndexMap::tidyUp(double zstop) {
142 map_t::reverse_iterator rit = mapRange2Element_m.rbegin();
143
144 if (rit != mapRange2Element_m.rend() && (*rit).second.empty() && zstop > (*rit).first.begin) {
145 key_t key{(*rit).first.begin, zstop};
146 value_t val;
147
148 mapRange2Element_m.erase(std::next(rit).base());
149 mapRange2Element_m.insert(std::pair<key_t, value_t>(key, val));

167void IndexMap::saveSDDS(double initialPathLength) const {
168 if (mapRange2Element_m.empty()) return;
169
170 std::vector<std::tuple<double, std::vector<double>, std::string> > sectors;
171
172 // add for each sector four rows:
173 // (s_i, 0)
174 // (s_i, 1)
175 // (s_f, 1)
176 // (s_f, 0)
177 // to the file, where
178 // s_i is the start of the range and
179 // s_f is the end of the range.
180 auto mapIti = mapRange2Element_m.begin();
181 auto mapItf = mapRange2Element_m.end();
182 for (; mapIti != mapItf; mapIti++) {
183 const auto& sectorElements = (*mapIti).second;
184 if (sectorElements.empty()) continue;
185
186 const auto& sectorRange = (*mapIti).first;
187
188 double sectorBegin = sectorRange.begin;
189 double sectorEnd = sectorRange.end;
190
191 std::vector<std::tuple<double, std::vector<double>, std::string> > currentSector(4);
192 std::get<0>(currentSector[0]) = sectorBegin;
193 std::get<0>(currentSector[1]) = sectorBegin;
194 std::get<0>(currentSector[2]) = sectorEnd;
195 std::get<0>(currentSector[3]) = sectorEnd;
196
197 for (unsigned short i = 0; i < 4; ++i) {
198 auto& flags = std::get<1>(currentSector[i]);
199 flags.resize(SIZE, 0);
200 }
201
202 for (auto element : sectorElements) {
203 auto elementPassages = mapElement2Range_m.equal_range(element);
204 auto passage = elementPassages.first;
205 auto end = elementPassages.second;
206 for (; passage != end; ++passage) {
207 const auto& elementRange = (*passage).second;
208 double elementBegin = elementRange.begin;
209 double elementEnd = elementRange.end;
210
211 if (elementBegin <= sectorBegin && elementEnd >= sectorEnd) {
212 break;
213 }
214 }
215
216 const auto& elementRange = (*passage).second;
217 if (elementRange.begin < sectorBegin) {
218 ::insertFlags(std::get<1>(currentSector[0]), element);
219 std::get<2>(currentSector[0]) += element->getName() + ", ";
220 }
221
222 ::insertFlags(std::get<1>(currentSector[1]), element);
223 std::get<2>(currentSector[1]) += element->getName() + ", ";
224
225 ::insertFlags(std::get<1>(currentSector[2]), element);
226 std::get<2>(currentSector[2]) += element->getName() + ", ";
227
228 if (elementRange.end > sectorEnd) {
229 ::insertFlags(std::get<1>(currentSector[3]), element);
230 std::get<2>(currentSector[3]) += element->getName() + ", ";
231 }
232 }
233
234 for (unsigned short i = 0; i < 4; ++i) {
235 sectors.push_back(currentSector[i]);
236 }
237 }
238
239 // make the entries of the rf cavities a zigzag line
240 const unsigned int numEntries = sectors.size();
241 auto it = mapElement2Range_m.begin();
242 auto end = mapElement2Range_m.end();
243 for (; it != end; ++it) {
244 auto element = (*it).first;
245 auto name = element->getName();
246 auto type = element->getType();
247 if (type != ElementType::RFCAVITY && type != ElementType::TRAVELINGWAVE) {
248 continue;
249 }
250
251 auto range = (*it).second;
252
253 unsigned int i = 0;
254 for (; i < numEntries; ++i) {
255 if (std::get<0>(sectors[i]) >= range.begin) {
256 break;
257 }
258 }
259
260 if (i == numEntries) continue;
261
262 unsigned int j = ++i;
263 while (std::get<0>(sectors[j]) < range.end) {
264 ++j;
265 }
266
267 double length = range.end - range.begin;
268 for (; i <= j; ++i) {
269 double pos = std::get<0>(sectors[i]);
270 auto& items = std::get<1>(sectors[i]);
271
272 items[RFCAVITY] = 1.0 - 2 * (pos - range.begin) / length;
273 }
274 }
275
276 // add row if range of first sector starts after initialPathLength
277 if (!sectors.empty() && std::get<0>(sectors[0]) > initialPathLength) {
278 auto tmp = sectors;
279 sectors = std::vector<std::tuple<double, std::vector<double>, std::string> >(1);
280 std::get<0>(sectors[0]) = initialPathLength;
281 std::get<1>(sectors[0]).resize(SIZE, 0.0);
282
283 sectors.insert(sectors.end(), tmp.begin(), tmp.end());
284 }
285
286 std::string fileName = Util::combineFilePath(
287 {OpalData::getInstance()->getAuxiliaryOutputDirectory(),
288 OpalData::getInstance()->getInputBasename() + "_ElementPositions.sdds"});
289 ElementPositionWriter writer(fileName);
290
291 for (auto sector : sectors) {
292 std::string names = std::get<2>(sector);
293 if (!names.empty()) {
294 names = names.substr(0, names.length() - 2);
295 }
296 names = "\"" + names + "\"";
297 writer.addRow(std::get<0>(sector), std::get<1>(sector), names);
298 }
299}

302 void insertFlags(std::vector<double>& flags, std::shared_ptr<Component> element) {
303 switch (element->getType()) {
304 case ElementType::MULTIPOLE: {
305 const Multipole* mult = static_cast<const Multipole*>(element.get());
306 switch (mult->getMaxNormalComponentIndex()) {
307 case 1:
308 flags[DIPOLE] = (mult->isFocusing(0) ? 1 : -1);
309 break;
310 case 2:
311 flags[QUADRUPOLE] = (mult->isFocusing(1) ? 1 : -1);
312 break;
313 case 3:
314 flags[SEXTUPOLE] = (mult->isFocusing(2) ? 1 : -1);
315 break;
316 case 4:
317 flags[OCTUPOLE] = (mult->isFocusing(3) ? 1 : -1);
318 break;
319 case 5:
320 flags[DECAPOLE] = (mult->isFocusing(4) ? 1 : -1);
321 break;
322 default:
323 flags[MULTIPOLE] = 1;
324 }
325 } break;
326 case ElementType::RFCAVITY:
327 case ElementType::TRAVELINGWAVE:
328 flags[RFCAVITY] = 1;
329 break;
330 case ElementType::SOLENOID:
331 flags[SOLENOID] = 1;
332 break;
333 case ElementType::CONSTANTEFIELDCAVITY:
334 flags[OTHER] = 1;
335 break;
336 default:
337 flags[OTHER] = 1;
338 break;
339 }

343IndexMap::key_t IndexMap::getRange(
344 const IndexMap::value_t::value_type& element, double position) const {
345 double minDistance = std::numeric_limits<double>::max();
346 key_t range{0.0, 0.0};
347 const std::pair<invertedMap_t::const_iterator, invertedMap_t::const_iterator> its =
348 mapElement2Range_m.equal_range(element);
349 if (std::distance(its.first, its.second) == 0)
350 throw OpalException(
351 "IndexMap::getRange()", "Element \"" + element->getName() + "\" not registered");
352
353 for (invertedMap_t::const_iterator it = its.first; it != its.second; ++it) {
354 double distance = std::min(
355 std::abs((*it).second.begin - position), std::abs((*it).second.end - position));
356 if (distance < minDistance) {
357 minDistance = distance;
358 range = (*it).second;
359 }
360 }
361
362 return range;
363}

365IndexMap::value_t IndexMap::getTouchingElements(const IndexMap::key_t& range) const {
366 map_t::const_iterator it = mapRange2Element_m.begin();
367 const map_t::const_iterator end = mapRange2Element_m.end();
368 value_t touchingElements;
369
370 for (; it != end; ++it) {
371 if (almostEqual(it->first.begin, range.begin) || almostEqual(it->first.end, range.end))
372 touchingElements.insert((it->second).begin(), (it->second).end());
373 }
374
375 return touchingElements;
