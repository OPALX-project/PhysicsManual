88template <typename ViewType>
89void computeStdDev(ViewType &view,
90 size_t nlocal,
91 size_t total_nparticles,
92 const double meanR[3],
93 double stddevR[3])
94{
95 double sumR2[3] = {0.0, 0.0, 0.0};
96
97 // Compute sum of squares locally
98 Kokkos::parallel_reduce(nlocal,
99 KOKKOS_LAMBDA(const int k, double& s0, double& s1, double& s2) {
100 s0 += view(k)[0] * view(k)[0];
101 s1 += view(k)[1] * view(k)[1];
102 s2 += view(k)[2] * view(k)[2];
103 },
104 Kokkos::Sum<double>(sumR2[0]),
105 Kokkos::Sum<double>(sumR2[1]),
106 Kokkos::Sum<double>(sumR2[2])
107 );
108 Kokkos::fence();
109
110 // MPI reduce to get global sums
111 double global_sumR2[3] = {0.0, 0.0, 0.0};
112 MPI_Allreduce(sumR2, global_sumR2, 3, MPI_DOUBLE, MPI_SUM,
113 ippl::Comm->getCommunicator());
114 ippl::Comm->barrier();
115
116 // Compute variance and stddev
117 for (int i = 0; i < 3; i++) {
118 double mean_square = global_sumR2[i] / static_cast<double>(total_nparticles);
119 double variance = mean_square - meanR[i] * meanR[i];
120 stddevR[i] = std::sqrt(variance);
121 }
122}

160template <typename ViewType>
161void computeMaxAbsR(ViewType &view,
162 size_t nlocal,
163 double global_maxAbsR[3],
164 double& global_maxRadius)
165{
166 // Local maxima initialized
167 double local_maxAbsR[3] = {0.0, 0.0, 0.0};
168 double local_maxRadius = 0.0;
169
170 // Compute local maxima using Kokkos
171 Kokkos::parallel_reduce(
172 nlocal,
173 KOKKOS_LAMBDA(const int k,
174 double& max0, double& max1, double& max2, double& maxr) {
175 double x = view(k)[0];
176 double y = view(k)[1];
177 double z = view(k)[2];
178
179 double ax = Kokkos::fabs(x);
180 double ay = Kokkos::fabs(y);
181 double az = Kokkos::fabs(z);
182 double r = Kokkos::sqrt(x*x + y*y + z*z);
183
184 if (ax > max0) max0 = ax;
185 if (ay > max1) max1 = ay;
186 if (az > max2) max2 = az;
187 if (r > maxr) maxr = r;
188 },
189 Kokkos::Max<double>(local_maxAbsR[0]),
190 Kokkos::Max<double>(local_maxAbsR[1]),
191 Kokkos::Max<double>(local_maxAbsR[2]),
192 Kokkos::Max<double>(local_maxRadius)
193 );
194 Kokkos::fence();
195
196
197 // MPI reduction to get global maxima
198 MPI_Allreduce(local_maxAbsR, global_maxAbsR, 3, MPI_DOUBLE, MPI_MAX,
199 ippl::Comm->getCommunicator());
200 MPI_Allreduce(&local_maxRadius, &global_maxRadius, 1, MPI_DOUBLE, MPI_MAX,
201 ippl::Comm->getCommunicator());
202
203 ippl::Comm->barrier();
204}

282void computeMeanRAndP(const std::shared_ptr<ParticleContainer<double,3>>& pc,
283 size_t total_nparticles,
284 double meanSample[6])
285{
286 auto Rview = pc->R.getView();
287 auto Pview = pc->P.getView();
288
289 double loc_sum[6] = {0.0};
290
291 // Kokkos parallel_reduce for both R and P
292
293 Kokkos::parallel_reduce(
294 pc->getLocalNum(),
295 KOKKOS_LAMBDA(const int k,
296 double& s0, double& s1, double& s2,
297 double& s3, double& s4, double& s5) {
298 s0 += Rview(k)[0];
299 s1 += Pview(k)[0];
300 s2 += Rview(k)[1];
301 s3 += Pview(k)[1];
302 s4 += Rview(k)[2];
303 s5 += Pview(k)[2];
304 },
305 Kokkos::Sum<double>(loc_sum[0]),
306 Kokkos::Sum<double>(loc_sum[1]),
307 Kokkos::Sum<double>(loc_sum[2]),
308 Kokkos::Sum<double>(loc_sum[3]),
309 Kokkos::Sum<double>(loc_sum[4]),
310 Kokkos::Sum<double>(loc_sum[5])
311 );
312 Kokkos::fence();
313
314 // MPI reduction to global sum
315 double global_sum[6] = {0.0};
316 MPI_Allreduce(loc_sum, global_sum, 6, MPI_DOUBLE, MPI_SUM,
317 ippl::Comm->getCommunicator());
318 ippl::Comm->barrier();
319
320 // Normalize to get mean
321 for (int i = 0; i < 6; i++)
322 meanSample[i] = global_sum[i] / static_cast<double>(total_nparticles);
323}

325void computeCovariance6x6(const std::shared_ptr<ParticleContainer<double,3>>& pc,
326 const double meanSample[6],
327 size_t total_nparticles,
328 double moment[6][6])
329{
330 auto Rview = pc->R.getView();
331 auto Pview = pc->P.getView();
332
333 Kokkos::Array<double, 6> mean = {
334 meanSample[0],
335 meanSample[1],
336 meanSample[2],
337 meanSample[3],
338 meanSample[4],
339 meanSample[5]
340 };
341
342 double loc_moment[6][6] = {{0.0}};
343
344 for (unsigned i = 0; i < 6; ++i) {
345 const unsigned row = i;
346 Kokkos::parallel_reduce(
347 pc->getLocalNum(),
348 KOKKOS_LAMBDA(const int k,
349 double& s0, double& s1, double& s2,
350 double& s3, double& s4, double& s5) {
351 double part[6] = {
352 Rview(k)[0] - mean[0],
353 Pview(k)[0] - mean[1],
354 Rview(k)[1] - mean[2],
355 Pview(k)[1] - mean[3],
356 Rview(k)[2] - mean[4],
357 Pview(k)[2] - mean[5]
358 };
359 s0 += part[row] * part[0];
360 s1 += part[row] * part[1];
361 s2 += part[row] * part[2];
362 s3 += part[row] * part[3];
363 s4 += part[row] * part[4];
364 s5 += part[row] * part[5];
365 },
366 Kokkos::Sum<double>(loc_moment[row][0]),
367 Kokkos::Sum<double>(loc_moment[row][1]),
368 Kokkos::Sum<double>(loc_moment[row][2]),
369 Kokkos::Sum<double>(loc_moment[row][3]),
370 Kokkos::Sum<double>(loc_moment[row][4]),
371 Kokkos::Sum<double>(loc_moment[row][5])
372 );
373 Kokkos::fence();
374 }
375
376 // MPI reduction across all ranks
377 MPI_Allreduce(loc_moment, moment, 6 * 6, MPI_DOUBLE, MPI_SUM,
378 ippl::Comm->getCommunicator());
379 ippl::Comm->barrier();
380
381 // Normalize to get average covariance/moments
382 for (unsigned i = 0; i < 6; ++i) {
383 for (unsigned j = 0; j < 6; ++j) {
384 moment[i][j] /= static_cast<double>(total_nparticles);
385 }
386 }
387}

407 covExpected[i][j] = 0.0;
