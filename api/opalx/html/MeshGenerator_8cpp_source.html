55 major = horizontalRadius;

69 case ApertureType::RECTANGULAR:
70 case ApertureType::CONIC_RECTANGULAR:
71 case ApertureType::ELLIPTICAL:
72 case ApertureType::CONIC_ELLIPTICAL:
73 minor = apert.second[0];
74 major = apert.second[1];
75 return true;
76 default:
77 return false;
78 }
