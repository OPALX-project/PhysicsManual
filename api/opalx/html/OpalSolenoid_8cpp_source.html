58void OpalSolenoid::update() {
59 OpalElement::update();
60
61 SolenoidRep *sol =
62 dynamic_cast<SolenoidRep *>(getElement());
63 double length = Attributes::getReal(itsAttr[LENGTH]);
64 double Bz = Attributes::getReal(itsAttr[KS]) * OpalData::getInstance()->getP0() / Physics::c;
65 bool fast = Attributes::getBool(itsAttr[FAST]);
66
67 sol->setElementLength(length);
68 sol->setFieldMapFN(Attributes::getString(itsAttr[FMAPFN]));
69 sol->setFast(fast);
70 sol->setBz(Bz);
71 sol->setKS(Attributes::getReal(itsAttr[KS]));
72 sol->setDKS(Attributes::getReal(itsAttr[DKS]));
73
74 // Transmit "unknown" attributes.
75 OpalElement::updateUnknown(sol);
76}
