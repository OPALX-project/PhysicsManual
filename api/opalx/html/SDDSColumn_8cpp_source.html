25SDDSColumn::SDDSColumn(const std::string& name,
26 const std::string& type,
27 const std::string& unit,
28 const std::string& desc,
29 std::ios_base::fmtflags flags,
30 unsigned short prec):
31 name_m(name),
32 description_m(std::make_tuple(type, unit, desc)),
33 writeFlags_m(flags),
34 writePrecision_m(prec),
35 set_m(false)
36{
37 std::list<std::ios_base::fmtflags> numericalBase({std::ios_base::dec,
38 std::ios_base::hex,
39 std::ios_base::oct});
40 std::list<std::ios_base::fmtflags> floatFormat({std::ios_base::fixed,
41 std::ios_base::scientific});
42 std::list<std::ios_base::fmtflags> adjustmentFlags({std::ios_base::internal,
43 std::ios_base::left,
44 std::ios_base::right});
45
46 // This code ensures that for each group of flags only one flag is given
47 for (std::ios_base::fmtflags flag: numericalBase) {
48 if (writeFlags_m & flag) {
49 writeFlags_m = (flag | (writeFlags_m & ~std::ios_base::basefield));
50 break;
51 }
52 }
53 for (std::ios_base::fmtflags flag: floatFormat) {
54 if (writeFlags_m & flag) {
55 writeFlags_m = (flag | (writeFlags_m & ~std::ios_base::floatfield));
56 break;
57 }
58 }
59 for (std::ios_base::fmtflags flag: adjustmentFlags) {
60 if (writeFlags_m & flag) {
61 writeFlags_m = (flag | (writeFlags_m & ~std::ios_base::adjustfield));
62 break;
63 }
64 }

66}

84void SDDSColumn::writeValue(std::ostream& os) const {
85 if (!set_m) {
86 throw OpalException("SDDSColumn::writeValue",
87 "value for column '" + name_m + "' isn't set");
88 }
89
90 os.flags(writeFlags_m);
91 os.precision(writePrecision_m);
92 std::visit([&os](const auto& val) { os << val; }, value_m);
93 os << std::setw(10) << "\t";
94 set_m = false;
95}
